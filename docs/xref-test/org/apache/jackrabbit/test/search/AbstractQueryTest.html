18  import org.apache.jackrabbit.test.AbstractTest;

20  import javax.jcr.query.QueryResult;
21  import javax.jcr.*;
22  
23  /***
24   * Abstract base class for query test cases.
25   *
26   * @author Marcel Reutegger
27   * @version $Revision:  $, $Date:  $
28   */
29  public class AbstractQueryTest extends AbstractTest {
30  
31      /***
32       * Checks if the <code>result</code> contains a number of <code>hits</code>.
33       * @param result the <code>QueryResult</code>.
34       * @param hits the number of expected hits.
35       * @throws RepositoryException if an error occurs while iterating over
36       *  the result nodes.
37       */
38      protected void checkResult(QueryResult result, int hits)
39  	    throws RepositoryException {
40  	int count = 0;
41  	log.info("Nodes:");
42  	for (NodeIterator nodes = result.getNodes(); nodes.hasNext(); count++ ) {
43  	    Node n = nodes.nextNode();
44  	    log.info(" " + n.getPath());
45  	}
46  	if (count == 0) {
47  	    log.info(" NONE");
48  	}
49  	assertEquals("Wrong hit count.", hits, count);
50      }
51  
52      /***
53       * Checks if the <code>result</code> contains a number of <code>hits</code>
54       * and <code>properties</code>.
55       * @param result the <code>QueryResult</code>.
56       * @param hits the number of expected hits.
57       * @param properties the number of expected properties.
58       * @throws RepositoryException if an error occurs while iterating over
59       *  the result nodes.
60       */
61      protected void checkResult(QueryResult result, int hits, int properties)
62  	    throws RepositoryException {
63  	checkResult(result, hits);
64  	// now check property count
65  	int count = 0;
66  	log.info("Properties:");
67  	for (PropertyIterator it = result.getProperties(); it.hasNext(); count++ ) {
68  	    StringBuffer msg = new StringBuffer();
69  	    Property p = it.nextProperty();
70  	    msg.append("  ").append(p.getName()).append(": ");
71  	    Value[] values = null;
72  	    if (p.getDefinition().isMultiple()) {
73  		values = p.getValues();
74  	    } else {
75  		values = new Value[] { p.getValue() };
76  	    }
77  	    String sep = "";
78  	    for (int i = 0; i < values.length; i++) {
79  		msg.append(sep);
80  		msg.append(values[i].getString());
81  		sep = " | ";
82  	    }
83  	    log.info(msg);
84  	}
85  	if (count == 0) {
86  	    log.info("  NONE");
87  	}
88  	assertEquals("Wrong property count.", properties, count);
89      }
