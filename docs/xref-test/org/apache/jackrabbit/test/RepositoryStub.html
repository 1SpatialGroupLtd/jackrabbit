66      /***
67       * Implementations of this class must overwrite this constructor.
68       *
69       * @param env the environment variables. This parameter must not be null.
70       */
71      protected RepositoryStub(Properties env) {
72          if (env == null) throw new IllegalArgumentException("Parameter 'env' must not be null!");
73          environment = env;
74  	superuser = new SimpleCredentials(env.getProperty(PROP_SUPERUSER_NAME, ""),
75  		env.getProperty(PROP_SUPERUSER_PWD, "").toCharArray());
76  	readonly = new SimpleCredentials(env.getProperty(PROP_READONLY_NAME, ""),
77  		env.getProperty(PROP_READONLY_PWD, "").toCharArray());

82      /***
83       * Creates and/or returns the configured <code>RepositryStub</code>
84       * implementation.
85       * <p>
86       * The property file is located in the following sequence:
87       * <ol>
88       * <li>If the system property <code>-Djavax.jcr.tck.properties</code> is
89       * set, then the accroding file is used as configuration.</li>
90       * <li>If the system property <code>-Djavax.jcr.tck.properties</code> is
91       * not set, then the TCK tries to load the file <code>repositoryStubImpl.properties</code>
92       * as a resource from the ClassLoader of this <code>RepositryStub</code> class.</li>
93       * <li>If none of the above is found, a {@link RepositoryStubException} is thrown.
94       * </ol>
95       * @return a <code>RepositoryStub</code> implementation.
96       * @throws RepositoryStubException
97       */
98      public static synchronized RepositoryStub getInstance() throws RepositoryStubException {
99          if (instance == null) {
100             Properties props = null;
101             String implProp = System.getProperty(STUB_IMPL_SYS_PROPS);
102 	    if (implProp != null) {
103 		File implPropFile = new File(implProp);
104 		if (implPropFile.exists()) {
105 		    props = new Properties();
106 		    try {
107 			props.load(new FileInputStream(implPropFile));
108 		    } catch (IOException e) {
109 			throw new RepositoryStubException("Unable to load config file: "
110 				+ implProp + " " + e.toString());
111 		    }
112 		} else {
113 		    throw new RepositoryStubException("File does not exist: " + implProp);
114 		}
115 	    }
116 
117 	    if (props == null) {
118 		InputStream is = RepositoryStub.class.getClassLoader().getResourceAsStream(STUB_IMPL_PROPS);
119 		if (is == null) {
120 		    throw new RepositoryStubException(STUB_IMPL_PROPS + " not found in classpath!");
121 		}
122 		try {
123 		    props = new Properties();
124 		    props.load(is);
125 		} catch (IOException e) {
126 		    throw new RepositoryStubException("Exception reading "
127 			    + STUB_IMPL_PROPS + ": " + e.toString());
128 		}
129 	    }
130 
131             try {
132                 String className = props.getProperty(PROP_STUB_IMPL_CLASS);
133                 if (className == null || className.length() == 0) {
134                     throw new RepositoryStubException("Property " + PROP_STUB_IMPL_CLASS + " not defined!");
135                 }
136                 Class stubClass = Class.forName(className);
137                 Constructor constr = stubClass.getConstructor(new Class[] {Properties.class});
138                 instance = (RepositoryStub)constr.newInstance(new Object[] {props});
139             } catch (ClassCastException e) {
140                 throw new RepositoryStubException(e.toString());
141             } catch (NoSuchMethodException e) {
142                 throw new RepositoryStubException(e.toString());
143             } catch (ClassNotFoundException e) {
144                 throw new RepositoryStubException(e.toString());
145             } catch (InstantiationException e) {
146                 throw new RepositoryStubException(e.toString());
147             } catch (IllegalAccessException e) {
148                 throw new RepositoryStubException(e.toString());
149             } catch (InvocationTargetException e) {
150                 throw new RepositoryStubException(e.toString());
151             }
152         }
153         return instance;
154     }
