48      /***
49       * Failsafe conversion of internal <code>Path</code> to JCR path for use in
50       * error messages etc.
51       *
52       * @param path path to convert
53       * @return JCR path
54       */
55      public String safeGetJCRPath(Path path) {
56  	try {
57  	    return path.toJCRPath(nsResolver);
58  	} catch (NoPrefixDeclaredException npde) {
59  	    log.error("failed to convert " + path.toString() + " to JCR path.");
60  	    // return string representation of internal path as a fallback
61  	    return path.toString();
62  	}
63      }

65      /***
66       * Failsafe translation of internal <code>ItemId</code> to JCR path for use in
67       * error messages etc.
68       *
69       * @param id path to convert
70       * @return JCR path
71       */
72      public String safeGetJCRPath(ItemId id) {
73  	try {
74  	    return safeGetJCRPath(getPath(id));
75  	} catch (RepositoryException re) {
76  	    log.error(id + ": failed to determine path to");
77  	    // return string representation if id as a fallback
78  	    return id.toString();
79  	}
80      }

83      /***
84       * @see HierarchyManager#listParents(ItemId)
85       */
86      public NodeId[] listParents(ItemId id) throws ItemNotFoundException, RepositoryException {
87  	ArrayList list = new ArrayList();
88  	try {
89  	    if (id.denotesNode()) {
90  		NodeState state = (NodeState) getItemState(id, false);
91  		Iterator iter = state.getParentUUIDs().iterator();
92  		while (iter.hasNext()) {
93  		    list.add(new NodeId((String) iter.next()));
94  		}
95  	    } else {
96  		PropertyState state = (PropertyState) getItemState(id, false);
97  		list.add(new NodeId(state.getParentUUID()));
98  	    }
99  	} catch (NoSuchItemStateException e) {
100 	    String msg = "failed to retrieve state of item " + id;
101 	    log.error(msg, e);
102 	    throw new ItemNotFoundException(msg, e);
103 	} catch (ItemStateException e) {
104 	    String msg = "failed to retrieve state of item " + id;
105 	    log.error(msg, e);
106 	    throw new RepositoryException(msg, e);
107 	}
108 	return (NodeId[]) list.toArray(new NodeId[list.size()]);
109     }

111     /***
112      * @see HierarchyManager#listChildren(NodeId)
113      */
114     public ItemId[] listChildren(NodeId id) throws ItemNotFoundException, RepositoryException {
115 	NodeState parentState;
116 	try {
117 	    parentState = (NodeState) getItemState(id, false);
118 	} catch (NoSuchItemStateException e) {
119 	    String msg = "failed to retrieve state of parent node " + id;
120 	    log.error(msg, e);
121 	    throw new ItemNotFoundException(msg, e);
122 	} catch (ItemStateException e) {
123 	    String msg = "failed to retrieve state of parent node " + id;
124 	    log.error(msg, e);
125 	    throw new RepositoryException(msg, e);
126 	}
127 	ArrayList list = new ArrayList();
128 	Iterator iter = parentState.getPropertyEntries().iterator();
129 	while (iter.hasNext()) {
130 	    // properties
131 	    NodeState.PropertyEntry pe = (NodeState.PropertyEntry) iter.next();
132 	    list.add(new PropertyId(id.getUUID(), pe.getName()));
133 	}
134 	iter = parentState.getChildNodeEntries().iterator();
135 	while (iter.hasNext()) {
136 	    // child nodes
137 	    NodeState.ChildNodeEntry cne = (NodeState.ChildNodeEntry) iter.next();
138 	    list.add(new NodeId(cne.getUUID()));
139 	}
140 	return (ItemId[]) list.toArray(new ItemId[list.size()]);
141     }

178     /***
179      * @see HierarchyManager#resolvePath(Path)
180      */
181     public synchronized ItemId resolvePath(Path path)
182 	    throws PathNotFoundException, RepositoryException {
183 	// shortcut
184 	if (path.denotesRoot()) {
185 	    return rootNodeId;
186 	}
187 
188 	if (!path.isCanonical()) {
189 	    String msg = "path is not canonical";
190 	    log.error(msg);
191 	    throw new RepositoryException(msg);
192 	}
193 
194 	NodeState parentState;
195 	try {
196 	    parentState = (NodeState) getItemState(rootNodeId, false);
197 	} catch (ItemStateException e) {
198 	    String msg = "failed to retrieve state of root node";
199 	    log.error(msg, e);
200 	    throw new RepositoryException(msg, e);
201 	}
202 
203 	Path.PathElement[] elems = path.getElements();
204 	for (int i = 1; i < elems.length; i++) {
205 	    Path.PathElement elem = elems[i];
206 	    QName name = elem.getName();
207 	    int index = elem.getIndex();
208 	    if (parentState.hasChildNodeEntry(name, index == 0 ? 1 : index)) {
209 		// child node
210 		NodeState.ChildNodeEntry nodeEntry = parentState.getChildNodeEntry(name, index == 0 ? 1 : index);
211 		if (i == elems.length - 1) {
212 		    // last element in the path
213 		    return new NodeId(nodeEntry.getUUID());
214 		}
215 		try {
216 		    parentState = (NodeState) getItemState(new NodeId(nodeEntry.getUUID()), false);
217 		} catch (ItemStateException e) {
218 		    String msg = "failed to retrieve state of intermediary node";
219 		    log.error(msg, e);
220 		    throw new RepositoryException(msg, e);
221 		}
222 		continue;
223 	    } else if (parentState.hasPropertyEntry(name)) {
224 		// property
225 		if (index > 1) {
226 		    // properties can't have same name siblings
227 		    throw new PathNotFoundException(safeGetJCRPath(path));
228 		}
229 		if (i == elems.length - 1) {
230 		    return new PropertyId(parentState.getUUID(), name);
231 		} else {
232 		    // property is not the last element in the path
233 		    throw new PathNotFoundException(safeGetJCRPath(path));
234 		}
235 	    } else {
236 		// no such item
237 		throw new PathNotFoundException(safeGetJCRPath(path));
238 	    }
239 	}
240 
241 	throw new PathNotFoundException(safeGetJCRPath(path));
242     }

244     /***
245      * @see HierarchyManager#getPath(ItemId)
246      */
247     public synchronized Path getPath(ItemId id) throws ItemNotFoundException, RepositoryException {
248 	try {
249 	    Path.PathBuilder builder = new Path.PathBuilder();
250 
251 	    ItemState state = getItemState(id, false);
252 	    String parentUUID = state.getParentUUID();
253 	    if (parentUUID == null) {
254 		// specified id denotes the root node
255 		builder.addRoot();
256 		return builder.getPath();
257 	    }
258 
259 	    NodeState parent = (NodeState) getItemState(new NodeId(parentUUID), false);
260 	    do {
261 		if (state.isNode()) {
262 		    NodeState nodeState = (NodeState) state;
263 		    String uuid = nodeState.getUUID();
264 		    List entries = parent.getChildNodeEntries(uuid);
265 		    if (entries.isEmpty()) {
266 			String msg = "failed to build path of " + id + ": " + parent.getUUID() + " has no child entry for " + uuid;
267 			log.error(msg);
268 			throw new RepositoryException(msg);
269 		    }
270 		    // if the parent has more than one child node entries pointing
271 		    // to the same child node, always use the first one
272 		    NodeState.ChildNodeEntry entry = (NodeState.ChildNodeEntry) entries.get(0);
273 		    QName name = entry.getName();
274 		    // add to path
275 		    builder.addFirst(name.getNamespaceURI(), name.getLocalName(), entry.getIndex());
276 		} else {
277 		    PropertyState propState = (PropertyState) state;
278 		    QName name = propState.getName();
279 		    // add to path
280 		    builder.addFirst(name.getNamespaceURI(), name.getLocalName());
281 		}
282 		parentUUID = parent.getParentUUID();
283 		if (parentUUID != null) {
284 		    state = parent;
285 		    parent = (NodeState) getItemState(new NodeId(parentUUID), false);
286 		} else {
287 		    parent = null;
288 		    state = null;
289 		}
290 	    } while (parent != null);
291 
292 	    // add root to path
293 	    builder.addRoot();
294 	    return builder.getPath();
295 	} catch (NoSuchItemStateException nsise) {
296 	    String msg = "failed to build path of " + id;
297 	    log.error(msg, nsise);
298 	    throw new ItemNotFoundException(msg, nsise);
299 	} catch (ItemStateException ise) {
300 	    String msg = "failed to build path of " + id;
301 	    log.error(msg, ise);
302 	    throw new RepositoryException(msg, ise);
303 	} catch (MalformedPathException mpe) {
304 	    String msg = "failed to build path of " + id;
305 	    log.error(msg, mpe);
306 	    throw new RepositoryException(msg, mpe);
307 	}
308     }

310     /***
311      * @see HierarchyManager#getName(ItemId)
312      */
313     public QName getName(ItemId itemId) throws ItemNotFoundException, RepositoryException {
314 	if (itemId.denotesNode()) {
315 	    NodeId nodeId = (NodeId) itemId;
316 	    NodeState parentState;
317 	    if (!provider.hasItemState(nodeId)) {
318 		String msg = "failed to resolve name of " + nodeId;
319 		log.error(msg);
320 		throw new ItemNotFoundException(nodeId.toString());
321 	    }
322 	    try {
323 		NodeState nodeState = (NodeState) getItemState(nodeId, false);
324 		String parentUUID = nodeState.getParentUUID();
325 		if (parentUUID == null) {
326 		    // this is the root or an orphaned node
327 		    // FIXME
328 		    return new QName(NamespaceRegistryImpl.NS_DEFAULT_URI, "");
329 		}
330 		parentState = (NodeState) getItemState(new NodeId(parentUUID), true);
331 	    } catch (ItemStateException ise) {
332 		String msg = "failed to resolve name of " + nodeId;
333 		log.error(msg, ise);
334 		throw new RepositoryException(msg, ise);
335 	    }
336 
337 	    List entries = parentState.getChildNodeEntries(nodeId.getUUID());
338 	    if (entries.size() == 0) {
339 		String msg = "failed to resolve name of " + nodeId;
340 		log.error(msg);
341 		throw new RepositoryException(msg);
342 	    }
343 	    NodeState.ChildNodeEntry entry = (NodeState.ChildNodeEntry) entries.get(0);
344 	    return entry.getName();
345 	} else {
346 	    PropertyId propId = (PropertyId) itemId;
347 	    return propId.getName();
348 	}
349     }

358     /***
359      * @see HierarchyManager#getAllPaths(ItemId, boolean)
360      */
361     public synchronized Path[] getAllPaths(ItemId id, boolean includeZombies)
362 	    throws ItemNotFoundException, RepositoryException {
363 	Path.PathBuilder builder = new Path.PathBuilder();
364 	ArrayList list = new ArrayList();
365 	list.add(builder);
366 
367 	NodeId nodeId = null;
368 	if (!id.denotesNode()) {
369 	    try {
370 		PropertyState propState = (PropertyState) getItemState(id, includeZombies);
371 		QName name = propState.getName();
372 		// add to path
373 		builder.addFirst(name.getNamespaceURI(), name.getLocalName());
374 		nodeId = new NodeId(propState.getParentUUID());
375 	    } catch (NoSuchItemStateException nsise) {
376 		String msg = "failed to build path of " + id;
377 		log.error(msg, nsise);
378 		throw new ItemNotFoundException(msg, nsise);
379 	    } catch (ItemStateException ise) {
380 		String msg = "failed to build path of " + id;
381 		log.error(msg, ise);
382 		throw new RepositoryException(msg, ise);
383 	    }
384 	} else {
385 	    nodeId = (NodeId) id;
386 	}
387 
388 	// recursively traverse parent nodes
389 	recursiveBuildPaths(nodeId, builder, list, includeZombies);
390 
391 	Path[] paths = new Path[list.size()];
392 	int i = 0;
393 	Iterator iter = list.iterator();
394 	while (iter.hasNext()) {
395 	    Path.PathBuilder pb = (Path.PathBuilder) iter.next();
396 	    try {
397 		paths[i++] = pb.getPath();
398 	    } catch (MalformedPathException mpe) {
399 		String msg = "failed to build all paths of " + id;
400 		log.error(msg, mpe);
401 		throw new RepositoryException(msg, mpe);
402 	    }
403 	}
404 	return paths;
405     }

416 	    throws NoSuchItemStateException, ItemStateException {
417 	if (!includeZombies) {
418 	    // get transient/persistent state
419 	    return provider.getItemState(id);

423 	    // try attic first
424 	    return provider.getItemStateInAttic(id);

431     /***
432      * @param nodeId
433      * @param builder
434      * @param builders
435      * @param includeZombies
436      * @throws ItemNotFoundException
437      * @throws RepositoryException
438      */
439     private void recursiveBuildPaths(NodeId nodeId, Path.PathBuilder builder,
440 				     List builders, boolean includeZombies)
441 	    throws ItemNotFoundException, RepositoryException {
442 	try {
443 	    NodeState nodeState = (NodeState) getItemState(nodeId, includeZombies);
444 
445 	    String uuid = nodeState.getUUID();
446 	    /***
447 	     * the parent uuid list contains an entry for every parent-child
448 	     * link to the specified node. this list may contain duplicate
449 	     * entries if the same parent node has more than one child link to
450 	     * the same target node. because the following code expects unique
451 	     * entries in the parent uuid list, we put them into a set.
452 	     */
453 	    HashSet parentUUIDs = new HashSet(nodeState.getParentUUIDs());
454 	    if (includeZombies) {
455 		parentUUIDs.addAll(nodeState.getRemovedParentUUIDs());
456 	    }
457 	    if (parentUUIDs.size() == 0) {
458 		// nodeId denotes the root node
459 		builder.addRoot();
460 		return;
461 	    }
462 
463 	    // if the specified node has n parents,
464 	    // we need to create n - 1 path builder clones
465 	    LinkedList queue = new LinkedList();
466 	    queue.add(builder);
467 	    int n = parentUUIDs.size() - 1;
468 	    while (n-- > 0) {
469 		Path.PathBuilder clone = (Path.PathBuilder) builder.clone();
470 		queue.addLast(clone);
471 		// add to list of path builders
472 		builders.add(clone);
473 	    }
474 
475 	    Iterator iter = parentUUIDs.iterator();
476 	    while (iter.hasNext()) {
477 		String parentUUID = (String) iter.next();
478 		NodeState parent = (NodeState) getItemState(new NodeId(parentUUID), includeZombies);
479 		ArrayList entries = new ArrayList(parent.getChildNodeEntries(uuid));
480 		if (includeZombies) {
481 		    Iterator riter = parent.getRemovedChildNodeEntries().iterator();
482 		    while (riter.hasNext()) {
483 			NodeState.ChildNodeEntry entry = (NodeState.ChildNodeEntry) riter.next();
484 			if (entry.getUUID().equals(uuid)) {
485 			    entries.add(entry);
486 			}
487 		    }
488 		}
489 		if (entries.isEmpty()) {
490 		    String msg = "failed to build path of " + nodeId + ": " + parent.getUUID() + " has no child entry for " + uuid;
491 		    log.error(msg);
492 		    throw new RepositoryException(msg);
493 		}
494 		n = entries.size() - 1;
495 		while (n-- > 0) {
496 		    // the same parent has multiple child references to
497 		    // this node, more path builder clones are needed
498 
499 		    // since we are consuming the queue of path builders
500 		    // starting at the tail, we can safely clone 'builder'
501 		    // because it will be consumed as the very last queue entry
502 		    Path.PathBuilder clone = (Path.PathBuilder) builder.clone();
503 		    queue.add(clone);
504 		    // add to list of path builders
505 		    builders.add(clone);
506 		}
507 		for (int i = 0; i < entries.size(); i++) {
508 		    NodeState.ChildNodeEntry entry = (NodeState.ChildNodeEntry) entries.get(i);
509 		    QName name = entry.getName();
510 
511 		    // get a path builder clone from the tail of the queue
512 		    Path.PathBuilder pb = (Path.PathBuilder) queue.removeLast();
513 		    // add entry to path
514 		    pb.addFirst(name.getNamespaceURI(), name.getLocalName(), entry.getIndex());
515 
516 		    // recurse
517 		    recursiveBuildPaths(new NodeId(parentUUID), pb, builders, includeZombies);
518 		}
519 	    }
520 	} catch (NoSuchItemStateException nsise) {
521 	    String msg = "failed to build path of " + nodeId;
522 	    log.error(msg, nsise);
523 	    throw new ItemNotFoundException(msg, nsise);
524 	} catch (ItemStateException ise) {
525 	    String msg = "failed to build path of " + nodeId;
526 	    log.error(msg, ise);
527 	    throw new RepositoryException(msg, ise);
528 	}
529     }
