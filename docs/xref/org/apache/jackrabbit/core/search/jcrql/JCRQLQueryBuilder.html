41      public static QueryRootNode createQuery(String statement)
42  	    throws InvalidQueryException {
43  	try {
44  	    JCRQLQueryBuilder builder = new JCRQLQueryBuilder(JCRQLParser.parse(statement));
45  	    return builder.getRootNode();
46  	} catch (ParseException e) {
47  	    throw new InvalidQueryException(e.getMessage());
48  	}
49      }
50  
51      QueryRootNode getRootNode() {
52  	if (root == null) {
53  	    root = (QueryRootNode)stmt.jjtAccept(this, null);
54  	}
55  	return root;
56      }

60      public Object visit(SimpleNode node, Object data) {
61  	// FIXME throw exception? unsupported

65      public Object visit(ASTStatement node, Object data) {
66  	QueryRootNode root = new QueryRootNode();
67  
68  	node.childrenAccept(this, root);
69  	// pass to select, from, location, where, textsearch clause
70  	return root;
71      }

73      public Object visit(ASTSelectClause node, Object data) {
74  	QueryRootNode root = (QueryRootNode)data;
75  	String[] properties = node.getProperties();
76  	for (int i = 0; i < properties.length; i++) {
77  	    root.addSelectProperty(properties[i]);
78  	}
79  	return data;
80      }
81  
82      public Object visit(ASTFromClause node, Object data) {
83  	QueryRootNode root = (QueryRootNode)data;
84  	// add node type query to parent
85  	node.childrenAccept(this, root.getConstraintNode());
86  
87  	return data;
88      }

90      public Object visit(ASTNodeType node, Object data) {
91          NAryQueryNode parent = (NAryQueryNode)data;
92  	parent.addOperand(new NodeTypeQueryNode(parent, node.getName()));
93  	return parent;
94      }

96      public Object visit(ASTLocationClause node, Object data) {
97  	QueryRootNode root = (QueryRootNode)data;
98  	root.setLocationNode(new PathQueryNode(root, node.getStringLocation(),
99  		node.getType()));
100 	return root;
101     }

103     public Object visit(ASTWhereClause node, Object data) {
104 	QueryRootNode root = (QueryRootNode)data;

107     }

139     public Object visit(ASTRelExpr node, Object data) {
140 	NAryQueryNode parent = (NAryQueryNode)data;
141 	ASTValue value = (ASTValue)node.jjtGetChild(0);
142 	int type = value.getType();
143 	RelationQueryNode rel = null;
144 
145 	if (type == Constants.TYPE_DATE) {
146 	    Date date = ISO8601.parse(value.getValue()).getTime();
147 	    rel = new RelationQueryNode(parent,
148 		    node.getProperty(),
149 		    date,
150 		    node.getOperationType());
151 	} else if (type == Constants.TYPE_DOUBLE) {
152 	    double d = Double.parseDouble(value.getValue());
153 	    rel = new RelationQueryNode(parent,
154 		    node.getProperty(),
155 		    d,
156 		    node.getOperationType());
157 	} else if (type == Constants.TYPE_LONG) {
158 	    long l = Long.parseLong(value.getValue());
159 	    rel = new RelationQueryNode(parent,
160 		    node.getProperty(),
161 		    l,
162 		    node.getOperationType());
163 	} else if (type == Constants.TYPE_STRING) {
164 	    rel = new RelationQueryNode(parent,
165 		    node.getProperty(),
166 		    value.getValue(),
167 		    node.getOperationType());
168 	}
169 
170 	if (rel != null) {
171 	    parent.addOperand(rel);
172 	}
173 	return parent;
174     }
