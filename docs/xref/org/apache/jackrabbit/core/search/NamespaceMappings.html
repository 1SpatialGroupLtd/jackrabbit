21  import javax.jcr.NamespaceException;
22  import java.util.HashMap;
23  import java.util.Map;
24  import java.util.Properties;
25  import java.util.Iterator;

28  /***
29   * The class <code>NamespaceMappings</code> implements a {@link
30   * org.apache.jackrabbit.core.NamespaceResolver} that holds a namespace
31   * mapping that is used internally in the search index. Storing paths with the
32   * full uri of a namespace would require too much space in the search index.
33   * <p/>
34   * Whenever a yet unknown namespace uri to prefix mapping is requested, a new
35   * prefix is created on the fly and associated with the namespace. Known
36   * namespace mappings are stored in a properties file.
37   * 
38   * @author Marcel Reutegger
39   * @version $Revision:  $, $Date:  $
40   */
41  public class NamespaceMappings implements NamespaceResolver {
42  
43      /*** Default logger instance for this class */
44      private static Logger log = Logger.getLogger(NamespaceMappings.class);
45  
46      /*** Location of the file that persists the uri / prefix mappings */
47      private final File storage;
48  
49      /*** Map of uris indexed by prefixes */
50      private Map prefixToURI = new HashMap();
51  
52      /*** Map of prefixes indexed by uris */
53      private Map uriToPrefix = new HashMap();
54  
55      /*** Current prefix count. */
56      private int prefixCount;
57  
58      /***
59       * Creates <code>NamespaceMappings</code> instance. Initial mappings are
60       * loaded from <code>file</code>.
61       *
62       * @param file the <code>File</code> to load initial mappings.
63       * @throws IOException if an error occurs while reading initial namespace
64       *                     mappings from <code>file</code>.
65       */
66      public NamespaceMappings(File file) throws IOException {
67  	storage = file;
68  	load();
69      }
70  
71      /***
72       * Returns a namespace uri for a <code>prefix</code>.
73       *
74       * @param prefix the namespace prefix.
75       * @return the namespace uri.
76       *
77       * @throws NamespaceException if no namespace uri is registered for
78       *                            <code>prefix</code>.
79       */
80      public synchronized String getURI(String prefix) throws NamespaceException {
81  	if (!prefixToURI.containsKey(prefix)) {
82  	    throw new NamespaceException(prefix + ": is not a registered namespace prefix.");
83  	}
84  	return (String) prefixToURI.get(prefix);
85      }
86  
87      /***
88       * Returns a prefix for the namespace <code>uri</code>. If a namespace
89       * mapping exists, the already known prefix is returned; otherwise a new
90       * prefix is created and assigned to the namespace uri.
91       *
92       * @param uri the namespace uri.
93       * @return the prefix for the namespace uri.
94       *
95       * @throws NamespaceException if an yet unknown namespace uri / prefix
96       *                            mapping could not be stored.
97       */
98      public synchronized String getPrefix(String uri) throws NamespaceException {
99  	String prefix = (String)uriToPrefix.get(uri);
100 	if (prefix == null) {
101 	    // make sure prefix is not taken
102 	    while (prefixToURI.get(String.valueOf(prefixCount)) != null) {
103 		prefixCount++;
104 	    }
105 	    prefix = String.valueOf(prefixCount);
106 	    prefixToURI.put(prefix, uri);
107 	    uriToPrefix.put(uri, prefix);
108 	    log.debug("adding new namespace mapping: " + prefix + " -> " + uri);
109 	    try {
110 		store();
111 	    } catch (IOException e) {
112 		throw new NamespaceException("Could not obtain a prefix for uri: " + uri, e);
113 	    }
114 	}
115 	return prefix;
116     }
117 
118     /***
119      * Translates a property name from a session local namespace mapping
120      * into a search index private namespace mapping.
121      * @param name the property name to translate
122      * @param resolver the <code>NamespaceResolver</code> of the local session.
123      * @return the translated property name
124      */
125     public String translatePropertyName(String name, NamespaceResolver resolver)
126 	    throws MalformedPathException {
127 	Path path = Path.create(name, resolver, false);
128 	try {
129 	    return path.toJCRPath(this);
130 	} catch (NoPrefixDeclaredException e) {
131 	    // should never happen actually, because we create yet unknown
132 	    // uri mappings on the fly.
133 	    throw new MalformedPathException(e.toString());
134 	}
135     }

139     /***
140      * Loads currently known mappings from a .properties file.
141      *
142      * @throws IOException if an error occurs while reading from the file.
143      */
144     private void load() throws IOException {
145 	if (storage.exists()) {
146 	    InputStream in = new FileInputStream(storage);
147 	    try {
148 		Properties props = new Properties();
149 		log.debug("loading namespace mappings...");
150 		props.load(in);
151 
152 		// read mappings from properties
153 		Iterator iter = props.keySet().iterator();
154 		while (iter.hasNext()) {
155 		    String prefix = (String) iter.next();
156 		    String uri = props.getProperty(prefix);
157                     log.debug(prefix + " -> " + uri);
158 		    prefixToURI.put(prefix, uri);
159 		    uriToPrefix.put(uri, prefix);
160 		}
161 		prefixCount = props.size();
162 		log.debug("namespace mappings loaded.");
163 	    } finally {
164 		in.close();
165 	    }
166 	}
167     }
168 
169     /***
170      * Writes the currently known mappings into a .properties file.
171      * @throws IOException if an error occurs while writing the file.
172      */
173     private void store() throws IOException {
174 	Properties props = new Properties();
175 
176 	// store mappings in properties
177 	Iterator iter = prefixToURI.keySet().iterator();
178 	while (iter.hasNext()) {
179 	    String prefix = (String) iter.next();
180 	    String uri = (String) prefixToURI.get(prefix);
181 	    props.setProperty(prefix, uri);
182 	}
183 
184 	storage.getParentFile().mkdirs();
185 	OutputStream out = new BufferedOutputStream(new FileOutputStream(storage));
186 
187 	try {
188 	    props.store(out, null);
189 	} finally {
190 	    // make sure stream is closed
191 	    out.close();
192 	}
193     }
