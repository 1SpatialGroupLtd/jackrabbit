16  package org.apache.jackrabbit.core.search;

18  /***
19   * Implements a query node that defines a path restriction.
20   *
21   * @author Marcel Reutegger
22   * @version $Revision:  $, $Date:  $
23   */
24  public class PathQueryNode extends QueryNode {
25  
26      /*** Match path exact*/
27      public static final int TYPE_EXACT = 1;
28  
29      /*** Match child nodes of path */
30      public static final int TYPE_CHILDREN = 2;
31  
32      /*** Match descendant nodes of path */
33      public static final int TYPE_DESCENDANT = 3;
34  
35      /*** The base path */
36      private final String path;
37  
38      /*** Converted path without indexes. /bla[2] -> /bla */
39      private final String indexlessPath;
40  
41      /*** The match type for this query node */
42      private final int type;
43  
44      /*** Flag indicating if this path query contains indexed location steps */
45      private final boolean indexedName;
46  
47      /***
48       * Creates a new <code>PathQueryNode</code> instance.
49       *
50       * @param parent this parent node of this query node.
51       * @param path   the base path.
52       * @param type   one of {@link #TYPE_CHILDREN}, {@link #TYPE_DESCENDANT},
53       *               {@link #TYPE_EXACT}
54       */
55      public PathQueryNode(QueryNode parent, String path, int type) {
56  	super(parent);
57  	if (type < TYPE_EXACT || type > TYPE_DESCENDANT) throw new IllegalArgumentException(String.valueOf(type));
58  	this.path = path;
59  	this.type = type;
60  	this.indexedName = (path.indexOf('[') > -1);
61  	if (indexedName) {
62  	    // also create an indexless path
63  	    StringBuffer tmp = new StringBuffer(path);
64  	    int idx;
65  	    while ((idx = tmp.indexOf("[")) > -1) {
66  		int end = tmp.indexOf("]", idx);
67  		if (end > -1) {
68  		    tmp.replace(idx, end, "");
69  		} else {
70  		    // should never happen
71  		    // FIXME do some error logging?
72  		    break;
73  		}
74  	    }
75  	    this.indexlessPath = tmp.toString();
76  	} else {
77  	    this.indexlessPath = path;
78  	}
79      }
80  
81  
82      /***
83       * @see QueryNode#accept(org.apache.jackrabbit.core.search.QueryNodeVisitor, java.lang.Object)
84       */
85      public Object accept(QueryNodeVisitor visitor, Object data) {
86  	return visitor.visit(this, data);
87      }
88  
89      /***
90       * Returns the unmodified path for this query node, as passed to the
91       * constructor.
92       *
93       * @return the unmodified path for this query node.
94       */
95      public String getPath() {
96  	return path;
97      }
98  
99      /***
100      * Returns a normalized path without indexes.
101      *
102      * @return a normalized path without indexes.
103      */
104     public String getIndexlessPath() {
105 	return indexlessPath;
106     }
107 
108     /***
109      * Returns the type of this <code>PathQueryNode</code>.
110      *
111      * @return one of {@link #TYPE_CHILDREN}, {@link #TYPE_DESCENDANT}, {@link
112      *         #TYPE_EXACT}
113      */
114     public int getType() {
115 	return type;
116     }
117 
118     /***
119      * Returns <code>true</code> if the path contains an index. E.g. a location
120      * step in XPath has a position predicate. If the path does not contain
121      * any indexes <code>false</code> is returned.
122      *
123      * @return <code>true</code> if the path contains an indexed location step.
124      */
125     public boolean hasIndexedName() {
126 	return indexedName;
127     }
128 
129 
130     /***
131      * Returns a JCRQL representation for this query node.
132      * @return a JCRQL representation for this query node.
133      */
134     public String toJCRQLString() {
135 	StringBuffer jcrql = new StringBuffer("LOCATION ");
136 	jcrql.append(path);
137 	if (type == TYPE_CHILDREN) {
138 	    jcrql.append("/*");
139 	} else if (type == TYPE_DESCENDANT) {
140 	    jcrql.append("//");
141 	}
142 	return jcrql.toString();
143     }
