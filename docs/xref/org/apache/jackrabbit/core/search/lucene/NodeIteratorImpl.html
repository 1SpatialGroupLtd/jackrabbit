16  package org.apache.jackrabbit.core.search.lucene;
17  
18  import org.apache.jackrabbit.core.ItemManager;
19  import org.apache.jackrabbit.core.NodeId;
20  import org.apache.log4j.Logger;
21  
22  import javax.jcr.NodeIterator;
23  import javax.jcr.Node;
24  import javax.jcr.RepositoryException;
25  import java.util.NoSuchElementException;
26  
27  /***
28   * @author Marcel Reutegger
29   * @version $Revision:  $, $Date:  $
30   */
31  class NodeIteratorImpl implements NodeIterator {
32  
33      private static final Logger log = Logger.getLogger(NodeIteratorImpl.class);
34  
35      private final String[] uuids;
36  
37      private final ItemManager itemMgr;
38  
39      private int pos = 0;
40  
41      NodeIteratorImpl(ItemManager itemMgr,
42  		     String[] uuids) {
43  	this.itemMgr = itemMgr;
44  	this.uuids = uuids;
45      }
46  
47      public Node nextNode() {
48  	if (pos >= uuids.length) throw new NoSuchElementException();
49  	try {
50  	    return (Node)itemMgr.getItem(new NodeId(uuids[pos++]));
51  	} catch (RepositoryException e) {
52  	    log.error("Exception retrieving Node with UUID: "
53  		    + uuids[pos] + ": " + e.toString());
54  	    // FIXME this is bad error handling!
55  	    throw new NoSuchElementException();
56  	}
57      }
58  
59      public Object next() {
60  	return nextNode();
61      }
62  
63      public void skip(long skipNum) {
64  	if ((pos + skipNum) > uuids.length) throw new NoSuchElementException();
65  	pos += skipNum;
66      }
67  
68      public long getSize() {
69  	return uuids.length;
70      }
71  
72      public long getPos() {
73  	return pos;
74      }
75  
76      public boolean hasNext() {
77  	return pos < uuids.length;
78      }
79  
80      public void remove() {
81  	throw new UnsupportedOperationException("remove");
82      }
83  
84  }
