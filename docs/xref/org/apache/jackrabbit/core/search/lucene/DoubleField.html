16  package org.apache.jackrabbit.core.search.lucene;
17  
18  /***
19   * The <code>DoubleField</code> class is a utility to convert double
20   * values into <code>String</code> values that are lexicographically ordered
21   * according to the double value.
22   *
23   * @author Marcel Reutegger
24   * @version $Revision:  $, $Date:  $
25   */
26  class DoubleField {
27  
28      private static final long SIGN_MASK = 0x8000000000000000L;
29  
30      private static final int STRING_DOUBLE_LEN
31  	    = Long.toString(Long.MAX_VALUE, Character.MAX_RADIX).length() + 1;
32  
33      private DoubleField() {
34      }
35  
36      public static String doubleToString(double value) {
37  	long longValue = Double.doubleToLongBits(value);
38  	StringBuffer sb = new StringBuffer(STRING_DOUBLE_LEN);
39  
40  	if ((longValue & SIGN_MASK) == 0) {
41  	    // positive
42  	    String s = Long.toString(longValue, Character.MAX_RADIX);
43  	    // add sign character
44  	    sb.append('1');
45  	    while ((sb.length() + s.length()) < STRING_DOUBLE_LEN) {
46  		sb.append('0');
47  	    }
48  	    sb.append(s);
49  	} else {
50  	    // negative
51  	    // fold value by reversing sign
52  	    longValue = -longValue;
53  	    String s = Long.toString(longValue, Character.MAX_RADIX);
54  	    // pad with leading zeros
55  	    while ((sb.length() + s.length()) < STRING_DOUBLE_LEN) {
56  		sb.append('0');
57  	    }
58  	    sb.append(s);
59  	}
60  	return sb.toString();
61      }
62  
63      public static double stringToDouble(String value) {
64  	if (value.charAt(0) == '1') {
65  	    // positive
66  	    long longValue = Long.parseLong(value.substring(1), Character.MAX_RADIX);
67  	    return Double.longBitsToDouble(longValue);
68  	} else if (value.charAt(0) == '0') {
69  	    // negative
70  	    long longValue = Long.parseLong(value, Character.MAX_RADIX);
71  	    // switch sign
72  	    longValue = -longValue;
73  	    return Double.longBitsToDouble(longValue);
74  	} else {
75  	    throw new IllegalArgumentException("not a valid string representation of a double: " + value);
76  	}
77      }
78  }
