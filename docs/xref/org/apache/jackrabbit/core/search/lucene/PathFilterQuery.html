18  import org.apache.lucene.search.*;
19  import org.apache.lucene.index.IndexReader;

25  /***
26   * A PathFilterQuery is a wrapper around a native lucene query and
27   * adds a {@link org.apache.jackrabbit.core.search.lucene.PackageFilter} around that query.
28   *
29   * @version $Revision: 1.3 $, $Date: 2004/01/23 15:55:40 $
30   * @author Marcel Reutegger
31   */
32  class PathFilterQuery extends Query {
33  
34      /*** The lucene <code>Query</code> object, containing the real query */
35      private Query delegatee;
36  
37      /*** The {@link PackageFilter} used for filtering */
38      private final PackageFilter filter;
39  
40      /***
41       * Creates a new <code>PathFilterQuery</code> object.
42       *
43       * @param delegatee the real lucene query.
44       * @param filter the <code>PackageFilter</code> for filtering the result.
45       */
46      PathFilterQuery(Query delegatee, PackageFilter filter) {
47  	this.delegatee = delegatee;
48  	this.filter = filter;
49      }
50  
51      /***
52       * Returns the real lucene <code>Query</code> object.
53       * @return the real lucene <code>Query</code> object.
54       */
55      Query getQuery() {
56  	return delegatee;
57      }
58  
59      /***
60       * Returns a <code>String</code> representation of a field.
61       * @param field the field for which to return a <code>String</code>
62       *   representaion.
63       * @return a <code>String</code> representation of a field.
64       */
65      public String toString(String field) {
66  	return delegatee.toString(field);
67      }
68  
69      /***
70       * Returns the <code>Weight</code> object for this
71       * <code>HandleFilterQuery</code>.
72       * @param searcher the <code>Searcher</code> to create
73       *   <code>Weight</code> object.
74       * @return the <code>Weight</code> object for this query.
75       * @throws IOException if an error occurs while creating the
76       *   <code>Weight</code>.
77       */
78      public Weight weight(Searcher searcher)
79  	    throws IOException {
80  	return new FilteredWeight(searcher);
81      }
82  
83      /***
84       * Returns a clone of this query.
85       */
86      public Object clone() {
87  	PathFilterQuery clone = (PathFilterQuery)super.clone();
88  	clone.delegatee = (Query)this.delegatee.clone();
89  	return clone;
90      }
91  
92      /***
93       * Creates a <code>Weight</code> object for this query.
94       * @param searcher the <code>Searcher</code> to create
95       *   <code>Weight</code> object.
96       * @return a <code>Weight</code> object for this query.
97       */
98      protected Weight createWeight(Searcher searcher) {
99  	return new FilteredWeight(searcher);
100     }
101 
102     public Query rewrite(IndexReader reader) throws IOException {
103 	delegatee = delegatee.rewrite(reader);
104 	return this;
105     }

109     /***
110      * This is the <code>Weight</code> implementation for this query.
111      * It simply wraps the <code>Weight</code> object of the real
112      * query and creates a {@link org.apache.jackrabbit.core.search.lucene.FilteredScorer} in {@link #scorer}.
113      */
114     private class FilteredWeight implements Weight {
115 
116 	/*** the <code>Searcher</code> to create a <code>Weight</code> object */
117 	private Searcher searcher;
118 
119 	/*** The <code>Weight</code> created by the real query */
120 	private Weight weight;
121 
122 	/***
123 	 * Creates a <code>FilteredWeight</code> with a given
124 	 * <code>Searcher</code>.
125 	 * @param searcher the <code>Search</code> to create the
126 	 *   real <code>Weight</code>.
127 	 */
128 	FilteredWeight(Searcher searcher) {
129 	    this.searcher = searcher;
130 	}
131 
132 	/***
133 	 * @see Weight#explain
134 	 */
135 	public Explanation explain(IndexReader reader, int doc) throws IOException {
136 	    if (weight == null) {
137 		weight = delegatee.weight(searcher);
138 	    }
139 	    return weight.explain(reader, doc);
140 	}
141 
142 	/***
143 	 * @see Weight#getQuery
144 	 */
145 	public Query getQuery() {
146 	    return PathFilterQuery.this;
147 	}
148 
149 	/***
150 	 * @see Weight#getValue
151 	 */
152 	public float getValue() {
153 	    return weight.getValue();
154 	}
155 
156 	/***
157 	 * @see Weight#normalize
158 	 */
159 	public void normalize(float norm) {
160 	    weight.normalize(norm);
161 	}
162 
163 	/***
164 	 * Returns a {@link FilteredScorer} instance, which only
165 	 * scores hits that are not filtered by the {@link org.apache.jackrabbit.core.search.lucene.PackageFilter}.
166 	 * @param reader <code>IndexReader</code> for reading from the search
167 	 *   index.
168 	 * @return a {@link org.apache.jackrabbit.core.search.lucene.FilteredScorer} instance.
169 	 * @throws IOException if an error occurs while reading from the
170 	 *   search index.
171 	 */
172 	public Scorer scorer(IndexReader reader) throws IOException {
173 	    if (weight == null) {
174 		weight = delegatee.weight(searcher);
175 	    }
176 	    return new FilteredScorer(weight.scorer(reader), filter, reader);
177 	}
178 
179 	/***
180 	 * @see Weight#sumOfSquaredWeights
181 	 */
182 	public float sumOfSquaredWeights() throws IOException {
183 	    if (weight == null) {
184 		weight = delegatee.weight(searcher);
185 	    }
186 	    return weight.sumOfSquaredWeights();
187 	}
188     }
