111     /***
112      * Package private constructor.
113      *
114      * @param itemMgr   the <code>ItemManager</code> that created this <code>Item</code>
115      * @param session   the <code>Session</code> through which this <code>Item</code> is acquired
116      * @param id        id of this <code>Item</code>
117      * @param state     state associated with this <code>Item</code>
118      * @param listeners listeners on life cylce changes of this <code>ItemImpl</code>
119      */
120     ItemImpl(ItemManager itemMgr, SessionImpl session, ItemId id, ItemState state,
121 	     ItemLifeCycleListener[] listeners) {
122 	this.session = session;
123 	rep = (RepositoryImpl) session.getRepository();
124 	itemStateMgr = session.getItemStateManager();
125 	this.id = id;
126 	this.itemMgr = itemMgr;
127 	this.state = state;
128 	status = STATUS_NORMAL;
129 
130 	if (listeners != null) {
131 	    for (int i = 0; i < listeners.length; i++) {
132 		addLifeCycleListener(listeners[i]);
133 	    }
134 	}
135 	notifyCreated();
136 
137 	// add this item as listener to events of the underlying state object
138 	this.state.addListener(this);
139     }

144 	}
145     }

147     protected void checkItemState() throws InvalidItemStateException {
148 	// check status for read operation
149 	switch (status) {
150 	    case STATUS_NORMAL:
151 	    case STATUS_MODIFIED:
152 		return;
153 
154 	    case STATUS_DESTROYED:
155 	    case STATUS_INVALIDATED:
156 		throw new InvalidItemStateException(id + ": the item does not exist anymore");
157 	}
158     }

168     /***
169      * Marks this instance as 'removed' and notifies its listeners.
170      * The resulting state is either 'temporarily invalidated' or
171      * 'permanently invalidated', depending on the initial state.
172      *
173      * @throws RepositoryException if an error occurs
174      */
175     protected void setRemoved() throws RepositoryException {
176 	if (status == STATUS_INVALIDATED || status == STATUS_DESTROYED) {
177 	    // this instance is already 'invalid', get outta here
178 	    return;
179 	}
180 
181 	ItemState transientState = getOrCreateTransientItemState();
182 	if (transientState.getStatus() == ItemState.STATUS_NEW) {
183 	    // this is a 'new' item, simply dispose the transient state
184 	    // (it is no longer used); this will indirectly (through
185 	    // stateDiscarded listener method) invalidate this instance permanently
186 	    itemStateMgr.disposeTransientItemState(transientState);
187 	} else {
188 	    // this is an 'existing' item (i.e. it is backed by persistent
189 	    // state), mark it as 'removed'
190 	    transientState.setStatus(ItemState.STATUS_EXISTING_REMOVED);
191 	    // transfer the transient state to the attic
192 	    itemStateMgr.moveTransientItemStateToAttic(transientState);
193 
194 	    // set state of this instance to 'invalid'
195 	    status = STATUS_INVALIDATED;
196 	    // notify the listeners that this instance has been
197 	    // temporarily invalidated
198 	    notifyInvalidated();
199 	}
200     }

211     /***
212      * Notify the listeners that this instance has been discarded
213      * (i.e. it has been temporarily rendered 'invalid').
214      */
215     private void notifyCreated() {
216 	// copy listeners to array to avoid ConcurrentModificationException
217 	ItemLifeCycleListener[] la = new ItemLifeCycleListener[listeners.size()];
218 	Iterator iter = listeners.values().iterator();
219 	int cnt = 0;
220 	while (iter.hasNext()) {
221 	    la[cnt++] = (ItemLifeCycleListener) iter.next();
222 	}
223 	for (int i = 0; i < la.length; i++) {
224 	    if (la[i] != null) {
225 		la[i].itemCreated(this);
226 	    }
227 	}
228     }
229

325     private void validateTransientItems(Iterator iter)
326 	    throws AccessDeniedException, ConstraintViolationException, RepositoryException {
327 	/***
328 	 * the following validations/checks are performed on transient items:
329 	 *
330 	 * for every transient node:
331 	 * - if it is 'new', check that its node type satisfies the
332 	 *   'required node type' constraint specified in its definition
333 	 * - if new child nodes have been added to the node in question,
334 	 *   check the ADD_NODE permission
335 	 * - if child items have been removed from the node in question,
336 	 *   check the REMOVE_ITEM permission
337 	 * - check if 'mandatory' child items exist
338 	 *
339 	 * for every transient property:
340 	 * - check the SET_PROPERTY permission
341 	 * - check if the property value satisfies the value constraints specified
342 	 *   in the property's definition
343 	 *
344 	 * note that the protected flag is checked in Node.addNode/Node.remove
345 	 * (for adding/removing child entries of a node), in
346 	 * Node.addMixin/removeMixin (for mixin changes on nodes)
347 	 * and in Property.setValue (for properties to be modified).
348 	 */
349 
350 	AccessManagerImpl accessMgr = session.getAccessManager();
351 	// walk through list of transient items and validate each
352 	while (iter.hasNext()) {
353 	    ItemState itemState = (ItemState) iter.next();
354 
355 	    if (itemState.isNode()) {
356 		// the transient item is a node
357 		NodeState nodeState = (NodeState) itemState;
358 		ItemId id = nodeState.getId();
359 		NodeImpl node = (NodeImpl) itemMgr.getItem(id);
360 		NodeDef def = node.getDefinition();
361 		NodeTypeImpl nt = (NodeTypeImpl) node.getPrimaryNodeType();
362 		// if the transient node was added (i.e. if it is 'new'),
363 		// check its node's node type against the required node type
364 		// in its definition
365 		NodeType[] nta = def.getRequiredPrimaryTypes();
366 		for (int i = 0; i < nta.length; i++) {
367 		    NodeTypeImpl ntReq = (NodeTypeImpl) nta[i];
368 		    if (nodeState.getStatus() == ItemState.STATUS_NEW
369 			    && !(nt.getQName().equals(ntReq.getQName())
370 			    || nt.isDerivedFrom(ntReq.getQName()))) {
371 			// the transient node's node type does not satisfy the
372 			// 'required primary types' constraint
373 			String msg = node.safeGetJCRPath() + " must be of node type " + ntReq.getName();
374 			log.warn(msg);
375 			throw new ConstraintViolationException(msg);
376 		    }
377 		}
378 
379 		// check child removals
380 		if (!nodeState.getRemovedChildNodeEntries().isEmpty() || !nodeState.getRemovedPropertyEntries().isEmpty()) {
381 		    // check REMOVE_ITEM permission
382 		    if (!accessMgr.isGranted(id, Permission.REMOVE_ITEM)) {
383 			String msg = node.safeGetJCRPath() + ": not allowed to remove a child item";
384 			log.error(msg);
385 			throw new AccessDeniedException(msg);
386 		    }
387 
388 		    // no need to check the protected flag
389 		    // as this is checked in NodeImpl.remove(String)
390 		}
391 
392 		// check child additions
393 		// added child nodes
394 		Iterator addedIter = nodeState.getAddedChildNodeEntries().iterator();
395 		while (addedIter.hasNext()) {
396 		    NodeState.ChildNodeEntry entry = (NodeState.ChildNodeEntry) addedIter.next();
397 		    Node childNode = (Node) itemMgr.getItem(new NodeId(entry.getUUID()));
398 		    NodeDef childDef = childNode.getDefinition();
399 		    if (!childDef.isAutoCreate()) {
400 			// check ADD_NODE permission
401 			if (!accessMgr.isGranted(id, Permission.ADD_NODE)) {
402 			    String msg = node.safeGetJCRPath() + ": not allowed to add node " + childNode.getName();
403 			    log.error(msg);
404 			    throw new AccessDeniedException(msg);
405 			}
406 		    }
407 		}
408 		// mandatory child properties
409 		PropertyDef[] propDefs = nt.getMandatoryPropertyDefs();
410 		for (int i = 0; i < propDefs.length; i++) {
411 		    PropertyDefImpl pd = (PropertyDefImpl) propDefs[i];
412 		    if (!nodeState.hasPropertyEntry(pd.getQName())) {
413 			String msg = node.safeGetJCRPath() + ": mandatory property " + pd.getName() + " does not exist";
414 			log.warn(msg);
415 			throw new ConstraintViolationException(msg);
416 		    }
417 		}
418 		// mandatory child nodes
419 		NodeDef[] nodeDefs = nt.getMandatoryNodeDefs();
420 		for (int i = 0; i < nodeDefs.length; i++) {
421 		    NodeDefImpl nd = (NodeDefImpl) nodeDefs[i];
422 		    if (!nodeState.hasChildNodeEntry(nd.getQName())) {
423 			String msg = node.safeGetJCRPath() + ": mandatory child node " + nd.getName() + " does not exist";
424 			log.warn(msg);
425 			throw new ConstraintViolationException(msg);
426 		    }
427 		}
428 	    } else {
429 		// the transient item is a property
430 		PropertyState propState = (PropertyState) itemState;
431 		ItemId propId = propState.getId();
432 		NodeId nodeId = new NodeId(propState.getParentUUID());
433 		PropertyImpl prop = (PropertyImpl) itemMgr.getItem(propId);
434 		PropertyDefImpl def = (PropertyDefImpl) prop.getDefinition();
435 
436 		if (!def.isAutoCreate()) {
437 		    // check SET_PROPERTY permission
438 		    if (!accessMgr.isGranted(nodeId, Permission.SET_PROPERTY)) {
439 			String msg = itemMgr.safeGetJCRPath(nodeId) + ": not allowed to set property " + prop.getName();
440 			log.error(msg);
441 			throw new AccessDeniedException(msg);
442 		    }
443 		}
444 
445 		// check value constraints
446 		// (no need to check value constraints of protected properties
447 		// as those are set by the implementation only, i.e. they
448 		// cannot be set by the user through the api)
449 		if (!def.isProtected()) {
450 		    if (def.getValueConstraint() != null) {
451 			InternalValue[] values = propState.getValues();
452 			try {
453 			    NodeTypeImpl.checkSetPropertyValueConstraints(def, values);
454 			} catch (RepositoryException e) {
455 			    // repack exception for providing verboser error message
456 			    String msg = prop.safeGetJCRPath() + ": " + e.getMessage();
457 			    log.warn(msg);
458 			    throw new ConstraintViolationException(msg);
459 			}
460 		    }
461 		}
462 
463 		// no need to check the protected flag
464 		// as this is checked in PropertyImpl.setValue(Value)
465 	    }
466 	}
467     }

469     private void checkReferences(Iterator iterDirty, Iterator iterRemoved,
470 				 ReferenceManager refMgr)
471 	    throws ConstraintViolationException, RepositoryException {
472 	// map of target (node) id's and modified NodeReferences objects
473 	HashMap dirtyNodeRefs = new HashMap();
474 
475 	// walk through dirty items and process REFERENCE properties:
476 	// 1. verify that target node exists
477 	// 2. update and collect the affected NodeReferences objects of the
478 	//    target nodes in the dirtyNodeRefs map
479 	while (iterDirty.hasNext()) {
480 	    ItemState transientState = (ItemState) iterDirty.next();
481 	    if (!transientState.isNode()) {
482 		PropertyState propState = (PropertyState) transientState;
483 		if (propState.getType() == PropertyType.REFERENCE) {
484 		    if (propState.getStatus() == ItemState.STATUS_EXISTING_MODIFIED) {
485 			// this is a modified REFERENCE property:
486 			// remove the 'reference' stored in the old value
487 			PropertyState oldPropState = (PropertyState) propState.getOverlayedState();
488 			InternalValue[] vals = oldPropState.getValues();
489 			for (int i = 0; vals != null && i < vals.length; i++) {
490 			    String uuid = vals[i].toString();
491 			    NodeId targetId = new NodeId(uuid);
492 			    NodeReferences refs;
493 			    if (dirtyNodeRefs.containsKey(targetId)) {
494 				refs = (NodeReferences) dirtyNodeRefs.get(targetId);
495 			    } else {
496 				refs = refMgr.get(targetId);
497 				dirtyNodeRefs.put(targetId, refs);
498 			    }
499 			    // remove reference from target node
500 			    refs.removeReference((PropertyId) propState.getId());
501 			}
502 		    }
503 		    // add the reference stored in the new value
504 		    InternalValue[] vals = propState.getValues();
505 		    for (int i = 0; vals != null && i < vals.length; i++) {
506 			String uuid = vals[i].toString();
507 			NodeId targetId = new NodeId(uuid);
508 			// verify that target exists
509 			if (!itemMgr.itemExists(targetId)) {
510 			    String msg = itemMgr.safeGetJCRPath(propState.getId())
511 				    + ": target node of REFERENCE property does not exist";
512 			    log.warn(msg);
513 			    throw new ConstraintViolationException(msg);
514 			}
515 			// target is a new (unsaved) node; make sure that it is
516 			// within the scope of the current save operation
517 			// (by veryfying that it is a descendant of 'this' item)
518 			NodeImpl target = (NodeImpl) itemMgr.getItem(targetId);
519 			if (target.isNew()) {
520 			    try {
521 				if (!target.getPrimaryPath().isDescendantOf(getPrimaryPath())) {
522 				    String msg = itemMgr.safeGetJCRPath(propState.getId())
523 					    + ": target node of REFERENCE property is a new node and must therefore either be saved first or be within the scope of the current save operation.";
524 				    log.warn(msg);
525 				    throw new ConstraintViolationException(msg);
526 				}
527 			    } catch (MalformedPathException mpe) {
528 				// should never get here...
529 				String msg = itemMgr.safeGetJCRPath(propState.getId())
530 					+ ": failed to verify existence of target node";
531 				log.error(msg, mpe);
532 				throw new RepositoryException(msg, mpe);
533 			    }
534 			}
535 			NodeReferences refs;
536 			if (dirtyNodeRefs.containsKey(targetId)) {
537 			    refs = (NodeReferences) dirtyNodeRefs.get(targetId);
538 			} else {
539 			    refs = refMgr.get(targetId);
540 			    dirtyNodeRefs.put(targetId, refs);
541 			}
542 			// add reference to target node
543 			refs.addReference((PropertyId) propState.getId());
544 		    }
545 		}
546 	    }
547 	}
548 
549 	// walk through 'removed' items:
550 	// 1. build list of removed nodes
551 	// 2. process REFERENCE properties (update and collect the affected
552 	//    NodeReferences objects of the target nodes)
553 	ArrayList removedNodes = new ArrayList();
554 	while (iterRemoved.hasNext()) {
555 	    ItemState transientState = (ItemState) iterRemoved.next();
556 	    if (transientState.isNode()) {
557 		// removed node: collect for later processing
558 		removedNodes.add(transientState);
559 	    } else {
560 		PropertyState propState = (PropertyState) transientState;
561 		if (propState.getType() == PropertyType.REFERENCE) {
562 		    // this is a removed REFERENCE property:
563 		    // remove the 'reference' stored in the value
564 		    InternalValue[] vals = propState.getValues();
565 		    for (int i = 0; i < vals.length; i++) {
566 			String uuid = vals[i].toString();
567 			NodeId targetId = new NodeId(uuid);
568 			NodeReferences refs;
569 			if (dirtyNodeRefs.containsKey(targetId)) {
570 			    refs = (NodeReferences) dirtyNodeRefs.get(targetId);
571 			} else {
572 			    refs = refMgr.get(targetId);
573 			    dirtyNodeRefs.put(targetId, refs);
574 			}
575 			// remove reference to target node
576 			refs.removeReference((PropertyId) propState.getId());
577 		    }
578 		}
579 	    }
580 	}
581 
582 	// now that all NodeReferences objects have been updated,
583 	// walk through 'removed' nodes and verify that no node that is still
584 	// being referenced, is removed
585 	Iterator iter = removedNodes.iterator();
586 	while (iter.hasNext()) {
587 	    NodeState nodeState = (NodeState) iter.next();
588 	    // check if node is referenced
589 	    NodeId targetId = (NodeId) nodeState.getId();
590 	    NodeReferences refs;
591 	    if (dirtyNodeRefs.containsKey(targetId)) {
592 		refs = (NodeReferences) dirtyNodeRefs.get(targetId);
593 	    } else {
594 		refs = refMgr.get(targetId);
595 	    }
596 	    if (refs.hasReferences()) {
597 		String msg = nodeState.getId()
598 			+ ": the node cannot be removed because it is being referenced.";
599 		log.warn(msg);
600 		throw new ConstraintViolationException(msg);
601 	    }
602 	}
603 
604 	// persist dirty NodeReferences objects
605 	iter = dirtyNodeRefs.values().iterator();
606 	while (iter.hasNext()) {
607 	    NodeReferences refs = (NodeReferences) iter.next();

726     public void stateDiscarded(ItemState discarded) {
727 	// the state of this item has been discarded, probably as a result
728 	// of calling Node.revert() or ItemImpl.setRemoved()
729 	if (isTransient()) {
730 	    switch (state.getStatus()) {
731 		/***
732 		 * persistent item that has been transiently removed
733 		 */
734 		case ItemState.STATUS_EXISTING_REMOVED:
735 		    /***
736 		     * persistent item that has been transiently modified
737 		     */
738 		case ItemState.STATUS_EXISTING_MODIFIED:
739 		    /***
740 		     * persistent item that has been transiently modified or removed
741 		     * and the underlying persistent state has been externally
742 		     * modified since the transient modification/removal.
743 		     */
744 		case ItemState.STATUS_STALE_MODIFIED:
745 		    ItemState persistentState = state.getOverlayedState();
746 		    // the state is a transient wrapper for the underlying
747 		    // persistent state, therefor restore the
748 		    // persistent state and resurrect this item instance
749 		    // if necessary
750 		    state.removeListener(this);
751 		    state = persistentState;
752 		    state.addListener(this);
753 
754 		    if (status == STATUS_INVALIDATED) {
755 			// resurrect this instance
756 			status = STATUS_NORMAL;
757 			// notify the listeners
758 			notifyResurrected();
759 		    }
760 		    return;
761 
762 		    /***
763 		     * persistent item that has been transiently modified or removed
764 		     * and the underlying persistent state has been externally
765 		     * destroyed since the transient modification/removal.
766 		     */
767 		case ItemState.STATUS_STALE_DESTROYED:
768 		    // set state of this instance to 'destroyed'
769 		    status = STATUS_DESTROYED;
770 		    // dispose state
771 		    state.removeListener(this);
772 		    state = null;
773 		    // notify the listeners that this instance has been
774 		    // permanently invalidated
775 		    notifyDestroyed();
776 		    return;
777 
778 		    /***
779 		     * new item that has been transiently added
780 		     */
781 		case ItemState.STATUS_NEW:
782 		    // set state of this instance to 'destroyed'
783 		    status = STATUS_DESTROYED;
784 		    // dispose state
785 		    state.removeListener(this);
786 		    state = null;
787 		    // notify the listeners that this instance has been
788 		    // permanently invalidated
789 		    notifyDestroyed();
790 		    return;
791 	    }
792 	}
793 
794 	// render this instance 'invalid'
795 	status = STATUS_INVALIDATED;
796 	// notify the listeners
797 	notifyInvalidated();

840     public synchronized void save()
841 	    throws AccessDeniedException, LockException,
842 	    ConstraintViolationException, InvalidItemStateException,
843 	    RepositoryException {
844 	// check state of this instance
845 	checkItemState();
846 
847 	// list of transient states that should be persisted
848 	ArrayList dirty = new ArrayList();
849 	ItemState transientState;
850 
851 	// list of events that are generated by saved changes
852 	WorkspaceImpl wsp = (WorkspaceImpl) session.getWorkspace();
853 	ObservationManagerFactory obsFactory = rep.getObservationManagerFactory(wsp.getName());
854 	EventStateCollection events = obsFactory.createEventStateCollection(session,
855 		session.getItemStateManager(), session.getHierarchyManager());
856 
857 	// check status of this item's state
858 	if (isTransient()) {
859 	    switch (state.getStatus()) {
860 		case ItemState.STATUS_EXISTING_MODIFIED:
861 		    // add this item's state to the list
862 		    dirty.add(state);
863 		    // create event
864 		    events.createEventStates(state);
865 		    break;
866 
867 		case ItemState.STATUS_NEW:
868 		    {
869 			String msg = safeGetJCRPath() + ": cannot save a new item.";
870 			log.error(msg);
871 			throw new RepositoryException(msg);
872 		    }
873 
874 		case ItemState.STATUS_STALE_MODIFIED:
875 		    {
876 			String msg = safeGetJCRPath() + ": the item cannot be saved because it has been modified externally.";
877 			log.error(msg);
878 			throw new InvalidItemStateException(msg);
879 		    }
880 
881 		case ItemState.STATUS_STALE_DESTROYED:
882 		    {
883 			String msg = safeGetJCRPath() + ": the item cannot be saved because it has been deleted externally.";
884 			log.error(msg);
885 			throw new InvalidItemStateException(msg);
886 		    }
887 
888 		default:
889 		    log.debug("unexpected state status (" + state.getStatus() + ")");
890 		    // ignore
891 		    break;
892 	    }
893 	}
894 
895 	if (isNode()) {
896 	    // build list of 'new' or 'modified' descendents
897 	    Iterator iter = itemStateMgr.getDescendantTransientItemStates(id);
898 	    while (iter.hasNext()) {
899 		transientState = (ItemState) iter.next();
900 		switch (transientState.getStatus()) {
901 		    case ItemState.STATUS_NEW:
902 		    case ItemState.STATUS_EXISTING_MODIFIED:
903 			// add modified state to the list
904 			dirty.add(transientState);
905 			// create events
906 			events.createEventStates(transientState);
907 			break;
908 
909 		    case ItemState.STATUS_STALE_MODIFIED:
910 			{
911 			    String msg = transientState.getId() + ": the item cannot be saved because it has been modified externally.";
912 			    log.error(msg);
913 			    throw new InvalidItemStateException(msg);
914 			}
915 
916 		    case ItemState.STATUS_STALE_DESTROYED:
917 			{
918 			    String msg = transientState.getId() + ": the item cannot be saved because it has been deleted externally.";
919 			    log.error(msg);
920 			    throw new InvalidItemStateException(msg);
921 			}
922 
923 		    default:
924 			log.debug("unexpected state status (" + transientState.getStatus() + ")");
925 			// ignore
926 			break;
927 		}
928 	    }
929 	}
930 
931 	if (dirty.size() == 0) {
932 	    // no transient items, nothing to do here
933 	    return;
934 	}
935 
936 	// check that parent node is also included in the dirty items list
937 	// if dirty node was removed or added (adding/removing a parent/child
938 	// link requires that both parent and child are saved)
939 	Iterator iter = dirty.iterator();
940 	while (iter.hasNext()) {
941 	    transientState = (ItemState) iter.next();
942 	    if (transientState.isNode()) {
943 		NodeState nodeState = (NodeState) transientState;
944 		ArrayList dirtyParents = new ArrayList();
945 		// removed parents
946 		dirtyParents.addAll(nodeState.getRemovedParentUUIDs());
947 		// added parents
948 		dirtyParents.addAll(nodeState.getAddedParentUUIDs());
949 		Iterator parentsIter = dirtyParents.iterator();
950 		while (parentsIter.hasNext()) {
951 		    NodeId id = new NodeId((String) parentsIter.next());
952 		    NodeState parentState = null;
953 		    try {
954 			parentState = (NodeState) itemStateMgr.getTransientItemState(id);
955 		    } catch (ItemStateException ise) {
956 			// should never get here...
957 			String msg = "inconsistency: failed to retrieve transient state for " + itemMgr.safeGetJCRPath(id);
958 			log.error(msg);
959 			throw new RepositoryException(msg);
960 		    }
961 		    // check if parent is also going to be saved
962 		    if (!dirty.contains(parentState)) {
963 			// need to save the parent too
964 			String msg = itemMgr.safeGetJCRPath(id) + " needs to be saved also.";
965 			log.error(msg);
966 			throw new RepositoryException(msg);
967 		    }
968 		}
969 	    }
970 	}
971 
972 	// validate access and node type constraints
973 	// (this will also validate child removals)
974 	validateTransientItems(dirty.iterator());
975 
976 	// we need to make sure that we are not interrupted while
977 	// verifying/persisting node references
978 	ReferenceManager refMgr = wsp.getReferenceManager();
979 	synchronized (refMgr) {
980 
981 	    // build list of transient descendents in the attic
982 	    // (i.e. those marked as 'removed')
983 	    ArrayList removed = new ArrayList();
984 	    if (isNode()) {
985 		iter = itemStateMgr.getDescendantTransientItemStatesInAttic(id);
986 		while (iter.hasNext()) {
987 		    transientState = (ItemState) iter.next();
988 		    // check if stale
989 		    if (transientState.getStatus() == ItemState.STATUS_STALE_MODIFIED) {
990 			String msg = transientState.getId() + ": the item cannot be removed because it has been modified externally.";
991 			log.error(msg);
992 			throw new InvalidItemStateException(msg);
993 		    }
994 		    if (transientState.getStatus() == ItemState.STATUS_STALE_DESTROYED) {
995 			String msg = transientState.getId() + ": the item cannot be removed because it has already been deleted externally.";
996 			log.error(msg);
997 			throw new InvalidItemStateException(msg);
998 		    }
999 		    removed.add(transientState);
1000 		    events.createEventStates(transientState);
1001 		}
1002 	    }
1003 
1004 	    // referential integrity checks:
1005 	    // make sure that a referenced node cannot be removed and
1006 	    // that all references are updated and persisted
1007 	    checkReferences(dirty.iterator(), removed.iterator(), refMgr);
1008 
1009 	    // prepare for event dispatch
1010 	    // this will check access rights on items that will be removed
1011 	    events.prepare();
1012 
1013 	    // definitively remove transient items marked as 'removed'
1014 	    removeTransientItems(removed.iterator());
1015 
1016 	    // dispose the transient states marked 'removed'
1017 	    iter = removed.iterator();
1018 	    while (iter.hasNext()) {
1019 		transientState = (ItemState) iter.next();
1020 		// dispose the transient state, it is no longer used
1021 		// this will indirectly (through stateDiscarded listener method)
1022 		// permanently invalidate the wrapping Item instance
1023 		itemStateMgr.disposeTransientItemStateInAttic(transientState);
1024 	    }
1025 
1026 	    // initialize version histories for new nodes
1027 	    initVersionHistories(dirty.iterator());
1028 
1029 	    // persist 'new' or 'modified' transient states
1030 	    persistTransientItems(dirty.iterator());
1031 	} // synchronized(refMgr)
1032 
1033 	// now it is safe to dispose the transient states
1034 	iter = dirty.iterator();
1035 	while (iter.hasNext()) {
1036 	    transientState = (ItemState) iter.next();
1037 	    // dispose the transient state, it is no longer used
1038 	    itemStateMgr.disposeTransientItemState(transientState);
1039 	}
1040 
1041 	// all changes are persisted, now dispatch events
1042 	events.dispatch();
1043     }
