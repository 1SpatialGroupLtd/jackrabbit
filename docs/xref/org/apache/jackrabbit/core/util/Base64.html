18  import java.io.*;

20  /***
21   * <code>Base64</code> provides Base64 encoding/decoding of strings and streams.
22   *
23   * @author Stefan Guggisberg
24   * @version $Revision: 1.9 $, $Date: 2004/08/02 16:19:52 $
25   */
26  public class Base64 {
27      // charset used for base64 encoded data (7-bit ASCII)
28      private static final String CHARSET = "US-ASCII";
29  
30      // encoding table (the 64 valid base64 characters)
31      private static final char[] BASE64CHARS =
32  	    "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/".toCharArray();
33  
34      // decoding table (used to lookup original 6-bit with base64 character as table index)
35      private static final byte[] DECODETABLE = new byte[128];
36  
37      static {
38  	// initialize decoding table
39  	for (int i = 0; i < DECODETABLE.length; i++) {
40  	    DECODETABLE[i] = 0x7f;
41  	}
42  	// build decoding table
43  	for (int i = 0; i < BASE64CHARS.length; i++) {
44  	    DECODETABLE[BASE64CHARS[i]] = (byte) i;
45  	}
46      }
47  
48      // pad character
49      private static final char BASE64PAD = '=';
50  
51      /***
52       * empty private constructor
53       */
54      private Base64() {
55      }
56  
57      /***
58       * Calculates the size (i.e. number of bytes) of the base64 encoded output
59       * given the length (i.e. number of bytes) of the data to be encoded.
60       *
61       * @param dataLength length (i.e. number of bytes) of the data to be encoded
62       * @return size (i.e. number of bytes) of the base64 encoded output
63       */
64      public static long calcEncodedLength(long dataLength) {
65  	long encLen = dataLength * 4 / 3;
66  	encLen += (encLen + 4) % 4;
67  	return encLen;
68      }
69  
70      /***
71       * Pessimistically guesses the size (i.e. number of bytes) of the decoded output
72       * given the length (i.e. number of bytes) of the base64 encoded data.
73       *
74       * @param encLength length (i.e. number of bytes) of the base64 encoded data
75       * @return size (i.e. number of bytes) of the decoded output
76       */
77      public static long guessDecodedLength(long encLength) {
78  	long decLen = encLength * 3 / 4;
79  	return decLen + 3;
80      }

82      /***
83       * Outputs base64 representation of the specified stream data to an <code>OutputStream</code>.
84       *
85       * @param in     stream data to be encoded
86       * @param writer writer to output the encoded data
87       */
88      public static void encode(InputStream in, Writer writer)
89  	    throws IOException {
90  	// encode stream data in chunks;
91  	// chunksize must be a multiple of 3 in order
92  	// to avoid padding within output
93  	byte[] buffer = new byte[9 * 1024];
94  	int read = 0;
95  	while ((read = in.read(buffer)) > 0) {
96  	    encode(buffer, 0, read, writer);
97  	}
98      }

100     /***
101      * Outputs base64 representation of the specified stream data to an <code>OutputStream</code>.
102      *
103      * @param in  stream data to be encoded
104      * @param out stream where the encoded data should be written to
105      */
106     public static void encode(InputStream in, OutputStream out)
107 	    throws IOException {
108 	Writer writer = new OutputStreamWriter(out, CHARSET);
109 	encode(in, writer);
110     }

112     /***
113      * Outputs base64 representation of the specified data to an <code>OutputStream</code>.
114      *
115      * @param data   data to be encoded
116      * @param off    offset within data at which to start encoding
117      * @param len    length of data to encode
118      * @param writer writer to output the encoded data
119      */
120     public static void encode(byte[] data, int off, int len, Writer writer)
121 	    throws IOException {
122 	if (len == 0) {
123 	    return;
124 	}
125 	if (len < 0 || off >= data.length
126 		|| len + off > data.length) {
127 	    throw new IllegalArgumentException();
128 	}
129 	char[] enc = new char[4];
130 	while (len >= 3) {
131 	    int i = ((data[off] & 0xff) << 16)
132 		    + ((data[off + 1] & 0xff) << 8)
133 		    + (data[off + 2] & 0xff);
134 	    enc[0] = BASE64CHARS[i >> 18];
135 	    enc[1] = BASE64CHARS[(i >> 12) & 0x3f];
136 	    enc[2] = BASE64CHARS[(i >> 6) & 0x3f];
137 	    enc[3] = BASE64CHARS[i & 0x3f];
138 	    writer.write(enc, 0, 4);
139 	    off += 3;
140 	    len -= 3;
141 	}
142 	// add padding if necessary
143 	if (len == 1) {
144 	    int i = data[off] & 0xff;
145 	    enc[0] = BASE64CHARS[i >> 2];
146 	    enc[1] = BASE64CHARS[(i << 4) & 0x3f];
147 	    enc[2] = BASE64PAD;
148 	    enc[3] = BASE64PAD;
149 	    writer.write(enc, 0, 4);
150 	} else if (len == 2) {
151 	    int i = ((data[off] & 0xff) << 8) + (data[off + 1] & 0xff);
152 	    enc[0] = BASE64CHARS[i >> 10];
153 	    enc[1] = BASE64CHARS[(i >> 4) & 0x3f];
154 	    enc[2] = BASE64CHARS[(i << 2) & 0x3f];
155 	    enc[3] = BASE64PAD;
156 	    writer.write(enc, 0, 4);
157 	}
158     }

160     /***
161      * Decode base64 encoded data.
162      *
163      * @param reader reader for the base64 encoded data to be decoded
164      * @param out    stream where the decoded data should be written to
165      */
166     public static void decode(Reader reader, OutputStream out) throws IOException {
167 	char[] chunk = new char[8192];
168 	int read;
169 	while ((read = reader.read(chunk)) > -1) {
170 	    decode(chunk, 0, read, out);
171 	}
172     }

174     /***
175      * Decode base64 encoded data. The data read from the inputstream is
176      * assumed to be of charset "US-ASCII".
177      *
178      * @param in  inputstream of the base64 encoded data to be decoded
179      * @param out stream where the decoded data should be written to
180      */
181     public static void decode(InputStream in, OutputStream out) throws IOException {
182 	decode(new InputStreamReader(in, CHARSET), out);
183     }

185     /***

187      *

191     public static void decode(String data, OutputStream out) throws IOException {

206     /***
207      * Decode base64 encoded data.
208      *
209      * @param chars the base64 encoded data to be decoded
210      * @param off   offset within data at which to start decoding
211      * @param len   length of data to decode
212      * @param out   stream where the decoded data should be written to
213      */
214     public static void decode(char[] chars, int off, int len, OutputStream out) throws IOException {
215 	if (len == 0) {
216 	    return;
217 	}
218 	if (len < 0 || off >= chars.length
219 		|| len + off > chars.length) {
220 	    throw new IllegalArgumentException();
221 	}
222 	char[] chunk = new char[4];
223 	byte[] dec = new byte[3];
224 	int posChunk = 0;
225 	// decode in chunks of 4 characters
226 	for (int i = off; i < (off + len); i++) {
227 	    char c = chars[i];
228 	    if (c < DECODETABLE.length && DECODETABLE[c] != 0x7f
229 		    || c == BASE64PAD) {
230 		chunk[posChunk++] = c;
231 		if (posChunk == chunk.length) {
232 		    int b0 = DECODETABLE[chunk[0]];
233 		    int b1 = DECODETABLE[chunk[1]];
234 		    int b2 = DECODETABLE[chunk[2]];
235 		    int b3 = DECODETABLE[chunk[3]];
236 		    if (chunk[3] == BASE64PAD && chunk[2] == BASE64PAD) {
237 			dec[0] = (byte) (b0 << 2 & 0xfc | b1 >> 4 & 0x3);
238 			out.write(dec, 0, 1);
239 		    } else if (chunk[3] == BASE64PAD) {
240 			dec[0] = (byte) (b0 << 2 & 0xfc | b1 >> 4 & 0x3);
241 			dec[1] = (byte) (b1 << 4 & 0xf0 | b2 >> 2 & 0xf);
242 			out.write(dec, 0, 2);
243 		    } else {
244 			dec[0] = (byte) (b0 << 2 & 0xfc | b1 >> 4 & 0x3);
245 			dec[1] = (byte) (b1 << 4 & 0xf0 | b2 >> 2 & 0xf);
246 			dec[2] = (byte) (b2 << 6 & 0xc0 | b3 & 0x3f);
247 			out.write(dec, 0, 3);
248 		    }
249 		    posChunk = 0;
250 		}
251 	    } else {
252 		throw new IllegalArgumentException("specified data is not base64 encoded");
253 	    }
254 	}
255     }
