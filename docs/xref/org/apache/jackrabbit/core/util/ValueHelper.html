53      /***
54       * @param srcValue
55       * @param targetType
56       * @return
57       * @throws ValueFormatException
58       * @throws IllegalStateException
59       * @throws IllegalArgumentException
60       */
61      public static Value convert(Value srcValue, int targetType)
62  	    throws ValueFormatException, IllegalStateException, IllegalArgumentException {
63  	Value val = null;
64  	int srcType = srcValue.getType();
65  
66  	if (srcType == targetType) {
67  	    // no conversion needed, return original value
68  	    return srcValue;
69  	}
70  
71  	switch (targetType) {
72  	    case PropertyType.STRING:
73  		// convert to STRING
74  		try {
75  		    val = new StringValue(srcValue.getString());
76  		} catch (RepositoryException re) {
77  		    throw new ValueFormatException("conversion failed: "
78  			    + PropertyType.nameFromValue(srcType) + " to "
79  			    + PropertyType.nameFromValue(targetType), re);
80  		}
81  		break;
82  
83  	    case PropertyType.BINARY:
84  		// convert to BINARY
85  		try {
86  		    val = new BinaryValue(srcValue.getStream());
87  		} catch (RepositoryException re) {
88  		    throw new ValueFormatException("conversion failed: "
89  			    + PropertyType.nameFromValue(srcType) + " to "
90  			    + PropertyType.nameFromValue(targetType), re);
91  		}
92  		break;
93  
94  	    case PropertyType.BOOLEAN:
95  		// convert to BOOLEAN
96  		try {
97  		    val = new BooleanValue(srcValue.getBoolean());
98  		} catch (RepositoryException re) {
99  		    throw new ValueFormatException("conversion failed: "
100 			    + PropertyType.nameFromValue(srcType) + " to "
101 			    + PropertyType.nameFromValue(targetType), re);
102 		}
103 		break;
104 
105 	    case PropertyType.DATE:
106 		// convert to DATE
107 		try {
108 		    val = new DateValue(srcValue.getDate());
109 		} catch (RepositoryException re) {
110 		    throw new ValueFormatException("conversion failed: "
111 			    + PropertyType.nameFromValue(srcType) + " to "
112 			    + PropertyType.nameFromValue(targetType), re);
113 		}
114 		break;
115 
116 	    case PropertyType.DOUBLE:
117 		// convert to DOUBLE
118 		try {
119 		    val = new DoubleValue(srcValue.getDouble());
120 		} catch (RepositoryException re) {
121 		    throw new ValueFormatException("conversion failed: "
122 			    + PropertyType.nameFromValue(srcType) + " to "
123 			    + PropertyType.nameFromValue(targetType), re);
124 		}
125 		break;
126 
127 	    case PropertyType.LONG:
128 		// convert to LONG
129 		try {
130 		    val = new LongValue(srcValue.getLong());
131 		} catch (RepositoryException re) {
132 		    throw new ValueFormatException("conversion failed: "
133 			    + PropertyType.nameFromValue(srcType) + " to "
134 			    + PropertyType.nameFromValue(targetType), re);
135 		}
136 		break;
137 
138 	    case PropertyType.PATH:
139 		// convert to PATH
140 		switch (srcType) {
141 		    case PropertyType.PATH:
142 			// no conversion needed, return original value
143 			// (redundant code, just here for the sake of clarity)
144 			return srcValue;
145 
146 		    case PropertyType.BINARY:
147 		    case PropertyType.STRING:
148 		    case PropertyType.NAME:	// a name is always also a relative path
149 			// try conversion via string
150 			String path;
151 			try {
152 			    // get string value
153 			    path = srcValue.getString();
154 			} catch (RepositoryException re) {
155 			    // should never happen
156 			    throw new ValueFormatException("failed to convert source value to PATH value", re);
157 			}
158 			try {
159 			    // check path format
160 			    Path.checkFormat(path);
161 			} catch (MalformedPathException mpe) {
162 			    throw new ValueFormatException("source value " + path + " does not represent a valid path", mpe);
163 			}
164 			val = PathValue.valueOf(path);
165 			break;
166 
167 		    case PropertyType.BOOLEAN:
168 		    case PropertyType.DATE:
169 		    case PropertyType.DOUBLE:
170 		    case PropertyType.LONG:
171 		    case PropertyType.REFERENCE:
172 			throw new ValueFormatException("conversion failed: "
173 				+ PropertyType.nameFromValue(srcType) + " to "
174 				+ PropertyType.nameFromValue(targetType));
175 
176 		    default:
177 			throw new IllegalArgumentException("not a valid type constant: " + srcType);
178 		}
179 		break;
180 
181 	    case PropertyType.NAME:
182 		// convert to NAME
183 		switch (srcType) {
184 		    case PropertyType.NAME:
185 			// no conversion needed, return original value
186 			// (redundant code, just here for the sake of clarity)
187 			return srcValue;
188 			
189 		    case PropertyType.BINARY:
190 		    case PropertyType.STRING:
191 		    case PropertyType.PATH:	// path might be a name (relative, onle element long path) 
192 			// try conversion via string
193 			String name;
194 			try {
195 			    // get string value
196 			    name = srcValue.getString();
197 			} catch (RepositoryException re) {
198 			    // should never happen
199 			    throw new ValueFormatException("failed to convert source value to NAME value", re);
200 			}
201 			try {
202 			    // check name format
203 			    QName.checkFormat(name);
204 			} catch (IllegalNameException ine) {
205 			    throw new ValueFormatException("source value " + name + " does not represent a valid name", ine);
206 			}
207 			val = NameValue.valueOf(name);
208 			break;
209 
210 		    case PropertyType.BOOLEAN:
211 		    case PropertyType.DATE:
212 		    case PropertyType.DOUBLE:
213 		    case PropertyType.LONG:
214 		    case PropertyType.REFERENCE:
215 			throw new ValueFormatException("conversion failed: "
216 				+ PropertyType.nameFromValue(srcType) + " to "
217 				+ PropertyType.nameFromValue(targetType));
218 
219 		    default:
220 			throw new IllegalArgumentException("not a valid type constant: " + srcType);
221 		}
222 		break;
223 
224 	    case PropertyType.REFERENCE:
225 		// convert to REFERENCE
226 		switch (srcType) {
227 		    case PropertyType.REFERENCE:
228 			// no conversion needed, return original value
229 			// (redundant code, just here for the sake of clarity)
230 			return srcValue;
231 
232 		    case PropertyType.BINARY:
233 		    case PropertyType.STRING:
234 			// try conversion via string
235 			String uuid;
236 			try {
237 			    // get string value
238 			    uuid = srcValue.getString();
239 			} catch (RepositoryException re) {
240 			    // should never happen
241 			    throw new ValueFormatException("failed to convert source value to REFERENCE value", re);
242 			}
243 			val = ReferenceValue.valueOf(uuid);
244 			break;
245 
246 		    case PropertyType.BOOLEAN:
247 		    case PropertyType.DATE:
248 		    case PropertyType.DOUBLE:
249 		    case PropertyType.LONG:
250 		    case PropertyType.PATH:
251 		    case PropertyType.NAME:
252 			throw new ValueFormatException("conversion failed: "
253 				+ PropertyType.nameFromValue(srcType) + " to "
254 				+ PropertyType.nameFromValue(targetType));
255 
256 		    default:
257 			throw new IllegalArgumentException("not a valid type constant: " + srcType);
258 		}
259 		break;
260 
261 	    default:
262 		throw new IllegalArgumentException("not a valid type constant: " + targetType);
263 	}
264 
265 	return val;
266     }

268     /***
269      * @param srcVal
270      * @return
271      * @throws IllegalStateException
272      */
273     public static Value copy(Value srcVal) throws IllegalStateException {
274 	Value newVal = null;
275 	try {
276 	    switch (srcVal.getType()) {
277 		case PropertyType.BINARY:
278 		    newVal = new BinaryValue(srcVal.getStream());
279 		    break;
280 
281 		case PropertyType.BOOLEAN:
282 		    newVal = new BooleanValue(srcVal.getBoolean());
283 		    break;
284 
285 		case PropertyType.DATE:
286 		    newVal = new DateValue(srcVal.getDate());
287 		    break;
288 
289 		case PropertyType.DOUBLE:
290 		    newVal = new DoubleValue(srcVal.getDouble());
291 		    break;
292 
293 		case PropertyType.LONG:
294 		    newVal = new LongValue(srcVal.getLong());
295 		    break;
296 
297 		case PropertyType.PATH:
298 		    newVal = PathValue.valueOf(srcVal.getString());
299 		    break;
300 
301 		case PropertyType.NAME:
302 		    newVal = NameValue.valueOf(srcVal.getString());
303 		    break;
304 
305 		case PropertyType.REFERENCE:
306 		    newVal = ReferenceValue.valueOf(srcVal.getString());
307 		    break;
308 
309 		case PropertyType.STRING:
310 		    newVal = new StringValue(srcVal.getString());
311 		    break;
312 	    }
313 	} catch (RepositoryException re) {
314 	    // should never get here
315 	}
316 	return newVal;
317     }

319     /***
320      * @param srcVal
321      * @return
322      * @throws IllegalStateException
323      */
324     public static Value[] copy(Value[] srcVal) throws IllegalStateException {
325 	if (srcVal == null) {
326 	    return null;
327 	}
328 	Value[] newVal = new Value[srcVal.length];
329 	for (int i = 0; i < srcVal.length; i++) {
330 	    newVal[i] = copy(srcVal[i]);
331 	}
332 	return newVal;
333     }
