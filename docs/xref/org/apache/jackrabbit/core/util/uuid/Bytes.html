30  public final class Bytes {

38      /***
39       * Appends two bytes array into one.
40       *
41       * @param a A byte[].
42       * @param b A byte[].
43       * @return A byte[].
44       */
45      public static byte[] append(byte[] a, byte[] b) {
46          byte[] z = new byte[a.length + b.length];
47          System.arraycopy(a, 0, z, 0, a.length);
48          System.arraycopy(b, 0, z, a.length, b.length);
49          return z;
50      }

52      /***
53       * Returns a 8-byte array built from a long.
54       *
55       * @param n The number to convert.
56       * @return A byte[].
57       */
58      public static byte[] toBytes(long n) {
59          return toBytes(n, new byte[8]);
60      }
61  
62      /***
63       * Build a 8-byte array from a long.  No check is performed on the
64       * array length.
65       *
66       * @param n The number to convert.
67       * @param b The array to fill.
68       * @return A byte[].
69       */
70      public static byte[] toBytes(long n, byte[] b) {
71          b[7] = (byte) (n);
72          n >>>= 8;
73          b[6] = (byte) (n);
74          n >>>= 8;
75          b[5] = (byte) (n);
76          n >>>= 8;
77          b[4] = (byte) (n);
78          n >>>= 8;
79          b[3] = (byte) (n);
80          n >>>= 8;
81          b[2] = (byte) (n);
82          n >>>= 8;
83          b[1] = (byte) (n);
84          n >>>= 8;
85          b[0] = (byte) (n);
86  
87          return b;
88      }

90      /***
91       * Build a long from first 8 bytes of the array.
92       *
93       * @param b The byte[] to convert.
94       * @return A long.
95       */
96      public static long toLong(byte[] b) {
97          return ((((long) b[7]) & 0xFF)
98                  + ((((long) b[6]) & 0xFF) << 8)
99                  + ((((long) b[5]) & 0xFF) << 16)
100                 + ((((long) b[4]) & 0xFF) << 24)
101                 + ((((long) b[3]) & 0xFF) << 32)
102                 + ((((long) b[2]) & 0xFF) << 40)
103                 + ((((long) b[1]) & 0xFF) << 48)
104                 + ((((long) b[0]) & 0xFF) << 56));
105     }

107     /***
108     * Compares two byte arrays for equality.
109     *
110     * @param a A byte[].
111     * @param b A byte[].
112     * @return True if the arrays have identical contents.
113     */
114     public static boolean areEqual(byte[] a, byte[] b) {
115         int aLength = a.length;
116         if (aLength != b.length) {
117             return false;
118         }
119 
120         for (int i = 0; i < aLength; i++) {
121             if (a[i] != b[i]) {
122                 return false;
123             }
124         }
125         return true;
126     }

128     /***
129      * <p>Compares two byte arrays as specified by <code>Comparable</code>.
130      *
131      * @param lhs - left hand value in the comparison operation.
132      * @param rhs - right hand value in the comparison operation.
133      * @return  a negative integer, zero, or a positive integer as <code>lhs</code>
134      *  is less than, equal to, or greater than <code>rhs</code>.
135      */
136     public static int compareTo(byte[] lhs, byte[] rhs) {
137         if (lhs == rhs) {
138             return 0;
139         }
140         if (lhs == null) {
141             return -1;
142         }
143         if (rhs == null) {
144             return +1;
145         }
146         if (lhs.length != rhs.length) {
147             return ((lhs.length < rhs.length) ? -1 : +1);
148         }
149         for (int i = 0; i < lhs.length; i++) {
150             if (lhs[i] < rhs[i]) {
151                 return -1;
152             } else if (lhs[i] > rhs[i]) {
153                 return 1;
154             }
155         }
156         return 0;
157     }

159     /***
160      * Build a short from first 2 bytes of the array.
161      *
162      * @param b The byte[] to convert.
163      * @return A short.
164      */
165     public static short toShort(byte[] b) {
166         return  (short) ((b[1] & 0xFF) + ((b[0] & 0xFF) << 8));
167     }
