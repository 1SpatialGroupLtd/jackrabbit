61      public SearchManager(ItemStateProvider stateProvider,
62  			 HierarchyManager hmgr,
63  			 SessionImpl session,
64  			 String indexPath) throws IOException {
65  	this.stateProvider = stateProvider;
66  	this.hmgr = hmgr;
67  	this.session = session;
68  	index = new SearchIndex(indexPath, new StandardAnalyzer());
69  	nsMappings = new NamespaceMappings(new File(indexPath, "ns_mappings.properties"));
70      }

72      public void addNode(NodeState node, String path) throws IOException {
73  	// FIXME rather throw RepositoryException?
74  	log.debug("add node to index: " + path);
75  	Document doc = NodeIndexer.createDocument(node, stateProvider, path, nsMappings);
76  	index.addDocument(doc);
77      }
78  
79      public void updateNode(NodeState node, String path) throws IOException {
80  	log.debug("update index for node: " + path);
81  	deleteNode(path, node.getUUID());
82  	addNode(node, path);
83      }
84  
85      public void deleteNode(String path, String uuid) throws IOException {
86  	log.debug("remove node from index: " + path);
87  	index.removeDocument(new Term(FieldNames.UUID, uuid));
88      }
89  
90      public void close() {
91  	index.close();
92      }

94      public QueryResultImpl execute(ItemManager itemMgr,
95  				   QueryRootNode root,
96  				   SessionImpl session)
97  	    throws RepositoryException {
98  
99  	// build lucene query
100 	Query query = LuceneQueryBuilder.createQuery(root,
101 		session, nsMappings, index.getAnalyzer());
102 
103 	OrderQueryNode orderNode = root.getOrderNode();
104 	// FIXME according to spec this should be descending
105 	// by default. this contrasts to standard sql semantics
106 	// where default is ascending.
107 	boolean ascending = true;
108 	String[] orderProperties = null;
109 	if (orderNode != null) {
110 	    ascending = orderNode.isAscending();
111 	    orderProperties = orderNode.getOrderByProperties();
112 	} else {
113 	    orderProperties = new String[0];
114 	}
115 
116 
117 	List uuids;
118         AccessManagerImpl accessMgr = session.getAccessManager();
119 
120 	// execute it
121 	try {
122 	    Hits result = index.executeQuery(query, orderProperties, ascending);
123 	    uuids = new ArrayList(result.length());
124 	    for (int i = 0; i < result.length(); i++) {
125 		String uuid = result.doc(i).get(FieldNames.UUID);
126 		// check access
127 		if (accessMgr.isGranted(new NodeId(uuid), Permission.READ_ITEM)) {
128 		    uuids.add(uuid);
129 		}
130 	    }
131 	} catch (IOException e) {
132 	    uuids = Collections.EMPTY_LIST;
133 	}
134 
135 	// return QueryResult
136 	return new QueryResultImpl(itemMgr,
137 		(String[])uuids.toArray(new String[uuids.size()]),
138 		root.getSelectProperties());
139     }

143     public void onEvent(EventIterator events) {
144 	Set modified = new HashSet();
145 
146 	// FIXME optimize operations on index.
147 	// only one cycle of document removes and document adds
148 
149 	while (events.hasNext()) {
150 	    try {
151 		EventImpl e = (EventImpl)events.nextEvent();
152 		long type = e.getType();
153 		if (type == EventType.CHILD_NODE_ADDED) {
154 
155 		    Path path = Path.create(
156 			    e.getNodePath() + ((e.getNodePath().length() > 1) ? "/" : "") + e.getChildName(),
157 			    session.getNamespaceResolver(),
158 			    true);
159 
160 		    path = getIndexlessPath(path);
161 
162 		    ItemId id = new NodeId(e.getChildUUID());
163 		    addNode((NodeState)stateProvider.getItemState(id),
164 			    path.toJCRPath(nsMappings));
165 
166 		} else if (type == EventType.CHILD_NODE_REMOVED) {
167 
168 		    Path path = Path.create(e.getNodePath() + ((e.getNodePath().length() > 1) ? "/" : "") + e.getChildName(),
169 			    session.getNamespaceResolver(),
170 			    true);
171 		    deleteNode(path.toJCRPath(nsMappings), e.getChildUUID());
172 
173 		} else if (type == EventType.PROPERTY_ADDED
174 			|| type == EventType.PROPERTY_CHANGED
175 			|| type == EventType.PROPERTY_REMOVED) {
176 
177 		    Path path = Path.create(
178 			    e.getNodePath(),
179 			    session.getNamespaceResolver(),
180 			    true);
181 		    modified.add(path);
182 		}
183 	    } catch (MalformedPathException e) {
184 		log.error("error indexing node.", e);
185 	    } catch (NoPrefixDeclaredException e) {
186 		log.error("error indexing node.", e);
187 	    } catch (ItemStateException e) {
188 		log.error("error indexing node.", e);
189 	    } catch (RepositoryException e) {
190 		log.error("error indexing node.", e);
191 	    } catch (IOException e) {
192 		log.error("error indexing node.", e);
193 	    }
194 	}
195 
196 	for (Iterator it = modified.iterator(); it.hasNext(); ) {
197 	    try {
198 		Path path = (Path)it.next();
199 		ItemId id = hmgr.resolvePath(path);
200 		path = getIndexlessPath(path);
201 		updateNode((NodeState)stateProvider.getItemState(id),
202 			path.toJCRPath(nsMappings));
203 	    } catch (NoPrefixDeclaredException e) {
204 		log.error("error indexing node.", e);
205 	    } catch (ItemStateException e) {
206 		log.error("error indexing node.", e);
207 	    } catch (RepositoryException e) {
208 		log.error("error indexing node.", e);
209 	    } catch (IOException e) {
210 		log.error("error indexing node.", e);
211 	    }
212 	}
213     }

217     /***
218      * Returns a <code>Path</code>, which contains the same sequence of
219      * path elements as <code>p</code>, but has cut off any existing indexes
220      * on the path elements.
221      * @param p the source path, possibly containing indexed path elements.
222      * @return a <code>Path</code> without indexed path elements.
223      */
224     private Path getIndexlessPath(Path p) {
225 	boolean hasIndexes = false;
226 	Path.PathElement[] elements = p.getElements();
227 	for (int i = 0; i < elements.length && !hasIndexes; i++) {
228 	    hasIndexes = (elements[i].getIndex() > 0);
229 	}
230 
231 	if (hasIndexes) {
232 	    // create Path without indexes
233 	    Path.PathBuilder builder = new Path.PathBuilder();
234 	    builder.addRoot();
235 	    for (int i = 1; i < elements.length; i++) {
236 		builder.addLast(elements[i].getName());
237 	    }
238 	    try {
239 		return builder.getPath();
240 	    } catch (MalformedPathException e) {
241 		// will never happen, because Path p is always valid
242 		log.error("internal error: malformed path.", e);
243 	    }
244 	}
245 	// return original path if it does not contain indexed path elements
246 	return p;
247     }
