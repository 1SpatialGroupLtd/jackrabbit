55      public static final QName NODENAME_XMLTEXT =

110     /***
111      * @see AbstractSAXEventGenerator#leavingProperties(NodeState, QName, int)
112      */
113     protected void leavingProperties(NodeState state, QName name, int level)
114 	    throws RepositoryException, SAXException {
115 	if (name.equals(NODENAME_XMLTEXT)) {
116 	    // the node represents xml character data
117 	    Iterator iter = props.iterator();
118 	    while (iter.hasNext()) {
119 		PropertyState propState = (PropertyState) iter.next();
120 		QName propName = propState.getName();
121 		if (propName.equals(PROPNAME_XMLCHARACTERS)) {
122 		    InternalValue val = propState.getValues()[0];
123 		    char[] chars = val.toJCRValue(nsReg).getString().toCharArray();
124 		    contentHandler.characters(chars, 0, chars.length);
125 		}
126 	    }
127 	} else {
128 	    // regalur node
129 
130 	    // element name
131 	    String elemName;
132 	    try {
133 		if (state.getParentUUIDs().size() == 0) {
134 		    // root node needs a name
135 		    elemName = NODENAME_ROOT.toJCRName(nsReg);
136 		} else {
137 		    elemName = name.toJCRName(nsReg);
138 		}
139 	    } catch (NoPrefixDeclaredException npde) {
140 		// should never get here...
141 		String msg = "internal error: encountered unregistered namespace";
142 		log.error(msg, npde);
143 		throw new RepositoryException(msg, npde);
144 	    }
145 	    if (!XMLChar.isValidName(elemName)) {
146 		throw new InvalidSerializedDataException("the node name is not a valid xml element name: " + elemName);
147 	    }
148 
149 	    // attributes (properties)
150 	    AttributesImpl attrs = new AttributesImpl();
151 	    Iterator iter = props.iterator();
152 	    while (iter.hasNext()) {
153 		PropertyState propState = (PropertyState) iter.next();
154 		QName propName = propState.getName();
155 		// attribute name
156 		String attrName;
157 		try {
158 		    attrName = propName.toJCRName(nsReg);
159 		} catch (NoPrefixDeclaredException npde) {
160 		    // should never get here...
161 		    String msg = "internal error: encountered unregistered namespace";
162 		    log.error(msg, npde);
163 		    throw new RepositoryException(msg, npde);
164 		}
165 		if (!XMLChar.isValidName(attrName)) {
166 		    throw new InvalidSerializedDataException("the property name is not a valid xml attribute name: " + attrName);
167 		}
168 		// attribute value
169 		String attrValue;
170 		int type = propState.getType();
171 		// process property value (guaranteed to be single-valued and non-null)
172 		InternalValue val = propState.getValues()[0];
173 		if (type == PropertyType.BINARY) {
174 		    if (binaryAsLink) {
175 			try {
176 			    attrValue = hierMgr.getPath(new PropertyId(propState.getParentUUID(), propState.getName())).toJCRPath(nsReg);
177 			} catch (NoPrefixDeclaredException npde) {
178 			    // should never get here...
179 			    String msg = "internal error: encountered unregistered namespace";
180 			    log.error(msg, npde);
181 			    throw new RepositoryException(msg, npde);
182 			}
183 		    } else {
184 			// binary data, base64 encoding required
185 			BLOBFileValue blob = (BLOBFileValue) val.internalValue();
186 			InputStream in = blob.getStream();
187 			StringWriter writer = new StringWriter();
188 			try {
189 			    Base64.encode(in, writer);
190 			    in.close();
191 			    writer.close();
192 			} catch (IOException ioe) {
193 			    // check if the exception wraps a SAXException
194 			    Throwable t = ioe.getCause();
195 			    if (t != null && t instanceof SAXException) {
196 				throw (SAXException) t;
197 			    } else {
198 				throw new SAXException(ioe);
199 			    }
200 			}
201 			attrValue = writer.toString();
202 		    }
203 		} else {
204 		    attrValue = val.toJCRValue(nsReg).getString();
205 		}
206 		attrs.addAttribute(propName.getNamespaceURI(), propName.getLocalName(), attrName, CDATA_TYPE, attrValue);
207 	    }
208 	    // start element (node)
209 	    contentHandler.startElement(name.getNamespaceURI(), name.getLocalName(), elemName, attrs);
210 	}
211     }

213     /***
214      * @see AbstractSAXEventGenerator#leaving(NodeState, QName, int)
215      */
216     protected void leaving(NodeState state, QName name, int level)
217 	    throws RepositoryException, SAXException {
218 	if (name.equals(NODENAME_XMLTEXT)) {
219 	    // the node represents xml character data
220 	    // (already processed in leavingProperties(NodeState, QName, int)
221 	    return;
222 	}
223 	// element name
224 	String elemName;
225 	try {
226 	    if (state.getParentUUIDs().size() == 0) {
227 		// root node needs a name
228 		elemName = NODENAME_ROOT.toJCRName(nsReg);
229 	    } else {
230 		elemName = name.toJCRName(nsReg);
231 	    }
232 	} catch (NoPrefixDeclaredException npde) {
233 	    // should never get here...
234 	    String msg = "internal error: encountered unregistered namespace";
235 	    log.error(msg, npde);
236 	    throw new RepositoryException(msg, npde);
237 	}
238 
239 	// end element (node)
240 	contentHandler.endElement(name.getNamespaceURI(), name.getLocalName(), elemName);
241     }

243     /***
244      * @see AbstractSAXEventGenerator#entering(PropertyState, int)
245      */
246     protected void entering(PropertyState state, int level)
247 	    throws RepositoryException, SAXException {
248 	// @todo should properties declared in the 'jcr:' namespace be skipped in the document view export?
249 /*
250 	// skip 'system' properties (e.g. jcr:primaryType, jcr:mixinTypes, jcr:uuid)
251 	if (NamespaceRegistryImpl.NS_JCR_URI.equals(state.getName().getNamespaceURI())) {
252 	    return;
253 	}
254 */
255 	// collect property state in temporary list (will be processed in leavingProperties(NodeState, QName, int)
256 	InternalValue[] values = state.getValues();
257 	if (values != null && values.length > 0) {
258 	    if (values.length != 1) {
259 		throw new InvalidSerializedDataException("unable to serialize multi-valued property in document view: " + state.getName().getLocalName());
260 	    }
261 	    if (values[0] != null) {
262 		props.add(state);
263 	    }
264 	}
265     }
