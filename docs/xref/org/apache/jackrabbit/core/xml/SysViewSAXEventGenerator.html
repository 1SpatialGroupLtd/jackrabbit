89      /***
90       * @see AbstractSAXEventGenerator#entering(NodeState, QName, int)
91       */
92      protected void entering(NodeState state, QName name, int level)
93  	    throws RepositoryException, SAXException {
94  	AttributesImpl attrs = new AttributesImpl();
95  	// name attribute
96  	String nodeName;
97  	try {
98  	    if (state.getParentUUIDs().size() == 0) {
99  		// use dummy name for root node
100 		nodeName = NODENAME_ROOT.toJCRName(nsReg);
101 	    } else {
102 		nodeName = name.toJCRName(nsReg);
103 	    }
104 	} catch (NoPrefixDeclaredException npde) {
105 	    // should never get here...
106 	    String msg = "internal error: encountered unregistered namespace";
107 	    log.error(msg, npde);
108 	    throw new RepositoryException(msg, npde);
109 	}
110 	attrs.addAttribute(NS_SV_URI, NAME_ATTRIBUTE, NS_SV_PREFIX + ":" + NAME_ATTRIBUTE, CDATA_TYPE, nodeName);
111 	// start node element
112 	contentHandler.startElement(NS_SV_URI, NODE_ELEMENT, NS_SV_PREFIX + ":" + NODE_ELEMENT, attrs);
113     }

140     /***
141      * @see AbstractSAXEventGenerator#entering(PropertyState, int)
142      */
143     protected void entering(PropertyState state, int level)
144 	    throws RepositoryException, SAXException {
145 	String name;
146 	try {
147 	    name = state.getName().toJCRName(nsReg);
148 	} catch (NoPrefixDeclaredException npde) {
149 	    // should never get here...
150 	    String msg = "internal error: encountered unregistered namespace";
151 	    log.error(msg, npde);
152 	    throw new RepositoryException(msg, npde);
153 	}
154 	AttributesImpl attrs = new AttributesImpl();
155 	// name attribute
156 	attrs.addAttribute(NS_SV_URI, NAME_ATTRIBUTE, NS_SV_PREFIX + ":" + NAME_ATTRIBUTE, CDATA_TYPE, name);
157 	// type attribute
158 	int type = state.getType();
159 	String typeName;
160 	try {
161 	    typeName = PropertyType.nameFromValue(type);
162 	} catch (IllegalArgumentException iae) {
163 	    // should never be getting here
164 	    throw new RepositoryException("unexpected property-type ordinal: " + type, iae);
165 	}
166 	if (type == PropertyType.BINARY && binaryAsLink) {
167 	    typeName = PropertyType.TYPENAME_PATH;
168 	}
169 	attrs.addAttribute(NS_SV_URI, TYPE_ATTRIBUTE, NS_SV_PREFIX + ":" + TYPE_ATTRIBUTE, ENUMERATION_TYPE, typeName);
170 
171 	// start property element
172 	contentHandler.startElement(NS_SV_URI, PROPERTY_ELEMENT, NS_SV_PREFIX + ":" + PROPERTY_ELEMENT, attrs);
173 
174 	// values
175 	InternalValue[] values = state.getValues();
176 	if (values != null && values.length > 0) {
177 	    for (int i = 0; i < values.length; i++) {
178 		if (values[i] != null) {
179 		    // start value element
180 		    contentHandler.startElement(NS_SV_URI, VALUE_ELEMENT, NS_SV_PREFIX + ":" + VALUE_ELEMENT, new AttributesImpl());
181 
182 		    // characters
183 		    if (type == PropertyType.BINARY) {
184 			if (binaryAsLink) {
185 			    String path;
186 			    try {
187 				path = hierMgr.getPath(new PropertyId(state.getParentUUID(), state.getName())).toJCRPath(nsReg);
188 			    } catch (NoPrefixDeclaredException npde) {
189 				// should never get here...
190 				String msg = "internal error: encountered unregistered namespace";
191 				log.error(msg, npde);
192 				throw new RepositoryException(msg, npde);
193 			    }
194 			    char[] chars = path.toCharArray();
195 			    contentHandler.characters(chars, 0, chars.length);
196 			} else {
197 			    // binary data, base64 encoding required
198 			    BLOBFileValue blob = (BLOBFileValue) values[i].internalValue();
199 			    InputStream in = blob.getStream();
200 			    Writer writer = new Writer() {
201 				public void close() /*throws IOException*/ {
202 				}
203 
204 				public void flush() /*throws IOException*/ {
205 				}
206 
207 				public void write(char[] cbuf, int off, int len) throws IOException {
208 				    try {
209 					contentHandler.characters(cbuf, off, len);
210 				    } catch (SAXException se) {
211 					throw new IOException(se.toString());
212 				    }
213 				}
214 			    };
215 			    try {
216 				Base64.encode(in, writer);
217 				in.close();
218 				writer.close();
219 			    } catch (IOException ioe) {
220 				// check if the exception wraps a SAXException
221 				Throwable t = ioe.getCause();
222 				if (t != null && t instanceof SAXException) {
223 				    throw (SAXException) t;
224 				} else {
225 				    throw new SAXException(ioe);
226 				}
227 			    }
228 			}
229 		    } else {
230 			char[] chars = values[i].toJCRValue(nsReg).getString().toCharArray();
231 			contentHandler.characters(chars, 0, chars.length);
232 		    }
233 
234 		    // end value element
235 		    contentHandler.endElement(NS_SV_URI, VALUE_ELEMENT, NS_SV_PREFIX + ":" + VALUE_ELEMENT);
236 		}
237 	    }
238 	}
239     }
