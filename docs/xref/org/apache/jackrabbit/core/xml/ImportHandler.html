26  /***
27   * An <code>ImportHandler</code> instance can be used to import serialized
28   * data in System View XML or Document View XML. The actual task of importing
29   * is delegated to specialized <code>ContentHandler</code>s (i.e.
30   * <code>SysViewImportHandler</code> and <code>DocViewImportHandler</code>).
31   *
32   * @author Stefan Guggisberg
33   * @version $Revision: 1.3 $, $Date: 2004/08/27 15:48:21 $
34   */
35  public class ImportHandler extends DefaultHandler {
36  
37      private static Logger log = Logger.getLogger(ImportHandler.class);
38  
39      protected Locator locator;
40      protected ContentHandler targetHandler;
41  
42      protected SessionImpl session;
43      protected NodeImpl importTargetNode;
44      protected NamespaceRegistryImpl nsReg;
45      protected boolean systemViewXML;
46      protected boolean initialized;
47  
48      public ImportHandler(NodeImpl importTargetNode, NamespaceRegistryImpl nsReg, SessionImpl session) {
49  	this.importTargetNode = importTargetNode;
50  	this.session = session;
51  	this.nsReg = nsReg;
52      }

98      /***
99       * @see ContentHandler#startPrefixMapping(String, String)
100      */
101     public void startPrefixMapping(String prefix, String uri) throws SAXException {
102 	try {
103 	    String oldPrefix = session.getNamespacePrefix(uri);
104 	    // namespace is already registered; check prefix
105 	    if (!oldPrefix.equals(prefix)) {
106 		// namespace is mapped to different prefix;
107 		// try to remap it to given prefix
108 		try {
109 		    session.setNamespacePrefix(prefix, uri);
110 		} catch (RepositoryException re) {
111 		    throw new SAXException("failed to remap namespace " + uri + " to prefix " + prefix, re);
112 		}
113 	    }
114 	} catch (NamespaceException nse) {
115 	    // namespace is not yet registered, try to register it
116 	    try {
117 		nsReg.registerNamespace(prefix, uri);
118 	    } catch (RepositoryException re) {
119 		throw new SAXException("failed to register namespace " + uri + " with prefix " + prefix, re);
120 	    }
121 	}
122     }

124     /***
125      * @see ContentHandler#startElement(String, String, String, Attributes)
126      */
127     public void startElement(String namespaceURI, String localName, String qName, Attributes atts) throws SAXException {
128 	if (!initialized) {
129 	    // the namespace of the first element determines the type of XML
130 	    // (system view/document view)
131 	    String nsURI;
132 	    if (qName == null || "".equals(qName)) {
133 		nsURI = namespaceURI;
134 	    } else {
135 		try {
136 		    nsURI = QName.fromJCRName(qName, session.getNamespaceResolver()).getNamespaceURI();
137 		} catch (BaseException e) {
138 		    // should never happen...
139 		    String msg = "internal error: failed to parse/resolve element name " + qName;
140 		    log.error(msg, e);
141 		    throw new SAXException(msg, e);
142 		}
143 	    }
144 	    systemViewXML = NamespaceRegistryImpl.NS_SV_URI.equals(nsURI);
145 
146 	    if (systemViewXML) {
147 		targetHandler = new SysViewImportHandler(importTargetNode, session);
148 	    } else {
149 		targetHandler = new DocViewImportHandler(importTargetNode, session);
150 	    }
151 	    targetHandler.startDocument();
152 	    initialized = true;
153 	}
154 
155 	// delegate to target handler
156 	targetHandler.startElement(namespaceURI, localName, qName, atts);
157     }
