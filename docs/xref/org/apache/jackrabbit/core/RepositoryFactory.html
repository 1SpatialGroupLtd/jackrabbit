36  /***
37   * A <code>RepositoryFactory</code>.
38   * A singleton class. On creation, it reads in the default config file "config.xml" which describes
39   * the available repositories. A sample config.xml (including DTD):
40   * <pre>
41   *  &lt;?xml version="1.0" encoding="ISO-8859-1"?&gt;
42   *  &lt;!DOCTYPE Repositories [
43   *    &lt;!ELEMENT Repositories (Repository+)&gt;
44   *    &lt;!ELEMENT Repository (RepositoryStore,StableWorkspace+)&gt;
45   *    &lt;!ATTLIST Repository
46   *      name CDATA #REQUIRED&gt;
47   *    &lt;!ELEMENT RepositoryStore (FileSystem)&gt;
48   *    &lt;!ELEMENT StableWorkspace (PersistenceManager,WorkspaceStore,BLOBStore?,DynamicWorkspace*)&gt;
49   *    &lt;!ATTLIST StableWorkspace
50   *      name CDATA #REQUIRED&gt;
51   *    &lt;!ELEMENT PersistenceManager (EMPTY)&gt;
52   *    &lt;!ATTLIST PersistenceManager
53   *      class CDATA #REQUIRED&gt;
54   *    &lt;!ELEMENT WorkspaceStore (FileSystem)&gt;
55   *    &lt;!ELEMENT BLOBStore (FileSystem)&gt;
56   *    &lt;!ELEMENT DynamicWorkspace (PersistenceManager,WorkspaceStore,BLOBStore?)&gt;
57   *    &lt;!ATTLIST DynamicWorkspace
58   *      name CDATA #REQUIRED&gt;
59   *    &lt;!ELEMENT FileSystem (param*)&gt;
60   *    &lt;!ATTLIST FileSystem
61   *      class CDATA #REQUIRED&gt;
62   *    &lt;!ELEMENT param EMPTY&gt;
63   *    &lt;!ATTLIST param
64   *      name CDATA #REQUIRED
65   *      value CDATA #REQUIRED&gt;]&gt;
66   *  &lt;Repositories&gt;
67   *    &lt;!-- sample configuration for a repository that stores its state in a local file system --&gt;
68   *    &lt;Repository name="localfs"&gt;
69   *      &lt;RepositoryStore&gt;
70   *        &lt;FileSystem class="org.apache.jackrabbit.core.fs.local.LocalFileSystem"&gt;
71   *          &lt;param name="path" value="${factory.home}/localfs/repository"/&gt;
72   *        &lt;/FileSystem&gt;
73   *      &lt;/RepositoryStore&gt;
74   *      &lt;!-- main workspace --&gt;
75   *      &lt;StableWorkspace name="default"&gt;
76   *        &lt;PersistenceManager class="org.apache.jackrabbit.core.state.xml.XMLPersistenceManager"/&gt;
77   *        &lt;WorkspaceStore&gt;
78   *          &lt;FileSystem class="org.apache.jackrabbit.core.fs.local.LocalFileSystem"&gt;
79   *            &lt;param name="path" value="${factory.home}/localfs/workspaces/default/data"/&gt;
80   *          &lt;/FileSystem&gt;
81   *        &lt;/WorkspaceStore&gt;
82   *        &lt;BLOBStore&gt;
83   *          &lt;FileSystem class="org.apache.jackrabbit.core.fs.local.LocalFileSystem"&gt;
84   *            &lt;param name="path" value="${factory.home}/localfs/workspaces/default/blobs"/&gt;
85   *          &lt;/FileSystem&gt;
86   *        &lt;/BLOBStore&gt;
87   *        &lt;!-- dynamic workspace based on main workspace --&gt;
88   *        &lt;DynamicWorkspace name="test1"&gt;
89   *          &lt;PersistenceManager class="org.apache.jackrabbit.core.state.xml.XMLPersistenceManager"/&gt;
90   *          &lt;WorkspaceStore&gt;
91   *            &lt;FileSystem class="org.apache.jackrabbit.core.fs.local.LocalFileSystem"&gt;
92   *              &lt;param name="path" value="${factory.home}/localfs/workspaces/test1/data"/&gt;
93   *            &lt;/FileSystem&gt;
94   *          &lt;/WorkspaceStore&gt;
95   *        &lt;/DynamicWorkspace&gt;
96   *      &lt;/StableWorkspace&gt;
97   *    &lt;/Repository&gt;
98   *  &lt;/Repositories&gt;
99   * </pre>
100  *
101  * @author Stefan Guggisberg

253     /***
254      * @param name
255      * @return
256      * @throws RepositoryException
257      */
258     public synchronized Repository getRepository(String name) throws RepositoryException {
259 	if (reps.containsKey(name)) {
260 	    return (Repository) reps.get(name);
261 	}
262 	Element repConfig = getRepositoryConfig(name);
263 
264 	// read repository store config
265 	// (we know this element must exist since the XML has been checked
266 	// against the DTD)
267 	Element fsConfig = repConfig.getChild(REPOSITORY_STORE_ELEMENT).getChild(FILE_SYSTEM_ELEMENT);
268 	FileSystem fsRepStore = createFileSystem(fsConfig);
269 
270 	// read definitions of stable workspaces
271 	List wspList = repConfig.getChildren(STABLE_WORKSPACE_ELEMENT);
272 	Iterator iter = wspList.iterator();
273 	ArrayList list = new ArrayList();
274 	while (iter.hasNext()) {
275 	    Element wspConfig = (Element) iter.next();
276 	    StableWorkspaceDef wspDef = createStableWspDef(wspConfig);
277 	    list.add(wspDef);
278 	}
279 	StableWorkspaceDef[] wspDefs = (StableWorkspaceDef[]) list.toArray(new StableWorkspaceDef[list.size()]);
280 
281 	Repository rep = new RepositoryImpl(fsRepStore, wspDefs);
282 	reps.put(name, rep);
283 	return rep;
284     }

286     /***
287      * @param name
288      * @return
289      * @throws RepositoryException
290      */
291     private Element getRepositoryConfig(String name) throws RepositoryException {
292 	List repList = config.getRootElement().getChildren(REPOSITORY_ELEMENT);
293 	Element repConfig = null;
294 	Iterator iter = repList.iterator();
295 	while (iter.hasNext()) {
296 	    repConfig = (Element) iter.next();
297 	    String n = repConfig.getAttributeValue(NAME_ATTRIB);
298 	    if (n.equals(name)) {
299 		return repConfig;
300 	    }
301 	}
302 	String msg = "The repository " + name + " was not found in the config file " + configId;
303 	log.error(msg);
304 	throw new RepositoryException(msg);
305     }

307     /***
308      * @param stableWspConfig
309      * @return
310      * @throws RepositoryException
311      */
312     private StableWorkspaceDef createStableWspDef(Element stableWspConfig) throws RepositoryException {
313 	String name = stableWspConfig.getAttributeValue(NAME_ATTRIB);
314 	// we know this element must exist since the XML has been checked
315 	// against the DTD
316 
317 	// FQN of class implementing PersistenceManager interface
318 	String persistenceManagerClassName = stableWspConfig.getChild(PERSISTENCE_MANAGER_ELEMENT).getAttributeValue(CLASS_ATTRIB);
319 
320 	// read the PersistenceManager properties from the
321 	// param elements in the config
322 	HashMap persistenceManagerParams = new HashMap();
323 	List paramList = stableWspConfig.getChild(PERSISTENCE_MANAGER_ELEMENT).getChildren(PARAM_ELEMENT);
324 	for (Iterator i = paramList.iterator(); i.hasNext();) {
325 	    Element param = (Element) i.next();
326 	    String paramName = param.getAttributeValue(NAME_ATTRIB);
327 	    String paramValue = param.getAttributeValue(VALUE_ATTRIB);
328 	    persistenceManagerParams.put(paramName, paramValue);
329 	}
330 
331 	// main workspace store (mandatory)
332 	Element fsConfig = stableWspConfig.getChild(WORKSPACE_STORE_ELEMENT).getChild(FILE_SYSTEM_ELEMENT);
333 	FileSystem wspStore = createFileSystem(fsConfig);
334 
335 	// blob store is optional
336 	FileSystem blobStore = null;
337 	if (stableWspConfig.getChild(BLOB_STORE_ELEMENT) != null) {
338 	    fsConfig = stableWspConfig.getChild(BLOB_STORE_ELEMENT).getChild(FILE_SYSTEM_ELEMENT);
339 	    blobStore = createFileSystem(fsConfig);
340 	}
341 
342 	// read config of dynamic workspaces
343 	List wspList = stableWspConfig.getChildren(DYNAMIC_WORKSPACE_ELEMENT);
344 	Iterator iter = wspList.iterator();
345 	ArrayList list = new ArrayList();
346 	while (iter.hasNext()) {
347 	    Element wspConfig = (Element) iter.next();
348 	    DynamicWorkspaceDef wspDef = createDynWspDef(wspConfig, name);
349 	    list.add(wspDef);
350 	}
351 	DynamicWorkspaceDef[] wspDefs = (DynamicWorkspaceDef[]) list.toArray(new DynamicWorkspaceDef[list.size()]);
352 
353 	// read config of search index (optional)
354 	String searchIndexPath = null;
355 	if (stableWspConfig.getChild(SEARCH_INDEX_ELEMENT) != null) {
356 	    searchIndexPath = stableWspConfig.getChild(SEARCH_INDEX_ELEMENT).getAttributeValue(PATH_ATTRIB);
357 	    searchIndexPath = searchIndexPath.replaceAll("//$//{factory//.home//}", factoryHomeDir.replace('//', '/'));
358 	}
359 
360 	return new StableWorkspaceDef(name, wspStore, blobStore,
361 		persistenceManagerClassName, persistenceManagerParams, wspDefs,
362 		searchIndexPath);
363     }

404     /***
405      * @param fsConfig
406      * @return
407      * @throws RepositoryException
408      */
409     private FileSystem createFileSystem(Element fsConfig) throws RepositoryException {
410 	FileSystem fs;
411 	String className = "";
412 	try {
413 	    // Create the file system object
414 	    className = fsConfig.getAttributeValue(CLASS_ATTRIB);
415 	    Class c = Class.forName(className);
416 	    fs = (FileSystem) c.newInstance();
417 
418 	    // Set the properties of the file system object from the
419 	    // param elements in the config
420 	    BeanMap bm = new BeanMap(fs);
421 	    List paramList = fsConfig.getChildren(PARAM_ELEMENT);
422 	    for (Iterator i = paramList.iterator(); i.hasNext();) {
423 		Element param = (Element) i.next();
424 		String paramName = param.getAttributeValue(NAME_ATTRIB);
425 		String paramValue = param.getAttributeValue(VALUE_ATTRIB);
426 		// @todo FIXME need a cleaner way to specify/configure root of abstract file system
427 		int pos;
428 		int lastPos = 0;
429 		StringBuffer sb = new StringBuffer(paramValue.length());
430 		while ((pos = paramValue.indexOf(FACTORY_HOME_VARIABLE, lastPos)) != -1) {
431 		    sb.append(paramValue.substring(lastPos, pos));
432 		    sb.append(factoryHomeDir);
433 		    lastPos = pos + FACTORY_HOME_VARIABLE.length();
434 		}
435 		if (lastPos < paramValue.length()) {
436 		    sb.append(paramValue.substring(lastPos));
437 		}
438 		bm.put(paramName, sb.toString());
439 	    }
440 	    fs.init();
441 	} catch (Exception e) {
442 	    String msg = "Cannot instantiate implementing class " + className;
443 	    log.error(msg, e);
444 	    throw new RepositoryException(msg, e);
445 	}
446 	return fs;
447     }

449     /***
450      * Shuts down the repositories
451      */
452     public synchronized void shutdown() {
453 	Iterator iter = reps.keySet().iterator();
454 	while (iter.hasNext()) {
455 	    String name = (String) iter.next();
456 	    RepositoryImpl rep = (RepositoryImpl) reps.get(name);
457 	    log.info("Shutting down: " + name);
458 	    rep.shutdown();
459 	}
460 	reps.clear();
461     }
