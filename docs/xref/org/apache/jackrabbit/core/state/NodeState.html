39       * List of parent UUIDs: there's <i>one</i> entry for every parent although
40       * a parent might have more than one child entries refering to <i>this</i>
41       * node state.
42       * <p/>
43       * Furthermore:
44       * <p/>

59      /***
60       * Package private constructor
61       *
62       * @param overlayedState the backing node state being overlayed
63       * @param initialStatus  the initial status of the node state object
64       */
65      NodeState(NodeState overlayedState, int initialStatus) {
66  	super(overlayedState, initialStatus);
67  	nodeTypeName = overlayedState.getNodeTypeName();
68  	mixinTypeNames.addAll(overlayedState.getMixinTypeNames());
69  	defId = overlayedState.getDefinitionId();
70  	uuid = overlayedState.getUUID();
71  	parentUUIDs.addAll(overlayedState.getParentUUIDs());
72  	propertyEntries.addAll(overlayedState.getPropertyEntries());
73  	childNodeEntries.addAll(overlayedState.getChildNodeEntries());
74      }

76      /***
77       * Package private constructor
78       *
79       * @param uuid          the UUID of the this node
80       * @param nodeTypeName  node type of this node
81       * @param parentUUID    the UUID of the parent node
82       * @param initialStatus the initial status of the node state object
83       */
84      NodeState(String uuid, QName nodeTypeName, String parentUUID, int initialStatus) {
85  	super(parentUUID, new NodeId(uuid), initialStatus);
86  	if (parentUUID != null) {
87  	    parentUUIDs.add(parentUUID);
88  	}
89  	this.nodeTypeName = nodeTypeName;
90  	this.uuid = uuid;
91      }

241     /***
242      * Determines if there is a <code>ChildNodeEntry</code> with the
243      * specified <code>name</code> and <code>index</code>.
244      *
245      * @param name  <code>QName</code> object specifying a node name
246      * @param index 1-based index if there are same-name child node entries
247      * @return <code>true</code> if there is a <code>ChildNodeEntry</code> with
248      *         the specified <code>name</code> and <code>index</code>.
249      */
250     public synchronized boolean hasChildNodeEntry(QName name, int index) {
251 	if (index < 1) {
252 	    throw new IllegalArgumentException("index is 1-based");
253 	}
254 	Iterator iter = childNodeEntries.iterator();
255 	int count = 0;
256 	while (iter.hasNext()) {
257 	    ChildNodeEntry entry = (ChildNodeEntry) iter.next();
258 	    if (name.equals(entry.getName())) {
259 		if (++count == index) {
260 		    return true;
261 		}
262 	    }
263 	}
264 	return false;
265     }

630     /***
631      * <code>ChildNodeEntries</code> represents an insertion-ordered
632      * collection of <code>ChildNodeEntry</code>s that also maintains
633      * the index values of same-name siblings on insertion and removal.
634      */
635     private static class ChildNodeEntries implements Serializable {
636 
637 	// insertion-ordered collection of entries
638 	List entries;
639 	// mapping from names to list of same-name sibling entries
640 	Map names;
641 
642 	ChildNodeEntries() {
643 	    entries = new ArrayList();
644 	    names = new HashMap();
645 	}
646 
647 	ChildNodeEntry add(QName nodeName, String uuid) {
648 	    List siblings = (List) names.get(nodeName);
649 	    if (siblings == null) {
650 		siblings = new ArrayList();
651 		names.put(nodeName, siblings);
652 	    }
653 
654 	    int index = siblings.size() + 1;
655 
656 	    ChildNodeEntry entry = new ChildNodeEntry(nodeName, uuid, index);
657 	    siblings.add(entry);
658 	    entries.add(entry);
659 
660 	    return entry;
661 	}
662 
663 	void addAll(List entriesList) {
664 	    Iterator iter = entriesList.iterator();
665 	    while (iter.hasNext()) {
666 		ChildNodeEntry entry = (ChildNodeEntry) iter.next();
667 		// delegate to add(QName, String) to maintain consistency
668 		add(entry.getName(), entry.getUUID());
669 	    }
670 	}
671 
672 	public void removeAll() {
673 	    names.clear();
674 	    entries.clear();
675 	}
676 
677 	public boolean remove(ChildNodeEntry entry) {
678 	    return remove(entry.getName(), entry.getIndex());
679 	}
680 
681 	public boolean remove(QName nodeName, int index) {
682 	    if (index < 1) {
683 		throw new IllegalArgumentException("index is 1-based");
684 	    }
685 	    List siblings = (List) names.get(nodeName);
686 	    if (siblings == null) {
687 		return false;
688 	    }
689 	    if (index > siblings.size()) {
690 		return false;
691 	    }
692 	    // remove from siblings list
693 	    ChildNodeEntry removedEntry = (ChildNodeEntry) siblings.remove(index - 1);
694 	    // remove from entries list
695 	    entries.remove(removedEntry);
696 
697 	    if (siblings.size() == 0) {
698 		// short cut
699 		names.remove(nodeName);
700 		return true;
701 	    }
702 
703 	    // update indices of subsequent same-name siblings
704 	    for (int i = index - 1; i < siblings.size(); i++) {
705 		ChildNodeEntry oldEntry = (ChildNodeEntry) siblings.get(i);
706 		ChildNodeEntry newEntry = new ChildNodeEntry(nodeName, oldEntry.getUUID(), oldEntry.getIndex() - 1);
707 		// overwrite old entry with updated entry in siblings list
708 		siblings.set(i, newEntry);
709 		// overwrite old entry with updated entry in entries list
710 		entries.set(entries.indexOf(oldEntry), newEntry);
711 	    }
712 
713 	    return true;
714 	}
715 
716 	Iterator iterator() {
717 	    return entries.iterator();
718 	}
719 
720 	List entries() {
721 	    return Collections.unmodifiableList(entries);
722 	}
723     }
