16  package org.apache.jackrabbit.core.state.xml;
17  
18  import org.apache.jackrabbit.core.QName;
19  import org.apache.jackrabbit.core.state.ItemStateException;
20  import org.apache.jackrabbit.core.state.PersistableItemState;
21  import org.apache.jackrabbit.core.state.PersistenceManager;
22  import org.apache.jackrabbit.core.state.PersistentNodeState;
23  
24  import java.io.IOException;
25  import java.io.ObjectInputStream;
26  import java.io.ObjectOutputStream;
27  
28  /***
29   * <code>XMLNodeState</code> represents the persistent state of a
30   * <code>Node</code>.
31   *
32   * @author Stefan Guggisberg
33   * @version $Revision: 1.7 $, $Date: 2004/08/02 16:19:50 $
34   */
35  class XMLNodeState extends PersistentNodeState {
36  
37      static final long serialVersionUID = -5635736253672249916L;
38  
39      /***
40       * Package private constructor
41       *
42       * @param uuid         the UUID of the this node
43       * @param nodeTypeName node type of this node
44       * @param parentUUID   the UUID of the parent node
45       * @param persistMgr   the persistence manager
46       */
47      XMLNodeState(String uuid, QName nodeTypeName, String parentUUID, PersistenceManager persistMgr) {
48  	super(uuid, nodeTypeName, parentUUID, persistMgr);
49      }
50  
51      //-------------------------------------------------< PersistableItemState >
52      /***
53       * @see PersistableItemState#reload
54       */
55      public synchronized void reload() throws ItemStateException {
56  	persistMgr.reload(this);
57  	// reset status
58  	status = STATUS_EXISTING;
59      }
60  
61      /***
62       * @see PersistableItemState#store
63       */
64      public synchronized void store() throws ItemStateException {
65  	persistMgr.store(this);
66  	// notify listeners
67  	if (status == STATUS_NEW) {
68  	    notifyStateCreated();
69  	} else {
70  	    notifyStateModified();
71  	}
72  	// reset status
73  	status = STATUS_EXISTING;
74      }
75  
76      /***
77       * @see PersistableItemState#destroy
78       */
79      public synchronized void destroy() throws ItemStateException {
80  	persistMgr.destroy(this);
81  	// notify listeners
82  	notifyStateDestroyed();
83  	// reset status
84  	status = STATUS_UNDEFINED;
85      }
86  
87      //-------------------------------------------------< Serializable support >
88      private void writeObject(ObjectOutputStream out) throws IOException {
89  	// delegate to default implementation
90  	out.defaultWriteObject();
91      }
92  
93      private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException {
94  	// delegate to default implementation
95  	in.defaultReadObject();
96      }
97  }
