102     private String buildNodeFolderPath(String uuid) {
103 	StringBuffer sb = new StringBuffer();
104 	char[] chars = uuid.toCharArray();
105 	int cnt = 0;
106 	for (int i = 0; i < chars.length; i++) {
107 	    if (chars[i] == '-') {
108 		continue;
109 	    }
110 	    //if (cnt > 0 && cnt % 4 == 0) {
111 	    if (cnt == 4 || cnt == 8) {
112 		sb.append('/');
113 	    }
114 	    sb.append(chars[i]);
115 	    cnt++;
116 	}
117 	return sb.toString();
118     }

120     private String buildPropFilePath(String parentUUID, QName propName) {
121 	String fileName;
122 	try {
123 	    MessageDigest md5 = MessageDigest.getInstance("MD5");
124 	    md5.update(propName.getNamespaceURI().getBytes());
125 	    md5.update(propName.getLocalName().getBytes());
126 	    byte[] bytes = md5.digest();
127 	    char[] chars = new char[32];
128 	    for (int i = 0, j = 0; i < 16; i++) {
129 		chars[j++] = HEXDIGITS[(bytes[i] >> 4) & 0x0f];
130 		chars[j++] = HEXDIGITS[bytes[i] & 0x0f];
131 	    }
132 	    //fileName = new String(chars) + ".xml";
133 	    fileName = new String(chars);
134 	} catch (NoSuchAlgorithmException nsae) {
135 	    // should never get here as MD5 should always be available in the JRE
136 	    String msg = "MD5 not available: ";
137 	    log.fatal(msg, nsae);
138 	    throw new InternalError(msg + nsae);
139 	}
140 	return buildNodeFolderPath(parentUUID) + "/" + fileName;
141     }

156     private void readState(Element nodeElement, PersistentNodeState state)
157 	    throws ItemStateException {
158 	// first do some paranoid sanity checks
159 	if (!nodeElement.getName().equals(NODE_ELEMENT)) {
160 	    String msg = "invalid serialization format (unexpected element: " + nodeElement.getName() + ")";
161 	    log.error(msg);
162 	    throw new ItemStateException(msg);
163 	}
164 	// check uuid
165 	if (!state.getUUID().equals(nodeElement.getAttributeValue(UUID_ATTRIBUTE))) {
166 	    String msg = "invalid serialized state: uuid mismatch";
167 	    log.error(msg);
168 	    throw new ItemStateException(msg);
169 	}
170 	// check nodetype
171 	String ntName = nodeElement.getAttributeValue(NODETYPE_ATTRIBUTE);
172 	if (!QName.valueOf(ntName).equals(state.getNodeTypeName())) {
173 	    String msg = "invalid serialized state: nodetype mismatch";
174 	    log.error(msg);
175 	    throw new ItemStateException(msg);
176 	}
177 
178 	// now we're ready to read state
179 
180 	// primary parent
181 	String parentUUID = nodeElement.getAttributeValue(PARENTUUID_ATTRIBUTE);
182 	if (parentUUID.length() > 0) {
183 	    state.setParentUUID(parentUUID);
184 	}
185 
186 	// definition id
187 	String definitionId = nodeElement.getAttributeValue(DEFINITIONID_ATTRIBUTE);
188 	state.setDefinitionId(NodeDefId.valueOf(definitionId));
189 
190 	// parent uuid's
191 	Iterator iter = nodeElement.getChild(PARENTS_ELEMENT).getChildren(PARENT_ELEMENT).iterator();
192 	ArrayList parentUUIDs = new ArrayList();
193 	while (iter.hasNext()) {
194 	    Element parentElement = (Element) iter.next();
195 	    parentUUIDs.add(parentElement.getAttributeValue(UUID_ATTRIBUTE));
196 	}
197 	if (parentUUIDs.size() > 0) {
198 	    state.setParentUUIDs(parentUUIDs);
199 	}
200 
201 	// mixin types
202 	Element mixinsElement = nodeElement.getChild(MIXINTYPES_ELEMENT);
203 	if (mixinsElement != null) {
204 	    iter = mixinsElement.getChildren(MIXINTYPE_ELEMENT).iterator();
205 	    Set mixins = new HashSet();
206 	    while (iter.hasNext()) {
207 		Element mixinElement = (Element) iter.next();
208 		mixins.add(QName.valueOf(mixinElement.getAttributeValue(NAME_ATTRIBUTE)));
209 	    }
210 	    if (mixins.size() > 0) {
211 		state.setMixinTypeNames(mixins);
212 	    }
213 	}
214 
215 	// property entries
216 	iter = nodeElement.getChild(PROPERTIES_ELEMENT).getChildren(PROPERTY_ELEMENT).iterator();
217 	while (iter.hasNext()) {
218 	    Element propElement = (Element) iter.next();
219 	    String propName = propElement.getAttributeValue(NAME_ATTRIBUTE);
220 	    // @todo deserialize type and values
221 	    state.addPropertyEntry(QName.valueOf(propName));
222 	}
223 
224 	// child node entries
225 	iter = nodeElement.getChild(NODES_ELEMENT).getChildren(NODE_ELEMENT).iterator();
226 	while (iter.hasNext()) {
227 	    Element childNodeElement = (Element) iter.next();
228 	    String childName = childNodeElement.getAttributeValue(NAME_ATTRIBUTE);
229 	    String childUUID = childNodeElement.getAttributeValue(UUID_ATTRIBUTE);
230 	    state.addChildNodeEntry(QName.valueOf(childName), childUUID);
231 	}
232     }

234     private void readState(Element propElement, PersistentPropertyState state)
235 	    throws ItemStateException {
236 	// first do some paranoid sanity checks
237 	if (!propElement.getName().equals(PROPERTY_ELEMENT)) {
238 	    String msg = "invalid serialization format (unexpected element: " + propElement.getName() + ")";
239 	    log.error(msg);
240 	    throw new ItemStateException(msg);
241 	}
242 	// check name
243 	if (!state.getName().equals(QName.valueOf(propElement.getAttributeValue(NAME_ATTRIBUTE)))) {
244 	    String msg = "invalid serialized state: name mismatch";
245 	    log.error(msg);
246 	    throw new ItemStateException(msg);
247 	}
248 	// check parentUUID
249 	String parentUUID = propElement.getAttributeValue(PARENTUUID_ATTRIBUTE);
250 	if (!parentUUID.equals(state.getParentUUID())) {
251 	    String msg = "invalid serialized state: parentUUID mismatch";
252 	    log.error(msg);
253 	    throw new ItemStateException(msg);
254 	}
255 
256 	// now we're ready to read state
257 
258 	// type
259 	String typeName = propElement.getAttributeValue(TYPE_ATTRIBUTE);
260 	int type;
261 	try {
262 	    type = PropertyType.valueFromName(typeName);
263 	} catch (IllegalArgumentException iae) {
264 	    // should never be getting here
265 	    throw new ItemStateException("unexpected property-type: " + typeName, iae);
266 	}
267 	state.setType(type);
268 
269 	// definition id
270 	String definitionId = propElement.getAttributeValue(DEFINITIONID_ATTRIBUTE);
271 	state.setDefinitionId(PropDefId.valueOf(definitionId));
272 
273 	// values
274 	Iterator iter = propElement.getChild(VALUES_ELEMENT).getChildren(VALUE_ELEMENT).iterator();
275 	ArrayList values = new ArrayList();
276 	while (iter.hasNext()) {
277 	    Element valueElement = (Element) iter.next();
278 	    Filter filter = new ContentFilter(ContentFilter.TEXT | ContentFilter.CDATA);
279 	    List content = valueElement.getContent(filter);
280 
281 	    InternalValue val;
282 	    if (!content.isEmpty()) {
283 		// read serialized value
284 		String text = valueElement.getTextTrim();
285 		if (type == PropertyType.BINARY) {
286 		    // special handling required for binary value:
287 		    // the value stores the path to the actual binary file in the blob store
288 		    try {
289 			val = InternalValue.create(new FileSystemResource(blobStore, text));
290 		    } catch (IOException ioe) {
291 			String msg = "error while reading serialized binary valuey";
292 			log.error(msg, ioe);
293 			throw new ItemStateException(msg, ioe);
294 		    }
295 		} else {
296 		    val = InternalValue.valueOf(text, type);
297 		}
298 	    } else {
299 		// null value
300 		val = null;
301 	    }
302 	    values.add(val);
303 	}
304 	state.setValues((InternalValue[]) values.toArray(new InternalValue[values.size()]));
305     }

307     private void readState(Properties props, PersistentPropertyState state)
308 	    throws ItemStateException {
309 	// type
310 	String typeName = props.getProperty(TYPE_ATTRIBUTE);
311 	int type;
312 	try {
313 	    type = PropertyType.valueFromName(typeName);
314 	} catch (IllegalArgumentException iae) {
315 	    // should never be getting here
316 	    throw new ItemStateException("unexpected property-type: " + typeName, iae);
317 	}
318 	state.setType(type);
319 
320 	// definition id
321 	if (props.containsKey(DEFINITIONID_ATTRIBUTE)) {
322 	    state.setDefinitionId(PropDefId.valueOf(props.getProperty(DEFINITIONID_ATTRIBUTE)));
323 	}
324 
325 	// # of values
326 	int cnt = Integer.parseInt(props.getProperty(COUNT_ATTRIBUTE));
327 
328 	// values
329 	InternalValue[] values = new InternalValue[cnt];
330 	Enumeration names = props.propertyNames();
331 	while (names.hasMoreElements()) {
332 	    String name = (String) names.nextElement();
333 	    if (TYPE_ATTRIBUTE.equals(name) || COUNT_ATTRIBUTE.equals(name)
334 		    || DEFINITIONID_ATTRIBUTE.equals(name)) {
335 		continue;
336 	    }
337 	    int i = Integer.parseInt(name);
338 	    InternalValue val;
339 	    String text = props.getProperty(name);
340 	    if (text != null) {
341 		if (type == PropertyType.BINARY) {
342 		    // special handling required for binary value:
343 		    // the value stores the path to the actual binary file in the blob store
344 		    try {
345 			val = InternalValue.create(new FileSystemResource(blobStore, text));
346 		    } catch (IOException ioe) {
347 			String msg = "error while reading serialized binary valuey";
348 			log.error(msg, ioe);
349 			throw new ItemStateException(msg, ioe);
350 		    }
351 		} else {
352 		    val = InternalValue.valueOf(text, type);
353 		}
354 	    } else {
355 		// null value
356 		val = null;
357 	    }
358 	    values[i] = val;
359 	}
360 	state.setValues(values);
361     }

378      * @see PersistenceManager#loadNodeState
379      */
380     public synchronized PersistentNodeState loadNodeState(String uuid)
381 	    throws NoSuchItemStateException, ItemStateException {
382 	if (!initialized) {
383 	    throw new IllegalStateException("not initialized");
384 	}
385 
386 	Exception e = null;
387 	String nodeFilePath = buildNodeFilePath(uuid);
388 	try {
389 	    if (!itemStateStore.isFile(nodeFilePath)) {
390 		throw new NoSuchItemStateException(uuid);
391 	    }
392 	    InputStream in = itemStateStore.getInputStream(nodeFilePath);
393 
394 	    try {
395 		SAXBuilder builder = new SAXBuilder();
396 		Element rootElement = builder.build(in).getRootElement();
397 		String ntName = rootElement.getAttributeValue(NODETYPE_ATTRIBUTE);
398 
399 		PersistentNodeState state = createNodeStateInstance(uuid, QName.valueOf(ntName));
400 		readState(rootElement, state);
401 
402 		return state;
403 	    } finally {
404 		in.close();
405 	    }
406 	} catch (JDOMException jde) {
407 	    e = jde;
408 	    // fall through
409 	} catch (IOException ioe) {
410 	    e = ioe;
411 	    // fall through
412 	} catch (FileSystemException fse) {
413 	    e = fse;
414 	    // fall through
415 	}
416 	String msg = "failed to read node state: " + uuid;
417 	log.error(msg, e);
418 	throw new ItemStateException(msg, e);
419     }

571     /***
572      * @see PersistenceManager#store
573      */
574     public synchronized void store(PersistentNodeState state) throws ItemStateException {
575 	if (!initialized) {
576 	    throw new IllegalStateException("not initialized");
577 	}
578 
579 	String uuid = state.getUUID();
580 	String nodeFilePath = buildNodeFilePath(uuid);
581 	FileSystemResource nodeFile = new FileSystemResource(itemStateStore, nodeFilePath);
582 	try {
583 	    nodeFile.makeParentDirs();
584 	    OutputStream os = nodeFile.getOutputStream();
585 	    Writer writer = null;
586 	    try {
587 		String encoding = DEFAULT_ENCODING;
588 		try {
589 		    writer = new BufferedWriter(new OutputStreamWriter(os, encoding));
590 		} catch (UnsupportedEncodingException e) {
591 		    // should never get here!
592 		    OutputStreamWriter osw = new OutputStreamWriter(os);
593 		    encoding = osw.getEncoding();
594 		    writer = new BufferedWriter(osw);
595 		}
596 
597 		writer.write("<?xml version=\"1.0\" encoding=\"" + encoding + "\"?>\n");
598 		writer.write("<" + NODE_ELEMENT + " "
599 			+ UUID_ATTRIBUTE + "=\"" + state.getUUID() + "\" "
600 			+ PARENTUUID_ATTRIBUTE + "=\"" + (state.getParentUUID() == null ? "" : state.getParentUUID()) + "\" "
601 			+ DEFINITIONID_ATTRIBUTE + "=\"" + state.getDefinitionId().toString() + "\" "
602 			+ NODETYPE_ATTRIBUTE + "=\"" + state.getNodeTypeName() + "\">\n");
603 		// parents
604 		writer.write("\t<" + PARENTS_ELEMENT + ">\n");
605 		Iterator iter = state.getParentUUIDs().iterator();
606 		while (iter.hasNext()) {
607 		    writer.write("\t\t<" + PARENT_ELEMENT + " "
608 			    + UUID_ATTRIBUTE + "=\"" + iter.next() + "\"/>\n");
609 		}
610 		writer.write("\t</" + PARENTS_ELEMENT + ">\n");
611 
612 		// mixin types
613 		writer.write("\t<" + MIXINTYPES_ELEMENT + ">\n");
614 		iter = state.getMixinTypeNames().iterator();
615 		while (iter.hasNext()) {
616 		    writer.write("\t\t<" + MIXINTYPE_ELEMENT + " "
617 			    + NAME_ATTRIBUTE + "=\"" + iter.next() + "\"/>\n");
618 		}
619 		writer.write("\t</" + MIXINTYPES_ELEMENT + ">\n");
620 
621 		// properties
622 		writer.write("\t<" + PROPERTIES_ELEMENT + ">\n");
623 		iter = state.getPropertyEntries().iterator();
624 		while (iter.hasNext()) {
625 		    NodeState.PropertyEntry entry = (NodeState.PropertyEntry) iter.next();
626 		    writer.write("\t\t<" + PROPERTY_ELEMENT + " "
627 			    + NAME_ATTRIBUTE + "=\"" + entry.getName() + "\">\n");
628 		    // @todo serialize type, definition id and values
629 		    writer.write("\t\t</" + PROPERTY_ELEMENT + ">\n");
630 		}
631 		writer.write("\t</" + PROPERTIES_ELEMENT + ">\n");
632 
633 		// child nodes
634 		writer.write("\t<" + NODES_ELEMENT + ">\n");
635 		iter = state.getChildNodeEntries().iterator();
636 		while (iter.hasNext()) {
637 		    NodeState.ChildNodeEntry entry = (NodeState.ChildNodeEntry) iter.next();
