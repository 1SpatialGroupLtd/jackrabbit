45      /***
46       * Creates a new <code>PersistentItemStateManager</code> instance.
47       *
48       * @param persistMgr
49       * @param rootNodeUUID
50       * @param ntReg
51       */
52      public PersistentItemStateManager(PersistenceManager persistMgr,
53  				      String rootNodeUUID,
54  				      NodeTypeRegistry ntReg)
55  	    throws ItemStateException {
56  	this.persistMgr = persistMgr;
57  
58  	try {
59  	    root = getNodeState(new NodeId(rootNodeUUID));
60  	} catch (NoSuchItemStateException e) {
61  	    // create root node
62  	    root = createPersistentRootNodeState(rootNodeUUID, ntReg);
63  	}
64      }

66      private PersistentNodeState createPersistentRootNodeState(String rootNodeUUID,
67  							      NodeTypeRegistry ntReg)
68  	    throws ItemStateException {
69  	PersistentNodeState rootState = createNodeState(rootNodeUUID, NodeTypeRegistry.NT_UNSTRUCTURED, null);
70  
71  	// @todo FIXME need to manually setup root node by creating mandatory jcr:primaryType property
72  	NodeDefId nodeDefId = null;
73  	PropDefId propDefId = null;
74  
75  	try {
76  	    // FIXME relies on definition of nt:unstructured and nt:base:
77  	    // first (and only) child node definition in nt:unstructured is applied to root node
78  	    nodeDefId = new NodeDefId(ntReg.getNodeTypeDef(NodeTypeRegistry.NT_UNSTRUCTURED).getChildNodeDefs()[0]);
79  	    // first property definition in nt:base is jcr:primaryType
80  	    propDefId = new PropDefId(ntReg.getNodeTypeDef(NodeTypeRegistry.NT_BASE).getPropertyDefs()[0]);
81  	} catch (NoSuchNodeTypeException nsnte) {
82  	    String msg = "failed to create root node";
83  	    log.error(msg, nsnte);
84  	    throw new ItemStateException(msg, nsnte);
85  	}
86  	rootState.setDefinitionId(nodeDefId);
87  
88  	QName propName = new QName(NamespaceRegistryImpl.NS_JCR_URI, "primaryType");
89  	rootState.addPropertyEntry(propName);
90  
91  	PersistentPropertyState prop = createPropertyState(rootNodeUUID, propName);
92  	prop.setValues(new InternalValue[]{InternalValue.create(NodeTypeRegistry.NT_UNSTRUCTURED)});
93  	prop.setType(PropertyType.NAME);
94  	prop.setDefinitionId(propDefId);
95  
96  	rootState.store();
97  	prop.store();
98  
99  	return rootState;
100     }

102     /***
103      * Dumps the state of this <code>TransientItemStateManager</code> instance
104      * (used for diagnostic purposes).
105      *
106      * @param ps
107      */
108     public void dump(PrintStream ps) {
109 	ps.println("PersistentItemStateManager (" + this + ")");
110 	ps.println();
111 	ps.println("entries in cache:");
112 	ps.println();
113 	Iterator iter = keys();
114 	while (iter.hasNext()) {
115 	    ItemId id = (ItemId) iter.next();
116 	    ItemState state = retrieve(id);
117 	    dumpItemState(id, state, ps);
118 	}
119     }

121     private void dumpItemState(ItemId id, ItemState state, PrintStream ps) {
122 	ps.print(state.isNode() ? "Node: " : "Prop: ");
123 	switch (state.getStatus()) {
124 	    case ItemState.STATUS_EXISTING:
125 		ps.print("[existing]           ");
126 		break;
127 	    case ItemState.STATUS_EXISTING_MODIFIED:
128 		ps.print("[existing, modified] ");
129 		break;
130 	    case ItemState.STATUS_EXISTING_REMOVED:
131 		ps.print("[existing, removed]  ");
132 		break;
133 	    case ItemState.STATUS_NEW:
134 		ps.print("[new]                ");
135 		break;
136 	    case ItemState.STATUS_STALE_DESTROYED:
137 		ps.print("[stale, destroyed]   ");
138 		break;
139 	    case ItemState.STATUS_STALE_MODIFIED:
140 		ps.print("[stale, modified]    ");
141 		break;
142 	    case ItemState.STATUS_UNDEFINED:
143 		ps.print("[undefined]          ");
144 		break;
145 	}
146 	ps.println(id + " (" + state + ")");
147     }

192     /***
193      * @param id
194      * @return
195      * @throws NoSuchItemStateException
196      * @throws ItemStateException
197      */
198     synchronized PersistentNodeState getNodeState(NodeId id)
199 	    throws NoSuchItemStateException, ItemStateException {
200 	// check cache
201 	if (isCached(id)) {
202 	    return (PersistentNodeState) retrieve(id);
203 	}
204 	// load from persisted state
205 	PersistentNodeState state = persistMgr.loadNodeState(id.getUUID());
206 	state.setStatus(ItemState.STATUS_EXISTING);
207 	// put it in cache
208 	cache(state);
209 	// register as listener
210 	state.addListener(this);
211 	return state;
212     }

237     /***
238      * Creates a <code>PersistentNodeState</code> instance representing new,
239      * i.e. not yet existing state. Call <code>{@link PersistentNodeState#store()}</code>
240      * on the returned object to make it persistent.
241      *
242      * @param uuid
243      * @param nodeTypeName
244      * @param parentUUID
245      * @return
246      * @throws ItemStateException
247      */
248     public synchronized PersistentNodeState createNodeState(String uuid, QName nodeTypeName, String parentUUID)
249 	    throws ItemStateException {
250 	NodeId id = new NodeId(uuid);
251 	// check cache
252 	if (isCached(id)) {
253 	    String msg = "there's already a node state instance with id " + id;
254 	    log.error(msg);
255 	    throw new ItemStateException(msg);
256 	}
257 
258 	PersistentNodeState state = persistMgr.createNodeStateInstance(uuid, nodeTypeName);
259 	state.setParentUUID(parentUUID);
260 	// put it in cache
261 	cache(state);
262 	// register as listener
263 	state.addListener(this);
264 	return state;
265     }

267     /***
268      * Creates a <code>PersistentPropertyState</code> instance representing new,
269      * i.e. not yet existing state. Call <code>{@link PersistentPropertyState#store()}</code>
270      * on the returned object to make it persistent.
271      *
272      * @param parentUUID
273      * @param propName
274      * @return
275      * @throws ItemStateException
276      */
277     public synchronized PersistentPropertyState createPropertyState(String parentUUID, QName propName)
278 	    throws ItemStateException {
279 	PropertyId id = new PropertyId(parentUUID, propName);
280 	// check cache
281 	if (isCached(id)) {
282 	    String msg = "there's already a property state instance with id " + id;
283 	    log.error(msg);
284 	    throw new ItemStateException(msg);
285 	}
286 
287 	PersistentPropertyState state = persistMgr.createPropertyStateInstance(propName, parentUUID);
288 	// put it in cache
289 	cache(state);
290 	// register as listener
291 	state.addListener(this);
292 	return state;
293     }
