16  package org.apache.jackrabbit.core.state;
17  
18  import org.apache.commons.collections.ReferenceMap;
19  import org.apache.jackrabbit.core.NodeId;
20  import org.apache.log4j.Logger;
21  
22  import javax.jcr.RepositoryException;
23  import java.util.Map;
24  
25  /***
26   * <code>ReferenceManager</code> ...
27   *
28   * @author Stefan Guggisberg
29   * @version $Revision: 1.2 $, $Date: 2004/09/06 14:28:05 $
30   */
31  public class ReferenceManager {
32  
33      private static Logger log = Logger.getLogger(ReferenceManager.class);
34  
35      private final PersistenceManager persistMgr;
36  
37      /***
38       * A cache for <code>NodeReferences</code> objects created by this
39       *  <code>ReferenceManager</code>
40       */
41      private Map refsCache;
42  
43      /***
44       * Package private constructor
45       */
46      public ReferenceManager(PersistenceManager persistMgr) {
47  	this.persistMgr = persistMgr;
48  	// setup cache with soft references to <code>NodeReferences</code> objects
49  	refsCache = new ReferenceMap(ReferenceMap.HARD, ReferenceMap.SOFT);
50      }
51  
52      /***
53       *
54       * @param targetId
55       * @return
56       * @throws RepositoryException
57       */
58      public synchronized NodeReferences get(NodeId targetId) throws RepositoryException {
59  	if (refsCache.containsKey(targetId)) {
60  	    return (NodeReferences) refsCache.get(targetId);
61  	}
62  	NodeReferences refs;
63  	try {
64  	    // load persisted references
65  	    refs = persistMgr.loadNodeReferences(targetId.getUUID());
66  	} catch (NoSuchItemStateException nsise) {
67  	    // does not exist, create new
68  	    refs = persistMgr.createNodeReferencesInstance(targetId.getUUID());
69  	} catch (ItemStateException ise) {
70  	    String msg = "error while loading references";
71  	    log.error(msg, ise);
72  	    throw new RepositoryException(msg, ise);
73  	}
74  	// put it in cache
75  	refsCache.put(targetId, refs);
76  	return refs;
77      }
78  
79      /***
80       *
81       * @param refs
82       * @throws RepositoryException
83       */
84      public synchronized void save(NodeReferences refs) throws RepositoryException {
85  	if (!refs.hasReferences()) {
86  	    remove(refs);
87  	    return;
88  	}
89  	if (!refsCache.containsKey(refs.getTargetId())) {
90  	    // not yet in cache, put it in cache
91  	    refsCache.put(refs.getTargetId(), refs);
92  	}
93  	try {
94  	    // store references
95  	    persistMgr.store(refs);
96  	} catch (ItemStateException ise) {
97  	    String msg = "error while storing references";
98  	    log.error(msg, ise);
99  	    throw new RepositoryException(msg, ise);
100 	}
101     }
102 
103     /***
104      *
105      * @param refs
106      * @throws RepositoryException
107      */
108     public synchronized void remove(NodeReferences refs) throws RepositoryException {
109 	try {
110 	    // destroy persisted references
111 	    persistMgr.destroy(refs);
112 	} catch (ItemStateException ise) {
113 	    String msg = "error while destroying references";
114 	    log.error(msg, ise);
115 	    throw new RepositoryException(msg, ise);
116 	}
117 
118 	// remove from cache
119 	refsCache.remove(refs.getTargetId());
120     }
121 }
