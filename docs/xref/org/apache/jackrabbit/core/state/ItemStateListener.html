16  package org.apache.jackrabbit.core.state;
17  
18  /***
19   * The <code>ItemStateListener</code> interface allows an implementing object
20   * to be informed about changes on an <code>ItemState</code>.
21   *
22   * @author Stefan Guggisberg
23   * @version $Revision: 1.5 $, $Date: 2004/08/02 16:19:48 $
24   * @see ItemState#addListener
25   */
26  public interface ItemStateListener {
27  
28      /***
29       * Called when an <code>ItemState</code> has successfully
30       * been created (i.e. its underlying persistent state was created).
31       *
32       * @param created the <code>ItemState</code> that has been 'created'
33       */
34      public void stateCreated(ItemState created);
35  
36      /***
37       * Called when an <code>ItemState</code> has successfully
38       * been modified (i.e. its underlying persistent state has changed).
39       *
40       * @param modified the <code>ItemState</code> that has been 'modified'
41       */
42      public void stateModified(ItemState modified);
43  
44      /***
45       * Called when an <code>ItemState</code> has successfully been
46       * removed (i.e. its underlying persistent state has been destroyed).
47       *
48       * @param destroyed the <code>ItemState</code> that has been 'destroyed'
49       */
50      public void stateDestroyed(ItemState destroyed);
51  
52      /***
53       * Called when an <code>ItemState</code> has been discarded (i.e. it has
54       * been rendered 'invalid').
55       *
56       * @param discarded the <code>ItemState</code> that has been discarded
57       * @see ItemState#discard
58       */
59      public void stateDiscarded(ItemState discarded);
60  }
