149     //-------------------------------------------------< Serializable support >
150     private void writeObject(ObjectOutputStream out) throws IOException {
151 	// important: fields must be written in same order as they are
152 	// read in readObject(ObjectInputStream)
153 	out.writeObject(name);
154 	out.writeInt(type);
155 	if (values == null) {
156 	    out.writeObject(null);
157 	} else {
158 	    String[] strings = new String[values.length];
159 	    for (int i = 0; i < values.length; i++) {
160 		InternalValue val = values[i];
161 		try {
162 		    if (type == PropertyType.BINARY) {
163 			// special handling required for binary value
164 			BLOBFileValue blob = (BLOBFileValue) val.internalValue();
165 			InputStream in = blob.getStream();
166 			// use 32k initial buffer size as binary data is
167 			// probably not just a couple of bytes
168 			StringWriter writer = new StringWriter(32768);
169 			try {
170 			    Base64.encode(in, writer);
171 			} finally {
172 			    in.close();
173 			    writer.close();
174 			}
175 			strings[i] = writer.toString();
176 		    } else {
177 			strings[i] = val.toString();
178 		    }
179 		} catch (IllegalStateException ise) {
180 		    throw new IOException(ise.getMessage());
181 		} catch (RepositoryException re) {
182 		    throw new IOException(re.getMessage());
183 		}
184 	    }
185 	    out.writeObject(strings);
186 	}
187     }
188 
189     private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException {
190 	// important: fields must be read in same order as they are
191 	// written in writeObject(ObjectOutputStream)
192 	name = (QName) in.readObject();
193 	type = in.readInt();
194 	Object obj = in.readObject();
195 	if (obj == null) {
196 	    values = null;
197 	} else {
198 	    String[] strings = (String[]) obj;
199 	    values = new InternalValue[strings.length];
200 	    for (int i = 0; i < strings.length; i++) {
201 		String str = strings[i];
202 		if (type == PropertyType.BINARY) {
203 		    // special handling required for binary value
204 		    ByteArrayOutputStream bos = new ByteArrayOutputStream(str.length());
205 		    Base64.decode(str, bos);
206 		    bos.close();
207 		    values[i] = InternalValue.create(new ByteArrayInputStream(bos.toByteArray()));
208 		} else {
209 		    values[i] = InternalValue.valueOf(str, type);
210 		}
211 	    }
212 	}
213     }
