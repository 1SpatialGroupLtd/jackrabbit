85      /***
86       * @param in
87       * @throws IOException
88       * @throws InvalidNodeTypeDefException
89       * @throws RepositoryException
90       */
91      void load(InputStream in)
92  	    throws IOException, InvalidNodeTypeDefException, RepositoryException {
93  	SAXBuilder builder = new SAXBuilder();
94  	Element root = null;
95  	try {
96  	    Document doc = builder.build(in);
97  	    root = doc.getRootElement();
98  	} catch (JDOMException jde) {
99  	    String msg = "internal error: failed to parse persistent node type definitions";
100 	    log.error(msg, jde);
101 	    throw new RepositoryException(msg, jde);
102 	}
103 	// read namespace declarations of root element
104 	Iterator nsIter = root.getAdditionalNamespaces().iterator();
105 	final HashMap prefixToURI = new HashMap();
106 	final HashMap uriToPrefix = new HashMap();
107 	while (nsIter.hasNext()) {
108 	    Namespace ns = (Namespace) nsIter.next();
109 	    prefixToURI.put(ns.getPrefix(), ns.getURI());
110 	    uriToPrefix.put(ns.getURI(), ns.getPrefix());
111 	}
112 	// add default namespace (empty uri)
113 	prefixToURI.put(NamespaceRegistryImpl.NS_EMPTY_PREFIX, NamespaceRegistryImpl.NS_DEFAULT_URI);
114 	uriToPrefix.put(NamespaceRegistryImpl.NS_DEFAULT_URI, NamespaceRegistryImpl.NS_EMPTY_PREFIX);
115 
116 	NamespaceResolver nsResolver = new NamespaceResolver() {
117 	    public String getURI(String prefix) throws NamespaceException {
118 		if (!prefixToURI.containsKey(prefix)) {
119 		    throw new NamespaceException(prefix + ": unknown prefix");
120 		}
121 		return (String) prefixToURI.get(prefix);
122 	    }
123 
124 	    public String getPrefix(String uri) throws NamespaceException {
125 		if (!uriToPrefix.containsKey(uri)) {
126 		    throw new NamespaceException(uri + ": unknown URI");
127 		}
128 		return (String) uriToPrefix.get(uri);
129 	    }
130 	};
131 
132 	// read definitions
133 	Iterator iter = root.getChildren(NODETYPE_ELEMENT).iterator();
134 	while (iter.hasNext()) {
135 	    NodeTypeDef ntDef = readDef((Element) iter.next(), nsResolver);
136 	    add(ntDef);
137 	}
138     }

140     /***
141      * @param out
142      * @param nsReg
143      * @throws IOException
144      * @throws RepositoryException
145      */
146     void store(OutputStream out, NamespaceRegistryImpl nsReg)
147 	    throws IOException, RepositoryException {
148 	Element root = new Element(ROOT_ELEMENT);
149 
150 	// namespace declarations
151 	String[] prefixes = nsReg.getPrefixes();
152 	for (int i = 0; i < prefixes.length; i++) {
153 	    String prefix = prefixes[i];
154 	    if ("".equals(prefix)) {
155 		continue;
156 	    }
157 	    String uri = nsReg.getURI(prefix);
158 	    root.addNamespaceDeclaration(Namespace.getNamespace(prefix, uri));
159 	}
160 
161 	// node type definitions
162 	Iterator iter = all().iterator();
163 	while (iter.hasNext()) {
164 	    NodeTypeDef ntd = (NodeTypeDef) iter.next();
165 	    Element ntElem = new Element(NODETYPE_ELEMENT);
166 	    writeDef(ntd, ntElem, nsReg);
167 	    root.addContent(ntElem);
168 	}
169 
170 	XMLOutputter serializer = new XMLOutputter(Format.getPrettyFormat());
171 	serializer.output(new Document(root), out);
172     }

219     //------------------------------------------------------< private methods >
220     private NodeTypeDef readDef(Element ntElem, NamespaceResolver nsResolver)
221 	    throws InvalidNodeTypeDefException {
222 	String sntName = ntElem.getAttributeValue(NAME_ATTRIBUTE);
223 	NodeTypeDef ntDef = new NodeTypeDef();
224 
225 	// name
226 	QName qntName;
227 	try {
228 	    qntName = QName.fromJCRName(sntName, nsResolver);
229 	} catch (BaseException e) {
230 	    String msg = "invalid serialized node type definition [" + sntName + "]: invalid node type name: " + sntName;
231 	    log.error(msg, e);
232 	    throw new InvalidNodeTypeDefException(msg, e);
233 	}
234 	ntDef.setName(qntName);
235 
236 	// supertypes
237 	ArrayList list = new ArrayList();
238 	Element typesElem = ntElem.getChild(SUPERTYPES_ELEMENT);
239 	if (typesElem != null) {
240 	    Iterator iter = typesElem.getChildren(SUPERTYPE_ELEMENT).iterator();
241 	    while (iter.hasNext()) {
242 		Element typeElem = (Element) iter.next();
243 		Filter filter = new ContentFilter(ContentFilter.TEXT | ContentFilter.CDATA);
244 		List content = typeElem.getContent(filter);
245 		if (!content.isEmpty()) {
246 		    String name = typeElem.getTextTrim();
247 		    try {
248 			list.add(QName.fromJCRName(name, nsResolver));
249 		    } catch (BaseException e) {
250 			String msg = "invalid serialized node type definition [" + sntName + "]: invalid supertype: " + name;
251 			log.error(msg, e);
252 			throw new InvalidNodeTypeDefException(msg, e);
253 		    }
254 		}
255 	    }
256 	    if (!list.isEmpty()) {
257 		ntDef.setSupertypes((QName[]) list.toArray(new QName[list.size()]));
258 	    }
259 	}
260 
261 	// mixin
262 	String mixin = ntElem.getAttributeValue(MIXIN_ATTRIBUTE);
263 	if (mixin != null && mixin.length() > 0) {
264 	    ntDef.setMixin(Boolean.valueOf(mixin).booleanValue());
265 	}
266 
267 	// orderableChildNodes
268 	String orderableChildNodes = ntElem.getAttributeValue(ORDERABLECHILDNODES_ATTRIBUTE);
269 	if (orderableChildNodes != null && orderableChildNodes.length() > 0) {
270 	    ntDef.setOrderableChildNodes(Boolean.valueOf(orderableChildNodes).booleanValue());
271 	}
272 
273 	// property definitions
274 	list.clear();
275 	Iterator iter = ntElem.getChildren(PROPERTYDEF_ELEMENT).iterator();
276 	while (iter.hasNext()) {
277 	    Element elem = (Element) iter.next();
278 	    PropDef pd = new PropDef();
279 	    // declaring node type
280 	    pd.setDeclaringNodeType(qntName);
281 	    // name
282 	    String propName = elem.getAttributeValue(NAME_ATTRIBUTE);
283 	    if (propName != null && !propName.equals(WILDCARD)) {
284 		try {
285 		    pd.setName(QName.fromJCRName(propName, nsResolver));
286 		} catch (BaseException e) {
287 		    String msg = "invalid serialized node type definition [" + sntName + "]: invalid property name: " + propName;
288 		    log.error(msg, e);
289 		    throw new InvalidNodeTypeDefException(msg, e);
290 		}
291 	    } else {
292 		pd.setName(ChildItemDef.ANY_NAME);
293 	    }
294 	    // type
295 	    String typeName = elem.getAttributeValue(TYPE_ATTRIBUTE);
296 	    int type = PropertyType.UNDEFINED;
297 	    if (typeName != null && typeName.length() > 0) {
298 		try {
299 		    type = PropertyType.valueFromName(typeName);
300 		    pd.setRequiredType(type);
301 		} catch (IllegalArgumentException e) {
302 		    String msg = "invalid serialized node type definition [" + sntName + "]: invalid type: " + typeName;
303 		    log.error(msg, e);
304 		    throw new InvalidNodeTypeDefException(msg, e);
305 		}
306 	    }
307 	    // valueConstraints
308 	    Element constraintsElem = elem.getChild(VALUECONSTRAINTS_ELEMENT);
309 	    if (constraintsElem != null) {
310 		ArrayList list1 = new ArrayList();
311 		Iterator iter1 = constraintsElem.getChildren(VALUECONSTRAINT_ELEMENT).iterator();
312 		while (iter1.hasNext()) {
313 		    Element constraintElem = (Element) iter1.next();
314 		    Filter filter = new ContentFilter(ContentFilter.TEXT | ContentFilter.CDATA);
315 		    List content = constraintElem.getContent(filter);
316 		    if (!content.isEmpty()) {
317 			String constraint = constraintElem.getTextTrim();
318 			try {
319 			    list1.add(ValueConstraint.create(type, constraint));
320 			} catch (InvalidConstraintException e) {
321 			    String msg = "invalid serialized node type definition [" + sntName + "]: invalid constraint: " + constraint;
322 			    log.error(msg, e);
323 			    throw new InvalidNodeTypeDefException(msg, e);
324 			}
325 		    }
326 		}
327 		if (!list1.isEmpty()) {
328 		    pd.setValueConstraints((ValueConstraint[]) list1.toArray(new ValueConstraint[list1.size()]));
329 		}
330 	    }
331 	    // defaultValues
332 	    Element defValuesElem = elem.getChild(DEFAULTVALUES_ELEMENT);
333 	    if (defValuesElem != null) {
334 		int defValType = (type == PropertyType.UNDEFINED) ? PropertyType.STRING : type;
335 		ArrayList list1 = new ArrayList();
336 		Iterator iter1 = defValuesElem.getChildren(DEFAULTVALUE_ELEMENT).iterator();
337 		while (iter1.hasNext()) {
338 		    Element valueElem = (Element) iter1.next();
339 		    Filter filter = new ContentFilter(ContentFilter.TEXT | ContentFilter.CDATA);
340 		    List content = valueElem.getContent(filter);
341 		    if (!content.isEmpty()) {
342 			String defValue = valueElem.getTextTrim();
343 			try {
344 			    list1.add(InternalValue.valueOf(defValue, defValType));
345 			} catch (IllegalArgumentException e) {
346 			    String msg = "invalid serialized node type definition [" + sntName + "]: invalid defaultValue: " + defValue;
347 			    log.error(msg, e);
348 			    throw new InvalidNodeTypeDefException(msg, e);
349 			}
350 		    }
351 		}
352 		if (!list1.isEmpty()) {
353 		    pd.setDefaultValues((InternalValue[]) list1.toArray(new InternalValue[list1.size()]));
354 		}
355 	    }
356 	    // autoCreate
357 	    String autoCreate = elem.getAttributeValue(AUTOCREATE_ATTRIBUTE);
358 	    if (autoCreate != null && autoCreate.length() > 0) {
359 		pd.setAutoCreate(Boolean.valueOf(autoCreate).booleanValue());
360 	    }
361 	    // mandatory
362 	    String mandatory = elem.getAttributeValue(MANDATORY_ATTRIBUTE);
363 	    if (mandatory != null && mandatory.length() > 0) {
364 		pd.setMandatory(Boolean.valueOf(mandatory).booleanValue());
365 	    }
366 	    // onParentVersion
367 	    String onVersion = elem.getAttributeValue(ONPARENTVERSION_ATTRIBUTE);
368 	    if (onVersion != null && onVersion.length() > 0) {
369 		try {
370 		    pd.setOnParentVersion(OnParentVersionAction.valueFromName(onVersion));
371 		} catch (IllegalArgumentException e) {
372 		    String msg = "invalid serialized node type definition [" + sntName + "]: invalid onVersion: " + onVersion;
373 		    log.error(msg, e);
374 		    throw new InvalidNodeTypeDefException(msg, e);
375 		}
376 	    }
377 	    // protected
378 	    String writeProtected = elem.getAttributeValue(PROTECTED_ATTRIBUTE);
379 	    if (writeProtected != null && writeProtected.length() > 0) {
380 		pd.setProtected(Boolean.valueOf(writeProtected).booleanValue());
381 	    }
382 	    // primaryItem
383 	    String primaryItem = elem.getAttributeValue(PRIMARYITEM_ATTRIBUTE);
384 	    if (primaryItem != null && primaryItem.length() > 0) {
385 		pd.setPrimaryItem(Boolean.valueOf(primaryItem).booleanValue());
386 	    }
387 	    // multiple
388 	    String multiple = elem.getAttributeValue(MULTIPLE_ATTRIBUTE);
389 	    if (multiple != null && multiple.length() > 0) {
390 		pd.setMultiple(Boolean.valueOf(multiple).booleanValue());
391 	    }
392 
393 	    list.add(pd);
394 	}
395 	if (!list.isEmpty()) {
396 	    ntDef.setPropertyDefs((PropDef[]) list.toArray(new PropDef[list.size()]));
397 	}
398 
399 	// child-node definitions
400 	list.clear();
401 	iter = ntElem.getChildren(CHILDNODEDEF_ELEMENT).iterator();
402 	while (iter.hasNext()) {
403 	    Element elem = (Element) iter.next();
404 	    ChildNodeDef cnd = new ChildNodeDef();
405 	    // declaring node type
406 	    cnd.setDeclaringNodeType(qntName);
407 	    // name
408 	    String nodeName = elem.getAttributeValue(NAME_ATTRIBUTE);
409 	    if (nodeName != null && !nodeName.equals(WILDCARD)) {
410 		try {
411 		    cnd.setName(QName.fromJCRName(nodeName, nsResolver));
412 		} catch (BaseException e) {
413 		    String msg = "invalid serialized node type definition [" + sntName + "]: invalid child node name: " + nodeName;
414 		    log.error(msg, e);
415 		    throw new InvalidNodeTypeDefException(msg, e);
416 		}
417 	    } else {
418 		cnd.setName(ChildItemDef.ANY_NAME);
419 	    }
420 	    // requiredPrimaryTypes
421 	    Element reqTtypesElem = elem.getChild(REQUIREDPRIMARYTYPES_ELEMENT);
422 	    if (reqTtypesElem != null) {
423 		ArrayList list1 = new ArrayList();
424 		Iterator iter1 = reqTtypesElem.getChildren(REQUIREDPRIMARYTYPE_ELEMENT).iterator();
425 		while (iter1.hasNext()) {
426 		    Element typeElem = (Element) iter1.next();
427 		    Filter filter = new ContentFilter(ContentFilter.TEXT | ContentFilter.CDATA);
428 		    List content = typeElem.getContent(filter);
429 		    if (!content.isEmpty()) {
430 			String name = typeElem.getTextTrim();
431 			try {
432 			    list1.add(QName.fromJCRName(name, nsResolver));
433 			} catch (BaseException e) {
434 			    String msg = "invalid serialized node type definition [" + sntName + "]: invalid requiredPrimaryType: " + name;
435 			    log.error(msg, e);
436 			    throw new InvalidNodeTypeDefException(msg, e);
437 			}
438 		    }
439 		}
440 		if (!list1.isEmpty()) {
441 		    cnd.setRequiredPrimaryTypes((QName[]) list1.toArray(new QName[list1.size()]));
442 		}
443 	    }
444 	    // defaultPrimaryType
445 	    String defaultPrimaryType = elem.getAttributeValue(DEFAULTPRIMARYTYPE_ATTRIBUTE);
446 	    if (defaultPrimaryType != null && defaultPrimaryType.length() > 0) {
447 		try {
448 		    cnd.setDefaultPrimaryType(QName.fromJCRName(defaultPrimaryType, nsResolver));
449 		} catch (BaseException e) {
450 		    String msg = "invalid serialized node type definition [" + sntName + "]: invalid defaultPrimaryType: " + defaultPrimaryType;
451 		    log.error(msg, e);
452 		    throw new InvalidNodeTypeDefException(msg, e);
453 		}
454 	    }
455 	    // autoCreate
456 	    String autoCreate = elem.getAttributeValue(AUTOCREATE_ATTRIBUTE);
457 	    if (autoCreate != null && autoCreate.length() > 0) {
458 		cnd.setAutoCreate(Boolean.valueOf(autoCreate).booleanValue());
459 	    }
460 	    // mandatory
461 	    String mandatory = elem.getAttributeValue(MANDATORY_ATTRIBUTE);
462 	    if (mandatory != null && mandatory.length() > 0) {
463 		cnd.setMandatory(Boolean.valueOf(mandatory).booleanValue());
464 	    }
465 	    // onParentVersion
466 	    String onVersion = elem.getAttributeValue(ONPARENTVERSION_ATTRIBUTE);
467 	    if (onVersion != null && onVersion.length() > 0) {
468 		try {
469 		    cnd.setOnParentVersion(OnParentVersionAction.valueFromName(onVersion));
470 		} catch (IllegalArgumentException e) {
471 		    String msg = "invalid serialized node type definition [" + sntName + "]: invalid onVersion: " + onVersion;
472 		    log.error(msg, e);
473 		    throw new InvalidNodeTypeDefException(msg, e);
474 		}
475 	    }
476 	    // protected
477 	    String writeProtected = elem.getAttributeValue(PROTECTED_ATTRIBUTE);
478 	    if (writeProtected != null && writeProtected.length() > 0) {
479 		cnd.setProtected(Boolean.valueOf(writeProtected).booleanValue());
480 	    }
481 	    // primaryItem
482 	    String primaryItem = elem.getAttributeValue(PRIMARYITEM_ATTRIBUTE);
483 	    if (primaryItem != null && primaryItem.length() > 0) {
484 		cnd.setPrimaryItem(Boolean.valueOf(primaryItem).booleanValue());
485 	    }
486 	    // sameNameSibs
487 	    String sameNameSibs = elem.getAttributeValue(SAMENAMESIBS_ATTRIBUTE);
488 	    if (sameNameSibs != null && sameNameSibs.length() > 0) {
489 		cnd.setAllowSameNameSibs(Boolean.valueOf(sameNameSibs).booleanValue());
490 	    }
491 
492 	    list.add(cnd);
493 	}
494 	if (!list.isEmpty()) {
495 	    ntDef.setChildNodeDefs((ChildNodeDef[]) list.toArray(new ChildNodeDef[list.size()]));
496 	}
497 
498 	return ntDef;
499     }

501     private void writeDef(NodeTypeDef ntd, Element ntElem, NamespaceResolver nsResolver)
502 	    throws RepositoryException {
503 	try {
504 	    // name
505 	    ntElem.setAttribute(NAME_ATTRIBUTE, ntd.getName().toJCRName(nsResolver));
506 
507 	    // supertypes
508 	    QName[] qNames = ntd.getSupertypes();
509 	    if (qNames.length != 0) {
510 		Element typesElem = new Element(SUPERTYPES_ELEMENT);
511 		ntElem.addContent(typesElem);
512 		for (int i = 0; i < qNames.length; i++) {
513 		    Element typeElem = new Element(SUPERTYPE_ELEMENT);
514 		    typesElem.addContent(typeElem);
515 		    typeElem.setText(qNames[i].toJCRName(nsResolver));
516 		}
517 	    }
518 
519 	    // mixin
520 	    ntElem.setAttribute(MIXIN_ATTRIBUTE, Boolean.toString(ntd.isMixin()));
521 
522 	    // orderableChildNodes
523 	    ntElem.setAttribute(ORDERABLECHILDNODES_ATTRIBUTE, Boolean.toString(ntd.hasOrderableChildNodes()));
524 
525 	    // property definitions
526 	    PropDef[] pda = ntd.getPropertyDefs();
527 	    for (int i = 0; i < pda.length; i++) {
528 		PropDef pd = pda[i];
529 		Element elem = new Element(PROPERTYDEF_ELEMENT);
530 		ntElem.addContent(elem);
531 
532 		// name
533 		String name = pd.definesResidual() ? WILDCARD : pd.getName().toJCRName(nsResolver);
534 		elem.setAttribute(NAME_ATTRIBUTE, name);
535 		// type
536 		elem.setAttribute(TYPE_ATTRIBUTE, PropertyType.nameFromValue(pd.getRequiredType()));
537 		// valueConstraints
538 		ValueConstraint[] vca = pd.getValueConstraints();
539 		if (vca != null && vca.length != 0) {
540 		    Element constraintsElem = new Element(VALUECONSTRAINTS_ELEMENT);
541 		    elem.addContent(constraintsElem);
542 		    for (int j = 0; j < vca.length; j++) {
543 			Element constraintElem = new Element(VALUECONSTRAINTS_ELEMENT);
544 			constraintsElem.addContent(constraintElem);
545 			constraintElem.setText(vca[j].getDefinition());
546 		    }
547 		}
548 		// defaultValues
549 		InternalValue[] defVals = pd.getDefaultValues();
550 		if (defVals != null && defVals.length != 0) {
551 		    Element valuesElem = new Element(DEFAULTVALUES_ELEMENT);
552 		    elem.addContent(valuesElem);
553 		    for (int j = 0; j < defVals.length; j++) {
554 			Element valueElem = new Element(DEFAULTVALUE_ELEMENT);
555 			valuesElem.addContent(valueElem);
556 			valueElem.setText(defVals[j].toString());
557 		    }
558 		}
559 		// autoCreate
560 		String autoCreate = elem.getAttributeValue(AUTOCREATE_ATTRIBUTE);
561 		if (autoCreate != null && autoCreate.length() > 0) {
562 		    pd.setAutoCreate(Boolean.valueOf(autoCreate).booleanValue());
563 		}
564 		// mandatory
565 		elem.setAttribute(MANDATORY_ATTRIBUTE, Boolean.toString(pd.isMandatory()));
566 		// onParentVersion
567 		elem.setAttribute(ONPARENTVERSION_ATTRIBUTE, OnParentVersionAction.nameFromValue(pd.getOnParentVersion()));
568 		// protected
569 		elem.setAttribute(PROTECTED_ATTRIBUTE, Boolean.toString(pd.isProtected()));
570 		// primaryItem
571 		elem.setAttribute(PRIMARYITEM_ATTRIBUTE, Boolean.toString(pd.isPrimaryItem()));
572 		// multiple
573 		elem.setAttribute(MULTIPLE_ATTRIBUTE, Boolean.toString(pd.isMultiple()));
574 	    }
575 
576 	    // child-node definitions
577 	    ChildNodeDef[] nda = ntd.getChildNodeDefs();
578 	    for (int i = 0; i < nda.length; i++) {
579 		ChildNodeDef nd = nda[i];
580 		Element elem = new Element(CHILDNODEDEF_ELEMENT);
581 		ntElem.addContent(elem);
582 
583 		// name
584 		String name = nd.definesResidual() ? WILDCARD : nd.getName().toJCRName(nsResolver);
585 		elem.setAttribute(NAME_ATTRIBUTE, name);
