83      /***
84       * @param in
85       * @throws IOException
86       * @throws InvalidNodeTypeDefException
87       * @throws RepositoryException
88       */
89      void load(InputStream in)
90  	    throws IOException, InvalidNodeTypeDefException, RepositoryException {
91  	SAXBuilder builder = new SAXBuilder();
92  	Element root = null;
93  	try {
94  	    Document doc = builder.build(in);
95  	    root = doc.getRootElement();
96  	} catch (JDOMException jde) {
97  	    String msg = "internal error: failed to parse persistent node type definitions";
98  	    log.error(msg, jde);
99  	    throw new RepositoryException(msg, jde);
100 	}
101 	// read namespace declarations of root element
102 	Iterator nsIter = root.getAdditionalNamespaces().iterator();
103 	final HashMap prefixToURI = new HashMap();
104 	final HashMap uriToPrefix = new HashMap();
105 	while (nsIter.hasNext()) {
106 	    Namespace ns = (Namespace) nsIter.next();
107 	    prefixToURI.put(ns.getPrefix(), ns.getURI());
108 	    uriToPrefix.put(ns.getURI(), ns.getPrefix());
109 	}
110 	// add default namespace (empty uri)
111 	prefixToURI.put(NamespaceRegistryImpl.NS_EMPTY_PREFIX, NamespaceRegistryImpl.NS_DEFAULT_URI);
112 	uriToPrefix.put(NamespaceRegistryImpl.NS_DEFAULT_URI, NamespaceRegistryImpl.NS_EMPTY_PREFIX);
113 
114 	NamespaceResolver nsResolver = new NamespaceResolver() {
115 	    public String getURI(String prefix) throws NamespaceException {
116 		if (!prefixToURI.containsKey(prefix)) {
117 		    throw new NamespaceException(prefix + ": unknown prefix");
118 		}
119 		return (String) prefixToURI.get(prefix);
120 	    }
121 
122 	    public String getPrefix(String uri) throws NamespaceException {
123 		if (!uriToPrefix.containsKey(uri)) {
124 		    throw new NamespaceException(uri + ": unknown URI");
125 		}
126 		return (String) uriToPrefix.get(uri);
127 	    }
128 	};
129 
130 	// read definitions
131 	Iterator iter = root.getChildren(NODETYPE_ELEMENT).iterator();
132 	while (iter.hasNext()) {
133 	    NodeTypeDef ntDef = readDef((Element) iter.next(), nsResolver);
134 	    add(ntDef);
135 	}
136     }

138     /***
139      * @param out
140      * @param nsReg
141      * @throws IOException
142      * @throws RepositoryException
143      */
144     void store(OutputStream out, NamespaceRegistryImpl nsReg)
145 	    throws IOException, RepositoryException {
146 	Element root = new Element(ROOT_ELEMENT);
147 
148 	// namespace declarations
149 	String[] prefixes = nsReg.getPrefixes();
150 	for (int i = 0; i < prefixes.length; i++) {
151 	    String prefix = prefixes[i];
152 	    if ("".equals(prefix)) {
153 		continue;
154 	    }
155 	    String uri = nsReg.getURI(prefix);
156 	    root.addNamespaceDeclaration(Namespace.getNamespace(prefix, uri));
157 	}
158 
159 	// node type definitions
160 	Iterator iter = all().iterator();
161 	while (iter.hasNext()) {
162 	    NodeTypeDef ntd = (NodeTypeDef) iter.next();
163 	    Element ntElem = new Element(NODETYPE_ELEMENT);
164 	    writeDef(ntd, ntElem, nsReg);
165 	    root.addContent(ntElem);
166 	}
167 
168 	XMLOutputter serializer = new XMLOutputter(Format.getPrettyFormat());
169 	serializer.output(new Document(root), out);
170     }

217     //------------------------------------------------------< private methods >
218     private NodeTypeDef readDef(Element ntElem, NamespaceResolver nsResolver)
219 	    throws InvalidNodeTypeDefException {
220 	String sntName = ntElem.getAttributeValue(NAME_ATTRIBUTE);
221 	NodeTypeDef ntDef = new NodeTypeDef();
222 
223 	// name
224 	QName qntName;
225 	try {
226 	    qntName = QName.fromJCRName(sntName, nsResolver);
227 	} catch (BaseException e) {
228 	    String msg = "invalid serialized node type definition [" + sntName + "]: invalid node type name: " + sntName;
229 	    log.error(msg, e);
230 	    throw new InvalidNodeTypeDefException(msg, e);
231 	}
232 	ntDef.setName(qntName);
233 
234 	// supertypes
235 	ArrayList list = new ArrayList();
236 	Element typesElem = ntElem.getChild(SUPERTYPES_ELEMENT);
237 	if (typesElem != null) {
238 	    Iterator iter = typesElem.getChildren(SUPERTYPE_ELEMENT).iterator();
239 	    while (iter.hasNext()) {
240 		Element typeElem = (Element) iter.next();
241 		Filter filter = new ContentFilter(ContentFilter.TEXT | ContentFilter.CDATA);
242 		List content = typeElem.getContent(filter);
243 		if (!content.isEmpty()) {
244 		    String name = typeElem.getTextTrim();
245 		    try {
246 			list.add(QName.fromJCRName(name, nsResolver));
247 		    } catch (BaseException e) {
248 			String msg = "invalid serialized node type definition [" + sntName + "]: invalid supertype: " + name;
249 			log.error(msg, e);
250 			throw new InvalidNodeTypeDefException(msg, e);
251 		    }
252 		}
253 	    }
254 	    if (!list.isEmpty()) {
255 		ntDef.setSupertypes((QName[]) list.toArray(new QName[list.size()]));
256 	    }
257 	}
258 
259 	// mixin
260 	String mixin = ntElem.getAttributeValue(MIXIN_ATTRIBUTE);
261 	if (mixin != null && mixin.length() > 0) {
262 	    ntDef.setMixin(Boolean.valueOf(mixin).booleanValue());
263 	}
264 
265 	// orderableChildNodes
266 	String orderableChildNodes = ntElem.getAttributeValue(ORDERABLECHILDNODES_ATTRIBUTE);
267 	if (orderableChildNodes != null && orderableChildNodes.length() > 0) {
268 	    ntDef.setOrderableChildNodes(Boolean.valueOf(orderableChildNodes).booleanValue());
269 	}
270 
271 	// property definitions
272 	list.clear();
273 	Iterator iter = ntElem.getChildren(PROPERTYDEF_ELEMENT).iterator();
274 	while (iter.hasNext()) {
275 	    Element elem = (Element) iter.next();
276 	    PropDef pd = new PropDef();
277 	    // declaring node type
278 	    pd.setDeclaringNodeType(qntName);
279 	    // name
280 	    String propName = elem.getAttributeValue(NAME_ATTRIBUTE);
281 	    if (propName != null && propName.length() > 0) {
282 		try {
283 		    pd.setName(QName.fromJCRName(propName, nsResolver));
284 		} catch (BaseException e) {
285 		    String msg = "invalid serialized node type definition [" + sntName + "]: invalid property name: " + propName;
286 		    log.error(msg, e);
287 		    throw new InvalidNodeTypeDefException(msg, e);
288 		}
289 	    }
290 	    // type
291 	    String typeName = elem.getAttributeValue(TYPE_ATTRIBUTE);
292 	    int type = PropertyType.UNDEFINED;
293 	    if (typeName != null && typeName.length() > 0) {
294 		try {
295 		    type = PropertyType.valueFromName(typeName);
296 		    pd.setRequiredType(type);
297 		} catch (IllegalArgumentException e) {
298 		    String msg = "invalid serialized node type definition [" + sntName + "]: invalid type: " + typeName;
299 		    log.error(msg, e);
300 		    throw new InvalidNodeTypeDefException(msg, e);
301 		}
302 	    }
303 	    // valueConstraints
304 	    Element constraintsElem = elem.getChild(VALUECONSTRAINTS_ELEMENT);
305 	    if (constraintsElem != null) {
306 		ArrayList list1 = new ArrayList();
307 		Iterator iter1 = constraintsElem.getChildren(VALUECONSTRAINT_ELEMENT).iterator();
308 		while (iter1.hasNext()) {
309 		    Element constraintElem = (Element) iter1.next();
310 		    Filter filter = new ContentFilter(ContentFilter.TEXT | ContentFilter.CDATA);
311 		    List content = constraintElem.getContent(filter);
312 		    if (!content.isEmpty()) {
313 			String constraint = constraintElem.getTextTrim();
314 			try {
315 			    list1.add(ValueConstraint.create(type, constraint));
316 			} catch (InvalidConstraintException e) {
317 			    String msg = "invalid serialized node type definition [" + sntName + "]: invalid constraint: " + constraint;
318 			    log.error(msg, e);
319 			    throw new InvalidNodeTypeDefException(msg, e);
320 			}
321 		    }
322 		}
323 		if (!list1.isEmpty()) {
324 		    pd.setValueConstraints((ValueConstraint[]) list1.toArray(new ValueConstraint[list1.size()]));
325 		}
326 	    }
327 	    // defaultValues
328 	    Element defValuesElem = elem.getChild(DEFAULTVALUES_ELEMENT);
329 	    if (defValuesElem != null) {
330 		int defValType = (type == PropertyType.UNDEFINED) ? PropertyType.STRING : type;
331 		ArrayList list1 = new ArrayList();
332 		Iterator iter1 = defValuesElem.getChildren(DEFAULTVALUE_ELEMENT).iterator();
333 		while (iter1.hasNext()) {
334 		    Element valueElem = (Element) iter1.next();
335 		    Filter filter = new ContentFilter(ContentFilter.TEXT | ContentFilter.CDATA);
336 		    List content = valueElem.getContent(filter);
337 		    if (!content.isEmpty()) {
338 			String defValue = valueElem.getTextTrim();
339 			try {
340 			    list1.add(InternalValue.valueOf(defValue, defValType));
341 			} catch (IllegalArgumentException e) {
342 			    String msg = "invalid serialized node type definition [" + sntName + "]: invalid defaultValue: " + defValue;
343 			    log.error(msg, e);
344 			    throw new InvalidNodeTypeDefException(msg, e);
345 			}
346 		    }
347 		}
348 		if (!list1.isEmpty()) {
349 		    pd.setDefaultValues((InternalValue[]) list1.toArray(new InternalValue[list1.size()]));
350 		}
351 	    }
352 	    // autoCreate
353 	    String autoCreate = elem.getAttributeValue(AUTOCREATE_ATTRIBUTE);
354 	    if (autoCreate != null && autoCreate.length() > 0) {
355 		pd.setAutoCreate(Boolean.valueOf(autoCreate).booleanValue());
356 	    }
357 	    // mandatory
358 	    String mandatory = elem.getAttributeValue(MANDATORY_ATTRIBUTE);
359 	    if (mandatory != null && mandatory.length() > 0) {
360 		pd.setMandatory(Boolean.valueOf(mandatory).booleanValue());
361 	    }
362 	    // onParentVersion
363 	    String onVersion = elem.getAttributeValue(ONPARENTVERSION_ATTRIBUTE);
364 	    if (onVersion != null && onVersion.length() > 0) {
365 		try {
366 		    pd.setOnParentVersion(OnParentVersionAction.valueFromName(onVersion));
367 		} catch (IllegalArgumentException e) {
368 		    String msg = "invalid serialized node type definition [" + sntName + "]: invalid onVersion: " + onVersion;
369 		    log.error(msg, e);
370 		    throw new InvalidNodeTypeDefException(msg, e);
371 		}
372 	    }
373 	    // protected
374 	    String writeProtected = elem.getAttributeValue(PROTECTED_ATTRIBUTE);
375 	    if (writeProtected != null && writeProtected.length() > 0) {
376 		pd.setProtected(Boolean.valueOf(writeProtected).booleanValue());
377 	    }
378 	    // primaryItem
379 	    String primaryItem = elem.getAttributeValue(PRIMARYITEM_ATTRIBUTE);
380 	    if (primaryItem != null && primaryItem.length() > 0) {
381 		pd.setPrimaryItem(Boolean.valueOf(primaryItem).booleanValue());
382 	    }
383 	    // multiple
384 	    String multiple = elem.getAttributeValue(MULTIPLE_ATTRIBUTE);
385 	    if (multiple != null && multiple.length() > 0) {
386 		pd.setMultiple(Boolean.valueOf(multiple).booleanValue());
387 	    }
388 
389 	    list.add(pd);
390 	}
391 	if (!list.isEmpty()) {
392 	    ntDef.setPropertyDefs((PropDef[]) list.toArray(new PropDef[list.size()]));
393 	}
394 
395 	// child-node definitions
396 	list.clear();
397 	iter = ntElem.getChildren(CHILDNODEDEF_ELEMENT).iterator();
398 	while (iter.hasNext()) {
399 	    Element elem = (Element) iter.next();
400 	    ChildNodeDef cnd = new ChildNodeDef();
401 	    // declaring node type
402 	    cnd.setDeclaringNodeType(qntName);
403 	    // name
404 	    String nodeName = elem.getAttributeValue(NAME_ATTRIBUTE);
405 	    if (nodeName != null && nodeName.length() > 0) {
406 		try {
407 		    cnd.setName(QName.fromJCRName(nodeName, nsResolver));
408 		} catch (BaseException e) {
409 		    String msg = "invalid serialized node type definition [" + sntName + "]: invalid child node name: " + nodeName;
410 		    log.error(msg, e);
411 		    throw new InvalidNodeTypeDefException(msg, e);
412 		}
413 	    }
414 	    // requiredPrimaryTypes
415 	    Element reqTtypesElem = elem.getChild(REQUIREDPRIMARYTYPES_ELEMENT);
416 	    if (reqTtypesElem != null) {
417 		ArrayList list1 = new ArrayList();
418 		Iterator iter1 = reqTtypesElem.getChildren(REQUIREDPRIMARYTYPE_ELEMENT).iterator();
419 		while (iter1.hasNext()) {
420 		    Element typeElem = (Element) iter1.next();
421 		    Filter filter = new ContentFilter(ContentFilter.TEXT | ContentFilter.CDATA);
422 		    List content = typeElem.getContent(filter);
423 		    if (!content.isEmpty()) {
424 			String name = typeElem.getTextTrim();
425 			try {
426 			    list1.add(QName.fromJCRName(name, nsResolver));
427 			} catch (BaseException e) {
428 			    String msg = "invalid serialized node type definition [" + sntName + "]: invalid requiredPrimaryType: " + name;
429 			    log.error(msg, e);
430 			    throw new InvalidNodeTypeDefException(msg, e);
431 			}
432 		    }
433 		}
434 		if (!list1.isEmpty()) {
435 		    cnd.setRequiredPrimaryTypes((QName[]) list1.toArray(new QName[list1.size()]));
436 		}
437 	    }
438 	    // defaultPrimaryType
439 	    String defaultPrimaryType = elem.getAttributeValue(DEFAULTPRIMARYTYPE_ATTRIBUTE);
440 	    if (defaultPrimaryType != null && defaultPrimaryType.length() > 0) {
441 		try {
442 		    cnd.setDefaultPrimaryType(QName.fromJCRName(defaultPrimaryType, nsResolver));
443 		} catch (BaseException e) {
444 		    String msg = "invalid serialized node type definition [" + sntName + "]: invalid defaultPrimaryType: " + defaultPrimaryType;
445 		    log.error(msg, e);
446 		    throw new InvalidNodeTypeDefException(msg, e);
447 		}
448 	    }
449 	    // autoCreate
450 	    String autoCreate = elem.getAttributeValue(AUTOCREATE_ATTRIBUTE);
451 	    if (autoCreate != null && autoCreate.length() > 0) {
452 		cnd.setAutoCreate(Boolean.valueOf(autoCreate).booleanValue());
453 	    }
454 	    // mandatory
455 	    String mandatory = elem.getAttributeValue(MANDATORY_ATTRIBUTE);
456 	    if (mandatory != null && mandatory.length() > 0) {
457 		cnd.setMandatory(Boolean.valueOf(mandatory).booleanValue());
458 	    }
459 	    // onParentVersion
460 	    String onVersion = elem.getAttributeValue(ONPARENTVERSION_ATTRIBUTE);
461 	    if (onVersion != null && onVersion.length() > 0) {
462 		try {
463 		    cnd.setOnParentVersion(OnParentVersionAction.valueFromName(onVersion));
464 		} catch (IllegalArgumentException e) {
465 		    String msg = "invalid serialized node type definition [" + sntName + "]: invalid onVersion: " + onVersion;
466 		    log.error(msg, e);
467 		    throw new InvalidNodeTypeDefException(msg, e);
468 		}
469 	    }
470 	    // protected
471 	    String writeProtected = elem.getAttributeValue(PROTECTED_ATTRIBUTE);
472 	    if (writeProtected != null && writeProtected.length() > 0) {
473 		cnd.setProtected(Boolean.valueOf(writeProtected).booleanValue());
474 	    }
475 	    // primaryItem
476 	    String primaryItem = elem.getAttributeValue(PRIMARYITEM_ATTRIBUTE);
477 	    if (primaryItem != null && primaryItem.length() > 0) {
478 		cnd.setPrimaryItem(Boolean.valueOf(primaryItem).booleanValue());
479 	    }
480 	    // sameNameSibs
481 	    String sameNameSibs = elem.getAttributeValue(SAMENAMESIBS_ATTRIBUTE);
482 	    if (sameNameSibs != null && sameNameSibs.length() > 0) {
483 		cnd.setAllowSameNameSibs(Boolean.valueOf(sameNameSibs).booleanValue());
484 	    }
485 
486 	    list.add(cnd);
487 	}
488 	if (!list.isEmpty()) {
489 	    ntDef.setChildNodeDefs((ChildNodeDef[]) list.toArray(new ChildNodeDef[list.size()]));
490 	}
491 
492 	return ntDef;
493     }

495     private void writeDef(NodeTypeDef ntd, Element ntElem, NamespaceResolver nsResolver)
496 	    throws RepositoryException {
497 	try {
498 	    // name
499 	    ntElem.setAttribute(NAME_ATTRIBUTE, ntd.getName().toJCRName(nsResolver));
500 
501 	    // supertypes
502 	    QName[] qNames = ntd.getSupertypes();
503 	    if (qNames.length != 0) {
504 		Element typesElem = new Element(SUPERTYPES_ELEMENT);
505 		ntElem.addContent(typesElem);
506 		for (int i = 0; i < qNames.length; i++) {
507 		    Element typeElem = new Element(SUPERTYPE_ELEMENT);
508 		    typesElem.addContent(typeElem);
509 		    typeElem.setText(qNames[i].toJCRName(nsResolver));
510 		}
511 	    }
512 
513 	    // mixin
514 	    ntElem.setAttribute(MIXIN_ATTRIBUTE, Boolean.toString(ntd.isMixin()));
515 
516 	    // orderableChildNodes
517 	    ntElem.setAttribute(ORDERABLECHILDNODES_ATTRIBUTE, Boolean.toString(ntd.hasOrderableChildNodes()));
518 
519 	    // property definitions
520 	    PropDef[] pda = ntd.getPropertyDefs();
521 	    for (int i = 0; i < pda.length; i++) {
522 		PropDef pd = pda[i];
523 		Element elem = new Element(PROPERTYDEF_ELEMENT);
524 		ntElem.addContent(elem);
525 
526 		// name
527 		String name = pd.getName() == null ? "" : pd.getName().toJCRName(nsResolver);
528 		elem.setAttribute(NAME_ATTRIBUTE, name);
529 		// type
530 		elem.setAttribute(TYPE_ATTRIBUTE, PropertyType.nameFromValue(pd.getRequiredType()));
531 		// valueConstraints
532 		ValueConstraint[] vca = pd.getValueConstraints();
533 		if (vca != null && vca.length != 0) {
534 		    Element constraintsElem = new Element(VALUECONSTRAINTS_ELEMENT);
535 		    elem.addContent(constraintsElem);
536 		    for (int j = 0; j < vca.length; j++) {
537 			Element constraintElem = new Element(VALUECONSTRAINTS_ELEMENT);
538 			constraintsElem.addContent(constraintElem);
539 			constraintElem.setText(vca[j].getDefinition());
540 		    }
541 		}
542 		// defaultValues
543 		InternalValue[] defVals = pd.getDefaultValues();
544 		if (defVals != null && defVals.length != 0) {
545 		    Element valuesElem = new Element(DEFAULTVALUES_ELEMENT);
546 		    elem.addContent(valuesElem);
547 		    for (int j = 0; j < defVals.length; j++) {
548 			Element valueElem = new Element(DEFAULTVALUE_ELEMENT);
549 			valuesElem.addContent(valueElem);
550 			valueElem.setText(defVals[j].toString());
551 		    }
552 		}
553 		// autoCreate
554 		String autoCreate = elem.getAttributeValue(AUTOCREATE_ATTRIBUTE);
555 		if (autoCreate != null && autoCreate.length() > 0) {
556 		    pd.setAutoCreate(Boolean.valueOf(autoCreate).booleanValue());
557 		}
558 		// mandatory
559 		elem.setAttribute(MANDATORY_ATTRIBUTE, Boolean.toString(pd.isMandatory()));
560 		// onParentVersion
561 		elem.setAttribute(ONPARENTVERSION_ATTRIBUTE, OnParentVersionAction.nameFromValue(pd.getOnParentVersion()));
562 		// protected
563 		elem.setAttribute(PROTECTED_ATTRIBUTE, Boolean.toString(pd.isProtected()));
564 		// primaryItem
565 		elem.setAttribute(PRIMARYITEM_ATTRIBUTE, Boolean.toString(pd.isPrimaryItem()));
566 		// multiple
567 		elem.setAttribute(MULTIPLE_ATTRIBUTE, Boolean.toString(pd.isMultiple()));
568 	    }
569 
570 	    // child-node definitions
571 	    ChildNodeDef[] nda = ntd.getChildNodeDefs();
572 	    for (int i = 0; i < nda.length; i++) {
573 		ChildNodeDef nd = nda[i];
574 		Element elem = new Element(CHILDNODEDEF_ELEMENT);
575 		ntElem.addContent(elem);
576 
577 		// name
578 		String name = nd.getName() == null ? "" : nd.getName().toJCRName(nsResolver);
579 		elem.setAttribute(NAME_ATTRIBUTE, name);
580 		// requiredPrimaryTypes
581 		qNames = nd.getRequiredPrimaryTypes();
582 		if (qNames.length != 0) {
583 		    Element typesElem = new Element(REQUIREDPRIMARYTYPES_ELEMENT);
584 		    elem.addContent(typesElem);
