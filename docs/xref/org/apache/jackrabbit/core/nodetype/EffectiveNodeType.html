32  /***
33   * An <code>EffectiveNodeType</code> represents one or more
34   * <code>NodeType</code>s as one 'effective' node type where inheritance
35   * is resolved.
36   * <p/>
37   * Instances of <code>EffectiveNodeType</code> are immutable.
38   *
39   * @author Stefan Guggisberg
40   * @version $Revision: 1.39 $, $Date: 2004/09/09 15:23:43 $
41   */
42  public class EffectiveNodeType implements Cloneable {
43      private static Logger log = Logger.getLogger(EffectiveNodeType.class);

46      private final NodeTypeRegistry ntReg;
47

53      // (through inheritance) included node types.

55      // map of named item definitions

72      /***
73       * Factory method: creates an effective node type
74       * representation of an existing (i.e. registered) node type.
75       *
76       * @param ntReg
77       * @param nodeTypeName
78       * @return
79       * @throws NodeTypeConflictException
80       * @throws NoSuchNodeTypeException
81       */
82      static EffectiveNodeType create(NodeTypeRegistry ntReg, QName nodeTypeName)
83  	    throws NodeTypeConflictException, NoSuchNodeTypeException {
84  	return create(ntReg, ntReg.getNodeTypeDef(nodeTypeName));
85      }

87      /***
88       * Factory method: creates an effective node type
89       * representation of a node type definition. Whereas all referenced
90       * node types must exist (i.e. must be registered), the definition itself
91       * is not required to be registered.
92       *
93       * @param ntReg
94       * @param ntd
95       * @return
96       * @throws NodeTypeConflictException
97       * @throws NoSuchNodeTypeException
98       */
99      public static EffectiveNodeType create(NodeTypeRegistry ntReg, NodeTypeDef ntd)
100 	    throws NodeTypeConflictException, NoSuchNodeTypeException {
101 	// create empty effective node type instance
102 	EffectiveNodeType ent = new EffectiveNodeType(ntReg);
103 	QName ntName = ntd.getName();
104 
105 	// prepare new instance
106 	ent.mergedNodeTypes.add(ntName);
107 	ent.allNodeTypes.add(ntName);
108 
109 	ChildNodeDef[] cnda = ntd.getChildNodeDefs();
110 	for (int i = 0; i < cnda.length; i++) {
111 	    QName name = cnda[i].getName();
112 	    if (name == null) {
113 		// residual node definition
114 		ent.unnamedItemDefs.add(cnda[i]);
115 	    } else {
116 		// named node definition
117 		ent.namedItemDefs.put(name, cnda[i]);
118 	    }
119 	}
120 	PropDef[] pda = ntd.getPropertyDefs();
121 	for (int i = 0; i < pda.length; i++) {
122 	    QName name = pda[i].getName();
123 	    if (name == null) {
124 		// residual property definition
125 		ent.unnamedItemDefs.add(pda[i]);
126 	    } else {
127 		// named property definition
128 		ent.namedItemDefs.put(name, pda[i]);
129 	    }
130 	}
131 
132 	// resolve supertypes recursively
133 	QName[] supertypes = ntd.getSupertypes();
134 	if (supertypes != null && supertypes.length > 0) {
135 	    ent.internalMerge(ntReg.buildEffectiveNodeType(supertypes), true);
136 	}
137 
138 	// we're done
139 	return ent;
140     }

332     /***
333      * Tests if the value constraints defined in the property definition
334      * <code>pd</code> are satisfied by the the specified <code>values</code>.
335      * <p/>
336      * Note that the <i>protected</i> flag is not checked.
337      *
338      * @param pd
339      * @param values
340      * @throws ConstraintViolationException
341      * @throws ValueFormatException
342      */
343     public static void checkSetPropertyValueConstraints(PropDef pd, InternalValue[] values)
344 	    throws ConstraintViolationException, ValueFormatException, RepositoryException {
345 	// check multi-value flag
346 	if (!pd.isMultiple() && values != null && values.length > 1) {
347 	    throw new ConstraintViolationException("the property is not multi-valued");
348 	}
349 
350 	// check value constraints
351 	ValueConstraint[] constraints = pd.getValueConstraints();
352 	if (constraints != null && constraints.length != 0) {
353 	    for (int i = 0; i < constraints.length; i++) {
354 		if (values == null || values.length == 0) {
355 		    constraints[i].check(null);
356 		} else {
357 		    for (int j = 0; j < values.length; j++) {
358 			constraints[i].check(values[j]);
359 		    }
360 		}
361 	    }
362 	}
363     }

366      * @param name
367      * @throws ConstraintViolationException
368      */
369     public void checkAddNodeConstraints(QName name)
370 	    throws ConstraintViolationException {
371 	try {
372 	    getApplicableChildNodeDef(name, null);
373 	} catch (NoSuchNodeTypeException nsnte) {
374 	    String msg = "internal eror: inconsistent node type";
375 	    log.error(msg, nsnte);
376 	    throw new ConstraintViolationException(msg, nsnte);
377 	}
378     }

381      * @param name
382      * @param nodeTypeName
383      * @throws ConstraintViolationException
384      * @throws NoSuchNodeTypeException
385      */
386     public void checkAddNodeConstraints(QName name, QName nodeTypeName)
387 	    throws ConstraintViolationException, NoSuchNodeTypeException {
388 	ChildNodeDef nd = getApplicableChildNodeDef(name, nodeTypeName);
389 	if (nd.isProtected()) {
390 	    throw new ConstraintViolationException(name + " is protected");
391 	}
392 	if (nd.isAutoCreate()) {
393 	    throw new ConstraintViolationException(name + " is auto-created and can not be manually added");
394 	}
395     }

397     /***
398      * Returns the applicable child node definition for a child node with the
399      * specified name and node type.
400      *
401      * @param name
402      * @param nodeTypeName
403      * @return
404      * @throws NoSuchNodeTypeException
405      * @throws ConstraintViolationException if no applicable child node definition
406      *                                      could be found
407      */
408     public ChildNodeDef getApplicableChildNodeDef(QName name, QName nodeTypeName)
409 	    throws NoSuchNodeTypeException, ConstraintViolationException {
410 	ChildItemDef def = (ChildItemDef) namedItemDefs.get(name);
411 	if (def == null) {
412 	    // no item with that name defined;
413 	    // try residual node definitions
414 	    ChildNodeDef[] nda = getUnnamedNodeDefs();
415 	    for (int i = 0; i < nda.length; i++) {
416 		ChildNodeDef nd = nda[i];
417 		if (nodeTypeName != null) {
418 		    try {
419 			// check node type constraint
420 			checkRequiredPrimaryType(nodeTypeName, nd.getRequiredPrimaryTypes());
421 		    } catch (ConstraintViolationException e) {
422 			// ignore and try next
423 			continue;
424 		    }
425 		    return nd;
426 		} else {
427 		    // since no node type has been specified for the new node,
428 		    // it must be determined from the default node type;
429 		    if (nd.getDefaultPrimaryType() != null) {
430 			// found residual node definition with default node type
431 			return nd;
432 		    }
433 		}
434 	    }
435 	} else {
436 	    if (def.definesNode()) {
437 		ChildNodeDef nd = (ChildNodeDef) def;
438 		// node definition with that name exists
439 		if (nodeTypeName != null) {
440 		    // check node type constraints
441 		    checkRequiredPrimaryType(nodeTypeName, nd.getRequiredPrimaryTypes());
442 		    return nd;
443 		} else {
444 		    if (nd.getDefaultPrimaryType() == null) {
445 			// no default node type defined
446 			throw new ConstraintViolationException("node type for " + name + " can not be determined");
447 		    }
448 		}
449 	    }
450 	}
451 	throw new ConstraintViolationException("no matching child node definition found for " + name);
452     }

454     /***
455      * Returns the applicable property definition for a property with the
456      * specified name and type.
457      *
458      * @param name
459      * @param type
460      * @param multiValued
461      * @return
462      * @throws ConstraintViolationException if no applicable property definition
463      *                                      could be found
464      */
465     public PropDef getApplicablePropertyDef(QName name, int type, boolean multiValued)
466 	    throws ConstraintViolationException {
467 	ChildItemDef def = (ChildItemDef) namedItemDefs.get(name);
468 	if (def == null) {
469 	    // no item with that name defined;
470 	    // try residual property definitions
471 	    PropDef[] pda = getUnnamedPropDefs();
472 	    for (int i = 0; i < pda.length; i++) {
473 		PropDef pd = pda[i];
474 		int reqType = pd.getRequiredType();
475 		// match type
476 		if (reqType == PropertyType.UNDEFINED
477 			|| type == PropertyType.UNDEFINED
478 			|| reqType == type) {
479 		    // match multiValued flag
480 		    if (multiValued == pd.isMultiple()) {
481 			// found match
482 			return pd;
483 		    }
484 		}
485 	    }
486 	} else {
487 	    if (!def.definesNode()) {
488 		PropDef pd = (PropDef) def;
489 		int reqType = pd.getRequiredType();
490 		// property definition with that name exists
491 		// match type
492 		if (reqType == PropertyType.UNDEFINED
493 			|| type == PropertyType.UNDEFINED
494 			|| reqType == type) {
495 		    // match multiValued flag
496 		    if (multiValued == pd.isMultiple()) {
497 			// found match
498 			return pd;
499 		    }
500 		}
501 	    }
502 	}
503 	throw new ConstraintViolationException("no matching property definition found for " + name);
504     }

522     /***
523      * @param nodeTypeName
524      * @param requiredPrimaryTypes
525      * @throws ConstraintViolationException
526      * @throws NoSuchNodeTypeException
527      */
528     public void checkRequiredPrimaryType(QName nodeTypeName, QName[] requiredPrimaryTypes)
529 	    throws ConstraintViolationException, NoSuchNodeTypeException {
530 	if (requiredPrimaryTypes == null) {
531 	    // no constraint
532 	    return;
533 	}
534 	EffectiveNodeType ent;
535 	try {
536 	    ent = ntReg.getEffectiveNodeType(nodeTypeName);
537 	} catch (RepositoryException re) {
538 	    String msg = "failed to check node type constraint";
539 	    log.error(msg, re);
540 	    throw new ConstraintViolationException(msg, re);
541 	}
542 	for (int i = 0; i < requiredPrimaryTypes.length; i++) {
543 	    if (!ent.includesNodeType(requiredPrimaryTypes[i])) {
544 		throw new ConstraintViolationException("node type constraint not satisfied: " + requiredPrimaryTypes[i]);
545 	    }
546 	}
547     }

549     /***
550      * Merges another <code>EffectiveNodeType</code> with this one.
551      * Checks for merge conflicts.
552      *
553      * @param other
554      * @return
555      * @throws NodeTypeConflictException
556      */
557     EffectiveNodeType merge(EffectiveNodeType other)
558 	    throws NodeTypeConflictException {
559 	// create a clone of this instance and perform the merge on
560 	// the 'clone' to avoid a potentially inconsistant state
561 	// of this instance if an exception is thrown during
562 	// the merge.
563 	EffectiveNodeType copy = (EffectiveNodeType) clone();
564 	copy.internalMerge(other, false);
565 	return copy;
566     }

568     /***
569      * Internal helper method which merges another <code>EffectiveNodeType</code>
570      * instance with <i>this</i> instance.
571      * <p/>
572      * Warning: This instance might be in an inconsistent state if an exception
573      * is thrown.
574      *
575      * @param other
576      * @param supertype true if the merge is a result of inheritance, i.e. <code>other</code>
577      *                  represents one or more supertypes of this instance; otherwise false, i.e.
578      *                  the merge is the result of an explicit aggregation
579      * @throws NodeTypeConflictException
580      */
581     private synchronized void internalMerge(EffectiveNodeType other, boolean supertype)
582 	    throws NodeTypeConflictException {
583 	QName[] nta = other.getAllNodeTypes();
584 	int includedCount = 0;
585 	for (int i = 0; i < nta.length; i++) {
586 	    if (includesNodeType(nta[i])) {
587 		// redundant node type
588 		log.warn("node type '" + nta[i] + "' is already contained.");
589 		includedCount++;
590 	    }
591 	}
592 	if (includedCount == nta.length) {
593 	    // total overlap, ignore
594 	    return;
595 	}
596 
597 	// named item definitions
598 	ChildItemDef[] defs = other.getNamedItemDefs();
599 	for (int i = 0; i < defs.length; i++) {
600 	    ChildItemDef def = defs[i];
601 	    if (includesNodeType(def.getDeclaringNodeType())) {
602 		// ignore redundant definitions
603 		continue;
604 	    }
605 	    QName name = def.getName();
606 	    ChildItemDef existing = getNamedItemDef(name);
607 	    if (existing != null) {
608 		// conflict
609 		String msg = "The item definition for '" + name + "' in node type '" + def.getDeclaringNodeType() + "' conflicts with node type '" + existing.getDeclaringNodeType() + "': name collision";
610 		log.error(msg);
611 		throw new NodeTypeConflictException(msg);
612 	    }
613 	    namedItemDefs.put(name, def);
614 	}
615 
616 	// residual item definitions
617 	defs = other.getUnnamedItemDefs();
618 	for (int i = 0; i < defs.length; i++) {
619 	    ChildItemDef def = defs[i];
620 	    if (includesNodeType(def.getDeclaringNodeType())) {
621 		// ignore redundant definitions
622 		continue;
623 	    }
624 	    Iterator iter = unnamedItemDefs.iterator();
625 	    while (iter.hasNext()) {
626 		ChildItemDef existing = (ChildItemDef) iter.next();
627 		// compare with existing definition
628 		if (def.definesNode() == existing.definesNode()) {
629 		    if (!def.definesNode()) {
630 			// property definition
631 			PropDef pd = (PropDef) def;
632 			PropDef epd = (PropDef) existing;
633 			// compare type & multiValued flag
634 			if (pd.getRequiredType() == epd.getRequiredType()
635 				&& pd.isMultiple() == epd.isMultiple()) {
636 			    // conflict
637 			    String msg = "A property definition in node type '" + def.getDeclaringNodeType() + "' conflicts with node type '" + existing.getDeclaringNodeType() + "': ambiguos residual property definition";
638 			    log.error(msg);
639 			    throw new NodeTypeConflictException(msg);
640 			}
641 		    } else {
642 			// child node definition
643 			// conflict
644 			String msg = "A child node definition in node type '" + def.getDeclaringNodeType() + "' conflicts with node type '" + existing.getDeclaringNodeType() + "': ambiguos residual child node definition";
645 			log.error(msg);
