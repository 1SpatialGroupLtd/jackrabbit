136     /***
137      * Private constructor
138      *
139      * @param nsReg
140      * @param ntStore
141      * @throws RepositoryException
142      */
143     private NodeTypeRegistry(NamespaceRegistryImpl nsReg, FileSystem ntStore)
144 	    throws RepositoryException {
145 	this.nsReg = nsReg;
146 	this.ntStore = ntStore;
147 	customNodeTypesResource = new FileSystemResource(this.ntStore, CUSTOM_NODETYPES_RESOURCE_NAME);
148 	try {
149 	    // make sure path to resource exists
150 	    if (!customNodeTypesResource.exists()) {
151 		customNodeTypesResource.makeParentDirs();
152 	    }
153 	} catch (FileSystemException fse) {
154 	    String error = "internal error: invalid resource: " + customNodeTypesResource.getPath();
155 	    log.error(error, fse);
156 	    throw new RepositoryException(error, fse);
157 	}
158 
159 	entCache = new EffectiveNodeTypeCache();
160 	registeredNTDefs = new HashMap();
161 	propDefs = new HashMap();
162 	nodeDefs = new HashMap();
163 
164 	// setup definition of root node
165 	rootNodeDef = createRootNodeDef();
166 	nodeDefs.put(new NodeDefId(rootNodeDef), rootNodeDef);
167 
168 	// load and register pre-defined (i.e. built-in) node types
169 	builtInNTDefs = new NodeTypeDefStore();
170 	InputStream in = null;
171 	try {
172 	    in = getClass().getClassLoader().getResourceAsStream(BUILTIN_NODETYPES_RESOURCE_PATH);
173 	    builtInNTDefs.load(in);
174 	    internalRegister(builtInNTDefs.all());
175 	} catch (IOException ioe) {
176 	    String error = "internal error: failed to read built-in node type definitions stored in " + BUILTIN_NODETYPES_RESOURCE_PATH;
177 	    log.error(error);
178 	    throw new RepositoryException(error);
179 	} catch (InvalidNodeTypeDefException intde) {
180 	    String error = "internal error: invalid built-in node type definition stored in " + BUILTIN_NODETYPES_RESOURCE_PATH;
181 	    log.error(error);
182 	    throw new RepositoryException(error);
183 	} finally {
184 	    if (in != null) {
185 		try {
186 		    in.close();
187 		} catch (IOException ioe) {
188 		    // ignore
189 		}
190 	    }
191 	}
192 
193 	// load and register custom node types
194 	customNTDefs = new NodeTypeDefStore();
195 	in = null;
196 	try {
197 	    if (customNodeTypesResource.exists()) {
198 		in = customNodeTypesResource.getInputStream();
199 	    }
200 	} catch (FileSystemException fse) {
201 	    String error = "internal error: failed to access custom node type definitions stored in " + customNodeTypesResource.getPath();
202 	    log.error(error, fse);
203 	    throw new RepositoryException(error, fse);
204 	}
205 	if (in == null) {
206 	    log.info("no custom node type definitions found");
207 	} else {
208 	    try {
209 		customNTDefs.load(in);
210 		internalRegister(customNTDefs.all());
211 	    } catch (IOException ioe) {
212 		String error = "internal error: failed to read custom node type definitions stored in " + customNodeTypesResource.getPath();
213 		log.error(error, ioe);
214 		throw new RepositoryException(error, ioe);
215 	    } catch (InvalidNodeTypeDefException intde) {
216 		String error = "internal error: invalid custom node type definition stored in " + customNodeTypesResource.getPath();
217 		log.error(error, intde);
218 		throw new RepositoryException(error, intde);
219 	    } finally {
220 		try {
221 		    in.close();
222 		} catch (IOException ioe) {
223 		    // ignore
224 		}
225 	    }
226 	}
227     }

229     private static ChildNodeDef createRootNodeDef() {
230 	ChildNodeDef def = new ChildNodeDef();
231 
232 	//def.setName(null);
233 	// FIXME need a fake declaring node type
234 	def.setDeclaringNodeType(new QName(NamespaceRegistryImpl.NS_DEFAULT_URI, ""));
235 	def.setRequiredPrimaryTypes(new QName[]{NT_BASE});
236 	def.setDefaultPrimaryType(NT_UNSTRUCTURED);
237 	def.setMandatory(true);
238 	def.setProtected(false);
239 	def.setOnParentVersion(OnParentVersionAction.VERSION);
240 	def.setPrimaryItem(false);
241 	def.setAllowSameNameSibs(false);
242 	def.setAutoCreate(true);
243 	return def;
244     }

246     /***
247      * Validates and registers the specified collection of <code>NodeTypeDef</code>
248      * objects. An <code>InvalidNodeTypeDefException</code> is thrown if the
249      * validation of any of the contained <code>NodeTypeDef</code> objects fails.
250      * <p/>
251      * Note that in the case an exception is thrown, some node types might have
252      * been nevertheless successfully registered.
253      *
254      * @param ntDefs collection of <code>NodeTypeDef</code> objects
255      * @throws InvalidNodeTypeDefException
256      * @throws RepositoryException
257      * @see #registerNodeType
258      */
259     private synchronized void internalRegister(Collection ntDefs)
260 	    throws InvalidNodeTypeDefException, RepositoryException {
261 	ArrayList list = new ArrayList(ntDefs);
262 
263 	// iterate over definitions until there are no more definitions with
264 	// unresolved (i.e. unregistered) dependencies or an error occurs;
265 
266 	int count = -1;  // number of registered nt's per iteration
267 	while (list.size() > 0 && count != 0) {
268 	    count = 0;
269 	    Iterator iterator = list.iterator();
270 	    while (iterator.hasNext()) {
271 		NodeTypeDef ntd = (NodeTypeDef) iterator.next();
272 		// check if definition has unresolved dependencies
273 		if (registeredNTDefs.keySet().containsAll(ntd.getDependencies())) {
274 		    // try to register it
275 		    internalRegister(ntd);
276 		    // remove it from list
277 		    iterator.remove();
278 		    // increase count
279 		    count++;
280 		}
281 	    }
282 	}
283 	if (list.size() > 0) {
284 	    StringBuffer msg = new StringBuffer();
285 	    msg.append("the following node types could not be registered because of unresolvable dependencies: ");
286 	    Iterator iterator = list.iterator();
287 	    while (iterator.hasNext()) {
288 		msg.append(((NodeTypeDef) iterator.next()).getName());
289 		msg.append(" ");
290 	    }
291 	    log.error(msg.toString());
292 	    throw new InvalidNodeTypeDefException(msg.toString());
293 	}
294     }

296     private EffectiveNodeType internalRegister(NodeTypeDef ntd)
297 	    throws InvalidNodeTypeDefException, RepositoryException {
298 	QName name = ntd.getName();
299 	if (name != null && registeredNTDefs.containsKey(name)) {
300 	    String reason = name + " already exists";
301 	    log.error(reason);
302 	    throw new InvalidNodeTypeDefException(reason);
303 	}
304 
305 	EffectiveNodeType ent = validateNodeTypeDef(ntd);
306 
307 	// store new effective node type instance
308 	entCache.put(ent);
309 
310 	// register clone of node type definition
311 	try {
312 	    ntd = (NodeTypeDef) ntd.clone();
313 	} catch (CloneNotSupportedException e) {
314 	    // should never get here
315 	    log.fatal("internal error", e);
316 	    throw new InternalError(e.getMessage());
317 	}
318 	registeredNTDefs.put(name, ntd);
319 
320 	// store poperty & child node definitions of new node type by id
321 	PropDef[] pda = ntd.getPropertyDefs();
322 	for (int i = 0; i < pda.length; i++) {
323 	    PropDef def = pda[i];
324 	    PropDefId id = new PropDefId(def);
325 	    propDefs.put(id, def);
326 	}
327 	ChildNodeDef[] nda = ntd.getChildNodeDefs();
328 	for (int i = 0; i < nda.length; i++) {
329 	    ChildNodeDef def = nda[i];
330 	    NodeDefId id = new NodeDefId(def);
331 	    nodeDefs.put(id, def);
332 	}
333 
334 	return ent;
335     }

358     /***
359      * Notify the listeners that a node type <code>ntName</code> has been registered.
360      */
361     private void notifyRegistered(QName ntName) {
362 	// copy listeners to array to avoid ConcurrentModificationException
363 	NodeTypeRegistryListener[] la = new NodeTypeRegistryListener[listeners.size()];
364 	Iterator iter = listeners.values().iterator();
365 	int cnt = 0;
366 	while (iter.hasNext()) {
367 	    la[cnt++] = (NodeTypeRegistryListener) iter.next();
368 	}
369 	for (int i = 0; i < la.length; i++) {
370 	    if (la[i] != null) {
371 		la[i].nodeTypeRegistered(ntName);
372 	    }
373 	}
374     }

394     private EffectiveNodeType validateNodeTypeDef(NodeTypeDef ntd)
395 	    throws InvalidNodeTypeDefException, RepositoryException {
396 
397 	// the effective (i.e. merged and resolved) node type resulting from
398 	// the specified node type definition;
399 	// the effective node type will finally be created after the definition
400 	// has been verified and checked for conflicts etc.; in some cases it
401 	// will be created already at an earlier stage during the validation
402 	// of child node definitions
403 	EffectiveNodeType ent = null;
404 
405 	QName name = ntd.getName();
406 	if (name == null) {
407 	    String reason = "no name specified";
408 	    log.error(reason);
409 	    throw new InvalidNodeTypeDefException(reason);
410 	}
411 
412 	if (registeredNTDefs.containsKey(name)) {
413 	    String reason = name + " already exists";
414 	    log.error(reason);
415 	    throw new InvalidNodeTypeDefException(reason);
416 	}
417 
418 	// validate supertypes
419 	QName[] supertypes = ntd.getSupertypes();
420 	for (int i = 0; i < supertypes.length; i++) {
421 	    // simple check for infinite recursion
422 	    // (won't trap recursion on a deeper inheritance level)
423 	    if (name.equals(supertypes[i])) {
424 		String reason = "invalid supertype: " + supertypes[i] + " (infinite recursion))";
425 		log.error(reason);
426 		throw new InvalidNodeTypeDefException(reason);
427 	    }
428 	    if (!registeredNTDefs.containsKey(supertypes[i])) {
429 		String reason = "invalid supertype: " + supertypes[i];
430 		log.error(reason);
431 		throw new InvalidNodeTypeDefException(reason);
432 	    }
433 	}
434 
435 	// check for circularity in inheritance chain ('a' extends 'b' extends 'a'):
436 	Stack inheritanceChain = new Stack();
437 	inheritanceChain.push(name);
438 	checkForCircularInheritance(supertypes, inheritanceChain);
439 
440 	// note that infinite recursion through inheritance is automatically being checked
441 	// by the following call to getEffectiveNodeType
442 	// as it's impossible to register an node type definition which references a
443 	// supertype that isn't registered yet...
444 
445 	// build effective (i.e. merged and resolved) node type from supertypes
446 	// and check for conflicts
447 	if (supertypes.length > 0) {
448 	    try {
449 		buildEffectiveNodeType(supertypes);
450 	    } catch (NodeTypeConflictException ntce) {
451 		String reason = "failed to validate supertypes";
452 		log.error(reason, ntce);
453 		throw new InvalidNodeTypeDefException(reason, ntce);
454 	    } catch (NoSuchNodeTypeException nsnte) {
455 		String reason = "failed to validate supertypes";
456 		log.error(reason, nsnte);
457 		throw new InvalidNodeTypeDefException(reason, nsnte);
458 	    }
459 	}
460 
461 	ChildItemDef primaryItem = null;
462 
463 	// validate property definitions
464 	PropDef[] pda = ntd.getPropertyDefs();
465 	for (int i = 0; i < pda.length; i++) {
466 	    PropDef pd = pda[i];
467 	    // check primary item flag
468 	    if (pd.isPrimaryItem()) {
469 		if (pd.getName() == null) {
470 		    String reason = "primary item must specify a name";
471 		    log.error(reason);
472 		    throw new InvalidNodeTypeDefException(reason);
473 		}
474 		if (primaryItem != null) {
475 		    String reason = "more than one primary item specified";
476 		    log.error(reason);
477 		    throw new InvalidNodeTypeDefException(reason);
478 		} else {
479 		    primaryItem = pd;
480 		}
481 	    }
482 	    // check that auto-created properties specify a name
483 	    if (pd.getName() == null && pd.isAutoCreate()) {
484 		String reason = "auto-created properties must specify a name";
485 		log.error(reason);
486 		throw new InvalidNodeTypeDefException(reason);
487 	    }
488 	    ValueConstraint[] constraints = pd.getValueConstraints();
489 	    InternalValue[] defVals = pd.getDefaultValues();
490 	    if (constraints != null && constraints.length != 0
491 		    && defVals != null && defVals.length != 0) {
492 		// check that default values satisfy value constraints
493 		for (int j = 0; j < constraints.length; j++) {
494 		    for (int k = 0; k < defVals.length; k++) {
495 			try {
496 			    constraints[j].check(defVals[k]);
497 			} catch (ConstraintViolationException cve) {
498 			    String msg = "default value of property "
499 				    + (pd.getName() == null ? "[null]" : pd.getName().toString())
500 				    + " does not satisfy value constraint "
501 				    + constraints[j].getDefinition();
502 			    log.error(msg, cve);
503 			    throw new InvalidNodeTypeDefException(msg, cve);
504 			}
505 		    }
506 		}
507 	    }
508 	}
509 
510 	// validate child-node definitions
511 	ChildNodeDef[] cnda = ntd.getChildNodeDefs();
512 	for (int i = 0; i < cnda.length; i++) {
513 	    ChildNodeDef cnd = cnda[i];
514 	    // check primary item flag
515 	    if (cnd.isPrimaryItem()) {
516 		if (cnd.getName() == null) {
517 		    String msg = "primary item must specify a name";
518 		    log.error(msg);
519 		    throw new InvalidNodeTypeDefException(msg);
520 		}
521 		if (primaryItem != null) {
522 		    String msg = "more than one primary item specified";
523 		    log.error(msg);
524 		    throw new InvalidNodeTypeDefException(msg);
525 		} else {
526 		    primaryItem = cnd;
527 		}
528 	    }
529 	    // check that auto-created child-nodes specify a name
530 	    if (cnd.getName() == null && cnd.isAutoCreate()) {
531 		String msg = "auto-created child-nodes must specify a name";
532 		log.error(msg);
533 		throw new InvalidNodeTypeDefException(msg);
534 	    }
535 	    // check default primary type
536 	    QName dpt = cnd.getDefaultPrimaryType();
537 	    boolean referenceToSelf = false;
538 	    EffectiveNodeType defaultENT = null;
539 	    if (dpt != null) {
540 		// check if this node type specifies itself as default primary type
541 		if (name.equals(dpt)) {
542 		    referenceToSelf = true;
543 		}
544 		// the default primary type must be registered with one notable
545 		// exception: the node type just being registered
546 		if (!name.equals(dpt) && !registeredNTDefs.containsKey(dpt)) {
547 		    String msg = "Invalid default primary type: " + dpt + " in childnode definition " + cnd.getName();
548 		    log.error(msg);
549 		    throw new InvalidNodeTypeDefException(msg);
550 		}
551 		// build effective (i.e. merged and resolved) node type from
552 		// default primary type and check for conflicts
553 		try {
554 		    if (!referenceToSelf) {
555 			defaultENT = getEffectiveNodeType(dpt);
556 		    } else {
557 			// the default primary type is identical with the node type
558 			// just being registered; we have to instantiate it
559 			// 'manually'
560 			ent = EffectiveNodeType.create(this, ntd);
561 			defaultENT = ent;
562 		    }
563 		    if (cnd.isAutoCreate()) {
564 			// check for circularity through default primary types of
565 			// auto-created child nodes (node type 'a' defines
566 			// auto-created child node with default primary type 'a')
567 			Stack definingNTs = new Stack();
568 			definingNTs.push(name);
569 			checkForCircularNodeAutoCreation(defaultENT, definingNTs);
570 		    }
571 		} catch (NodeTypeConflictException ntce) {
572 		    String msg = "failed to validate default primary type";
573 		    log.error(msg, ntce);
574 		    throw new InvalidNodeTypeDefException(msg, ntce);
575 		} catch (NoSuchNodeTypeException nsnte) {
576 		    String msg = "failed to validate default primary type";
577 		    log.error(msg, nsnte);
578 		    throw new InvalidNodeTypeDefException(msg, nsnte);
579 		}
580 	    }
581 
582 	    // check required primary types
583 	    QName[] reqTypes = cnd.getRequiredPrimaryTypes();
584 	    if (reqTypes != null && reqTypes.length > 0) {
585 		for (int n = 0; n < reqTypes.length; n++) {
586 		    QName rpt = reqTypes[n];
587 		    referenceToSelf = false;
588 		    // check if this node type specifies itself as required primary type
589 		    if (name.equals(rpt)) {
590 			referenceToSelf = true;
591 		    }
592 		    // the required primary type must be registered with one notable
593 		    // exception: the node type just being registered
594 		    if (!name.equals(rpt) && !registeredNTDefs.containsKey(rpt)) {
595 			String msg = "invalid required primary type: " + rpt;
596 			log.error(msg);
597 			throw new InvalidNodeTypeDefException(msg);
598 		    }
599 		    // check if default primary type satisfies the required primary type constraint
600 		    if (defaultENT != null && !defaultENT.includesNodeType(rpt)) {
601 			String msg = "default primary type does not satisfy required primary type constraint " + rpt;
602 			log.error(msg);
603 			throw new InvalidNodeTypeDefException(msg);
604 		    }
605 		    // build effective (i.e. merged and resolved) node type from
606 		    // required primary type constraint and check for conflicts
607 		    try {
608 			if (!referenceToSelf) {
609 			    getEffectiveNodeType(rpt);
610 			} else {
611 			    // the required primary type is identical with the node type
612 			    // just being registered; we have to instantiate it
613 			    // 'manually'
614 			    if (ent == null) {
615 				ent = EffectiveNodeType.create(this, ntd);
616 			    }
617 			}
618 		    } catch (NodeTypeConflictException ntce) {
619 			String msg = "failed to validate required primary type constraint";
620 			log.error(msg, ntce);
621 			throw new InvalidNodeTypeDefException(msg, ntce);
622 		    } catch (NoSuchNodeTypeException nsnte) {
623 			String msg = "failed to validate required primary type constraint";
624 			log.error(msg, nsnte);
625 			throw new InvalidNodeTypeDefException(msg, nsnte);
626 		    }
627 		}
628 	    }
629 	}
630 
631 	// now build effective (i.e. merged and resolved) node type from
632 	// this node type definition; this will potentially detect more
633 	// conflicts or problems
634 	if (ent == null) {
635 	    try {
636 		ent = EffectiveNodeType.create(this, ntd);
637 	    } catch (NodeTypeConflictException ntce) {
638 		String msg = "failed to resolve node type definition";
639 		log.error(msg, ntce);
640 		throw new InvalidNodeTypeDefException(msg, ntce);
641 	    } catch (NoSuchNodeTypeException nsnte) {
642 		String msg = "failed to resolve node type definition";
643 		log.error(msg, nsnte);
644 		throw new InvalidNodeTypeDefException(msg, nsnte);
645 	    }
646 	}
647 	return ent;
648     }

678     /***
679      * @param ntNames
680      * @return
681      */
682     public synchronized EffectiveNodeType buildEffectiveNodeType(QName[] ntNames)
683 	    throws NodeTypeConflictException, NoSuchNodeTypeException {
684 	// 1. make sure every single node type is registered
685 	for (int i = 0; i < ntNames.length; i++) {
686 	    if (!registeredNTDefs.containsKey(ntNames[i])) {
687 		throw new NoSuchNodeTypeException(ntNames[i].toString());
688 	    }
689 	}
690 
691 	WeightedKey key = new WeightedKey(ntNames);
692 
693 	// 2. check if aggregate has already been build
694 	if (entCache.contains(key)) {
695 	    return entCache.get(key);
696 	}
697 
698 	// 3. build aggregate
699 	EffectiveNodeType result = null;
700 
701 	// build list of 'best' existing sub-aggregates
702 	ArrayList tmpResults = new ArrayList();
703 	while (key.size() > 0) {
704 	    // check if we've already built this aggregate
705 	    if (entCache.contains(key)) {
706 		tmpResults.add(entCache.get(key));
707 		// subtract the result from the temporary key
708 		// (which is 'empty' now)
709 		key = key.subtract(key);
710 		break;
711 	    }
712 	    // walk list of existing aggregates sorted by 'weight' of
713 	    // aggregate (i.e. the cost of building it)
714 	    boolean foundSubResult = false;
715 	    Iterator iter = entCache.keys();
716 	    while (iter.hasNext()) {
717 		WeightedKey k = (WeightedKey) iter.next();
718 		// check if the existing aggregate is a 'subset' of the one
719 		// we're looking for
720 		if (key.contains(k)) {
721 		    tmpResults.add(entCache.get(k));
722 		    // subtract the result from the temporary key
723 		    key = key.subtract(k);
724 		    foundSubResult = true;
725 		    break;
726 		}
727 	    }
728 	    if (!foundSubResult) {
729 		// no matching sub-aggregates found:
730 		// build aggregate of remaining node types through iteration
731 		QName[] remainder = key.toArray();
732 		for (int i = 0; i < remainder.length; i++) {
733 		    EffectiveNodeType ent = null;
734 		    ent = EffectiveNodeType.create(this, remainder[i]);
735 		    // store new effective node type
736 		    entCache.put(ent);
737 		    if (result == null) {
738 			result = ent;
739 		    } else {
740 			result = result.merge(ent);
741 			// store intermediate result (sub-aggregate)
742 			entCache.put(result);
743 		    }
744 		}
745 		// add aggregate of remaining node types to result list
746 		tmpResults.add(result);
747 		break;
748 	    }
749 	}
750 	// merge the sub-aggregates into new effective node type
751 	for (int i = 0; i < tmpResults.size(); i++) {
752 	    if (result == null) {
753 		result = (EffectiveNodeType) tmpResults.get(i);
754 	    } else {
755 		result = result.merge((EffectiveNodeType) tmpResults.get(i));
756 		// store intermediate result
757 		entCache.put(result);
758 	    }
759 	}
760 	// we're done
761 	return result;
762     }

764     void checkForCircularInheritance(QName[] supertypes, Stack inheritanceChain)
765 	    throws InvalidNodeTypeDefException, RepositoryException {
766 	for (int i = 0; i < supertypes.length; i++) {
767 	    QName nt = supertypes[i];
768 	    int pos = inheritanceChain.lastIndexOf(nt);
769 	    if (pos >= 0) {
770 		StringBuffer buf = new StringBuffer();
771 		for (int j = 0; j < inheritanceChain.size(); j++) {
772 		    if (j == pos) {
773 			buf.append("--> ");
774 		    }
775 		    buf.append(inheritanceChain.get(j));
776 		    buf.append(" extends ");
777 		}
778 		buf.append("--> ");
779 		buf.append(nt);
780 		throw new InvalidNodeTypeDefException("circular inheritance detected: " + buf.toString());
781 	    }
782 
783 	    try {
784 		QName[] sta = getNodeTypeDef(nt).getSupertypes();
785 		if (sta != null && sta.length > 0) {
786 		    // check recursively
787 		    inheritanceChain.push(nt);
788 		    checkForCircularInheritance(sta, inheritanceChain);
789 		    inheritanceChain.pop();
790 		}
791 	    } catch (NoSuchNodeTypeException nsnte) {
792 		String reason = "unknown supertype: " + nt;
793 		log.error(reason, nsnte);
794 		throw new InvalidNodeTypeDefException(reason, nsnte);
795 	    }
796 	}
797     }

799     void checkForCircularNodeAutoCreation(EffectiveNodeType childNodeENT, Stack definingParentNTs)
800 	    throws InvalidNodeTypeDefException {
801 	// check for circularity through default node types of auto-created child nodes
802 	// (node type 'a' defines auto-created child node with default node type 'a')
803 	QName[] childNodeNTs = childNodeENT.getAllNodeTypes();
804 	for (int i = 0; i < childNodeNTs.length; i++) {
805 	    QName nt = childNodeNTs[i];
806 	    int pos = definingParentNTs.lastIndexOf(nt);
807 	    if (pos >= 0) {
808 		StringBuffer buf = new StringBuffer();
809 		for (int j = 0; j < definingParentNTs.size(); j++) {
810 		    if (j == pos) {
811 			buf.append("--> ");
812 		    }
813 		    buf.append("node type ");
814 		    buf.append(definingParentNTs.get(j));
815 		    buf.append(" defines auto-created child node with default ");
816 		}
817 		buf.append("--> ");
818 		buf.append("node type ");
819 		buf.append(nt);
820 		throw new InvalidNodeTypeDefException("circular node auto-creation detected: " + buf.toString());
821 	    }
822 	}
823 
824 	ChildNodeDef[] nodeDefs = childNodeENT.getAutoCreateNodeDefs();
825 	for (int i = 0; i < nodeDefs.length; i++) {
826 	    QName dnt = nodeDefs[i].getDefaultPrimaryType();
827 	    QName definingNT = nodeDefs[i].getDeclaringNodeType();
828 	    try {
829 		if (dnt != null) {
830 		    // check recursively
831 		    definingParentNTs.push(definingNT);
832 		    checkForCircularNodeAutoCreation(getEffectiveNodeType(dnt), definingParentNTs);
833 		    definingParentNTs.pop();
834 		}
835 	    } catch (NoSuchNodeTypeException nsnte) {
836 		String reason = definingNT + " defines invalid default node type for child node " + nodeDefs[i].getName();
837 		log.error(reason, nsnte);
838 		throw new InvalidNodeTypeDefException(reason, nsnte);
839 	    }
840 	}

845      * a registered <code>EffectiveNodeType</code> instance.
846      * <p/>
847      * The validation includes the following checks:
848      * <ul>
849      * <li>Supertypes must exist and be registered</li>
850      * <li>Inheritance graph must not be circular</li>
851      * <li>Aggregation of supertypes must not result in name conflicts,

865      * <li>Nodetypes specified as constraints in child-node definitions

867      * <li>The aggregation of the node types specified as constraints in
868      * child-node definitions must not result in name conflicts, ambiguities,
869      * etc.</li>
870      * <li>Default node types in child-node definitions must satisfy
871      * node type constraints in the same child-node definition</li>
872      * </ul>
873      *

1163     /***
1164      * A <code>WeightedKey</code> uniquely identifies
1165      * a combination (i.e. an aggregation) of one or more node types.
1166      * The weight is an indicator for the cost involved in building such an
1167      * aggregate (an aggregation multiple complex node types with deep
1168      * inheritance trees is more costly to build/validate than an agreggation
1169      * of two very simple node types with just one property definition each).
1170      * <p/>
1171      * A very simple (and not very accurate) approximation of the weight would
1172      * be the number of explicitly aggregated node types (ignoring inheritance
1173      * and complexity of each involved node type). A better approximation would
1174      * be the number of <b>all</b>, explicitly and implicitly (note that
1175      * inheritance is also an aggregation) aggregated node types.
1176      * <p/>
1177      * The more accurate the weight definition, the more efficient is the
1178      * the building of new aggregates.
1179      * <p/>
1180      * It is important to note that the weight is not part of the key value,
1181      * i.e. it is not considered by the <code>hashCode()</code> and
1182      * <code>equals(Object)</code> methods. It does however affect the order
1183      * of <code>WeightedKey</code> instances. See
1184      * <code>{@link #compareTo(Object)}</code> for more information.
1185      * <p/>
1186      * Let's assume we have an aggregation of node types named "b", "a" and "c".
1187      * Its key would be "[a, b, c]" and the weight 3 (using the simple
1188      * approximation).
1189      *
1190      * @author Stefan Guggisberg
1191      */
1192     static class WeightedKey implements Comparable {
1193 	/***
1194 	 * set of node type names, sorted in ascending order
1195 	 */
1196 	private final TreeSet set;
1197 	private final int weight;
1198 
1199 	/***
1200 	 * @param ntNames
1201 	 */
1202 	WeightedKey(QName[] ntNames) {
1203 	    this(ntNames, ntNames.length);
1204 	}
1205 
1206 	/***
1207 	 * @param ntNames
1208 	 * @param weight
1209 	 */
1210 	WeightedKey(QName[] ntNames, int weight) {
1211 	    this.weight = weight;
