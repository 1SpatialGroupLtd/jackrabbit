45      /***
46       * Package private constructor
47       * <p/>
48       * Creates a valid node type instance.
49       * We assume that the node type definition is valid and all referenced
50       * node types (supertypes, required node types etc.) do exist and are valid.
51       *
52       * @param ent        the effective (i.e. merged and resolved) node type representation
53       * @param ntd        the definition of this node type
54       * @param ntMgr      the node type manager associated with this node type
55       * @param nsResolver namespace resolver
56       */
57      NodeTypeImpl(EffectiveNodeType ent, NodeTypeDef ntd, NodeTypeManagerImpl ntMgr, NamespaceResolver nsResolver) {
58  	this.ent = ent;
59  	this.ntMgr = ntMgr;
60  	this.nsResolver = nsResolver;
61  	try {
62  	    // store a clone of the definition
63  	    this.ntd = (NodeTypeDef) ntd.clone();
64  	} catch (CloneNotSupportedException e) {
65  	    // should never get here
66  	    log.fatal("internal error", e);
67  	    throw new InternalError(e.getMessage());
68  	}
69      }

245     /***
246      * Returns all <i>inherited</i> supertypes of this node type.
247      *
248      * @return an array of <code>NodeType</code> objects.
249      * @see #getSupertypes
250      * @see #getDeclaredSupertypes
251      */
252     public NodeType[] getInheritedSupertypes() {
253 	// declared supertypes
254 	QName[] ntNames = ntd.getSupertypes();
255 	HashSet declared = new HashSet();
256 	for (int i = 0; i < ntNames.length; i++) {
257 	    declared.add(ntNames[i]);
258 	}
259 	// all supertypes
260 	ntNames = ent.getInheritedNodeTypes();
261 
262 	// filter from all supertypes those that are not declared
263 	ArrayList inherited = new ArrayList();
264 	for (int i = 0; i < ntNames.length; i++) {
265 	    if (!declared.contains(ntNames[i])) {
266 		try {
267 		    inherited.add(ntMgr.getNodeType(ntNames[i]));
268 		} catch (NoSuchNodeTypeException e) {
269 		    // should never get here
270 		    log.error("undefined supertype", e);
271 		    return new NodeType[0];
272 		}
273 	    }
274 	}
275 
276 	return (NodeType[]) inherited.toArray(new NodeType[inherited.size()]);
277     }

379     /***
380      * @see NodeType#canSetProperty(String, Value)
381      */
382     public boolean canSetProperty(String propertyName, Value value) {
383 	if (value == null) {
384 	    // setting a property to null is equivalent of removing it
385 	    return canRemoveItem(propertyName);
386 	}
387 	try {
388 	    QName name = QName.fromJCRName(propertyName, nsResolver);
389 	    int type = (value == null) ? PropertyType.UNDEFINED : value.getType();
390 	    PropertyDefImpl def = getApplicablePropertyDef(name, type, false);
391 	    if (def.isProtected()) {
392 		return false;
393 	    }
394 	    if (def.isMultiple()) {
395 		return false;
396 	    }
397 	    InternalValue internalValue = InternalValue.create(value, nsResolver);
398 	    checkSetPropertyValueConstraints(def, new InternalValue[]{internalValue});
399 	    return true;
400 	} catch (BaseException be) {
401 	    // implementation specific exception, fall through
402 	} catch (RepositoryException re) {
403 	    // fall through
404 	}
405 	return false;
406     }

408     /***
409      * @see NodeType#canSetProperty(String, Value[])
410      */
411     public boolean canSetProperty(String propertyName, Value[] values) {
412 	if (values == null) {
413 	    // setting a property to null is equivalent of removing it
414 	    return canRemoveItem(propertyName);
415 	}
416 	try {
417 	    QName name = QName.fromJCRName(propertyName, nsResolver);
418 	    int type = (values == null || values.length == 0) ? PropertyType.UNDEFINED : values[0].getType();
419 	    PropertyDefImpl def = getApplicablePropertyDef(name, type, true);
420 	    if (def.isProtected()) {
421 		return false;
422 	    }
423 	    if (!def.isMultiple()) {
424 		return false;
425 	    }
426 	    ArrayList list = new ArrayList();
427 	    // convert values and compact array (purge null entries)
428 	    for (int i = 0; i < values.length; i++) {
429 		if (values[i] != null) {
430 		    InternalValue internalValue = InternalValue.create(values[i], nsResolver);
431 		    list.add(internalValue);
432 		}
433 	    }
434 	    InternalValue[] internalValues = (InternalValue[]) list.toArray(new InternalValue[list.size()]);
435 	    checkSetPropertyValueConstraints(def, internalValues);
436 	    return true;
437 	} catch (BaseException be) {
438 	    // implementation specific exception, fall through
439 	} catch (RepositoryException re) {
440 	    // fall through
441 	}
442 	return false;
443     }
