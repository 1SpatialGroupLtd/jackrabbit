16  package org.apache.jackrabbit.core.nodetype;
17  
18  import java.io.Serializable;
19  
20  /***
21   * <code>PropDefId</code> uniquely identifies a <code>PropDef</code> in the
22   * node type registry.
23   *
24   * @author Stefan Guggisberg
25   * @version $Revision: 1.2 $
26   */
27  public class PropDefId implements Serializable {
28  
29      static final long serialVersionUID = 3675238890036653593L;
30  
31      private final int id;
32  
33      public PropDefId(PropDef def) {
34  	if (def == null) {
35  	    throw new IllegalArgumentException("PropDef argument can not be null");
36  	}
37  	// build key (format: <declaringNodeType>/<name>/<requiredType>/<multiple>)
38  	StringBuffer sb = new StringBuffer();
39  
40  	sb.append(def.getDeclaringNodeType().toString());
41  	sb.append('/');
42  	if (def.definesResidual()) {
43  	    sb.append('*');
44  	} else {
45  	    sb.append(def.getName().toString());
46  	}
47  	sb.append('/');
48  	sb.append(def.getRequiredType());
49  	sb.append('/');
50  	sb.append(def.isMultiple() ? 1 : 0);
51  
52  	id = sb.toString().hashCode();
53      }
54  
55      private PropDefId(int id) {
56  	this.id = id;
57      }
58  
59      /***
60       * Returns a <code>PropDefId</code> holding the value of the specified
61       * string. The string must be in the format returned by the
62       * <code>PropDefId.toString()</code> method.
63       *
64       * @param s a <code>String</code> containing the <code>PropDefId</code>
65       *          representation to be parsed.
66       * @return the <code>PropDefId</code> represented by the argument
67       * @throws IllegalArgumentException if the specified string can not be parsed
68       *                                  as a <code>PropDefId</code>.
69       * @see #toString()
70       */
71      public static PropDefId valueOf(String s) {
72  	if (s == null) {
73  	    throw new IllegalArgumentException("invalid PropDefId literal");
74  	}
75  	return new PropDefId(Integer.parseInt(s));
76      }
77  
78      public boolean equals(Object obj) {
79  	if (this == obj) {
80  	    return true;
81  	}
82  	if (obj instanceof PropDefId) {
83  	    PropDefId other = (PropDefId) obj;
84  	    return id == other.id;
85  	}
86  	return false;
87      }
88  
89      public String toString() {
90  	return Integer.toString(id);
91      }
92  
93      public int hashCode() {
94  	return id;
95      }
96  }
