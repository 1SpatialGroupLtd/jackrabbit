23  /***
24   * <code>QName</code> represents the qualified name of a repository item
25   * (i.e. <code>Node</code> or <code>Property</code>) or a node type.
26   *
27   * @author Stefan Guggisberg
28   * @version $Revision: 1.15 $, $Date: 2004/08/27 15:48:19 $
29   */
30  public class QName implements Cloneable, Comparable, Serializable {

32      static final long serialVersionUID = -2712313010017755368L;

34      /***
35       * Pattern used to validate and parse name:<p>
36       * <ul>
37       * <li>group 1 is namespace prefix incl. delimiter (colon)
38       * <li>group 2 is namespace prefix excl. delimiter (colon)
39       * <li>group 3 is localName
40       * </ul>
41       */
42      private static final Pattern NAME_PATTERN = Pattern.compile("(([^ /://[//]*'\"|](?:[^/://[//]*'\"|]*[^ /://[//]*'\"|])?):)?([^ /://[//]*'\"|](?:[^/://[//]*'\"|]*[^ /://[//]*'\"|])?)");
43  
44      private transient int hash;
45      private transient String string;
46  
47      protected final String namespaceURI;
48      protected final String localName;
49  
50      /***
51       * Creates a new <code>QName</code> instance with the given <code>namespaceURI</code>
52       * and <code>localName</code>.
53       * @param namespaceURI
54       * @param localName
55       */
56      public QName(String namespaceURI, String localName) {
57  	if (namespaceURI == null) {
58  	    throw new IllegalArgumentException("invalid namespaceURI specified");
59  	}
60  	// an empty localName is valid though (e.g. the root node name)
61  	if (localName == null) {
62  	    throw new IllegalArgumentException("invalid localName specified");
63  	}
64  	this.namespaceURI = namespaceURI;
65  	this.localName = localName;
66  	hash = 0;
67      }

70      /***
71       * @param rawName
72       * @param resolver
73       * @return
74       * @throws IllegalNameException
75       * @throws UnknownPrefixException
76       */
77      public static QName fromJCRName(String rawName, NamespaceResolver resolver)
78  	    throws IllegalNameException, UnknownPrefixException {
79  	if (rawName == null || rawName.length() == 0) {
80  	    throw new IllegalNameException("empty name");
81  	}
82  
83  	String prefix = null;
84  	String localName = null;
85  
86  	Matcher matcher = NAME_PATTERN.matcher(rawName);
87  	if (matcher.matches()) {
88  	    // check for prefix (group 1)
89  	    if (matcher.group(1) != null) {
90  		// prefix specified
91  		// group 2 is namespace prefix excl. delimiter (colon)
92  		prefix = matcher.group(2);
93  	    } else {
94  		// no prefix specified
95  		prefix = "";
96  	    }
97  
98  	    // group 3 is localName
99  	    localName = matcher.group(3);
100 	} else {
101 	    // illegal syntax for name
102 	    throw new IllegalNameException("'" + rawName + "' is not a valid name");
103 	}
104 
105 	String uri;
106 	try {
107 	    uri = resolver.getURI(prefix);
108 	} catch (NamespaceException nse) {
109 	    throw new UnknownPrefixException(prefix);
110 	}
111 
112 	return new QName(uri, localName);
113     }

115     /***
116      * Returns a <code>QName</code> holding the value of the specified
117      * string. The string must be in the format returned by the
118      * <code>QName.toString()</code> method, i.e.
119      * <p/>
120      * <code><b>{</b>namespaceURI<b>}</b>localName</code>
121      *
122      * @param s a <code>String</code> containing the <code>QName</code>
123      *          representation to be parsed.
124      * @return the <code>QName</code> represented by the argument
125      * @throws IllegalArgumentException if the specified string can not be parsed
126      *                                  as a <code>QName</code>.
127      * @see #toString()
128      */
129     public static QName valueOf(String s) {
130 	if ("".equals(s) || s == null) {
131 	    throw new IllegalArgumentException("invalid QName literal");
132 	}
133 
134 	if (s.charAt(0) == '{') {
135 	    int i = s.indexOf('}');
136 
137 	    if (i == -1) {
138 		throw new IllegalArgumentException("invalid QName literal");
139 	    }
140 
141 	    if (i == s.length() - 1) {
142 		throw new IllegalArgumentException("invalid QName literal");
143 	    } else {
144 		return new QName(s.substring(1, i), s.substring(i + 1));
145 	    }
146 	} else {
147 	    throw new IllegalArgumentException("invalid QName literal");
148 	}
149     }

152     /***
153      * Checks if <code>jcrName</code> is a valid JCR-style name.
154      *
155      * @param jcrName the name to be checked
156      * @throws IllegalNameException If <code>jcrName</code> is not a valid
157      *                                JCR-style name.
158      */
159     public static void checkFormat(String jcrName) throws IllegalNameException {
160 	if (jcrName == null || jcrName.length() == 0) {
161 	    throw new IllegalNameException("empty name");
162 	}
163 
164 	Matcher matcher = NAME_PATTERN.matcher(jcrName);
165 	if (!matcher.matches()) {
166 	    // illegal syntax for name
167 	    throw new IllegalNameException("'" + jcrName + "' is not a valid name");
168 	}
169     }

186     /***
187      * @param resolver
188      * @return
189      * @throws NoPrefixDeclaredException
190      */
191     public String toJCRName(NamespaceResolver resolver) throws NoPrefixDeclaredException {
192 	StringBuffer sb = new StringBuffer();
193 	// prefix
194 	String prefix;
195 	try {
196 	    prefix = resolver.getPrefix(namespaceURI);
197 	} catch (NamespaceException nse) {
198 	    throw new NoPrefixDeclaredException("no prefix declared for URI: " + namespaceURI);
199 	}
200 	if (prefix.length() == 0) {
201 	    // default prefix (empty string)
202 	} else {
203 	    sb.append(prefix);
204 	    sb.append(':');
205 	}
206 	// name
207 	sb.append(localName);
208 	return sb.toString();
209     }
