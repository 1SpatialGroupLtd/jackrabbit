38  /***
39   * There's one <code>ItemManager</code> instance per <code>Session</code>
40   * instance. It is the factory for <code>Node</code> and <code>Property</code>
41   * instances.
42   * <p/>
43   * The <code>ItemManager</code>'s responsabilities are:
44   * <ul>
45   * <li>providing access to <code>Item</code> instances by <code>ItemId</code>
46   * whereas <code>Node</code> and <code>Item</code> are only providing relative access.
47   * <li>returning the instance of an existing <code>Node</code> or <code>Property</code>,
48   * given its absolute path.
49   * <li>creating the per-session instance of a <code>Node</code>
50   * or <code>Property</code> that doesn't exist yet and needs to be created first.
51   * <li>guaranteeing that there aren't multiple instances representing the same
52   * <code>Node</code> or <code>Property</code> associated with the same
53   * <code>Session</code> instance.
54   * <li>maintaining a cache of the item instances it created.
55   * <li>checking access rights of associated <code>Session</code> in all methods.
56   * </ul>
57   *
58   * @author Stefan Guggisberg
59   * @version $Revision: 1.91 $, $Date: 2004/09/10 15:23:37 $
60   */
61  public class ItemManager implements ItemLifeCycleListener {

81       * Creates a new per-workspace instance <code>ItemManager</code> instance.
82       *
83       * @param itemStateProvider the item state provider associated with
84       *                          the new instance
85       * @param session           the session associated with the new instance
86       * @param rootNodeDef       the definition of the root node
87       * @param rootNodeUUID      the UUID of the root node
88       */
89      ItemManager(ItemStateProvider itemStateProvider, HierarchyManager hierMgr,
90  		SessionImpl session, NodeDef rootNodeDef, String rootNodeUUID) {
91  	this.itemStateProvider = itemStateProvider;
92  	this.hierMgr = hierMgr;
93  	this.session = session;
94  	this.rootNodeDef = rootNodeDef;
95  	rootNodeId = new NodeId(rootNodeUUID);
96  	// setup item cache with soft references to items
97  	itemCache = new ReferenceMap(ReferenceMap.HARD, ReferenceMap.SOFT);
98      }

100     /***
101      * Returns the root node instance of the repository.
102      *
103      * @return the root node.
104      * @throws RepositoryException
105      */
106     private synchronized NodeImpl getRoot() throws RepositoryException {
107 	// lazy instantiation of root node
108 	// to avoid chicken & egg kind of problems
109 	if (root == null) {
110 	    try {
111 		NodeState rootState = (NodeState) itemStateProvider.getItemState(rootNodeId);
112 		// keep a hard reference to root node
113 		root = createNodeInstance(rootState, rootNodeDef);
114 	    } catch (ItemStateException ise) {
115 		String msg = "failed to retrieve state of root node";
116 		log.error(msg, ise);
117 		throw new RepositoryException(msg, ise);
118 	    }
119 	}
120 	return root;
121     }

124      * Dumps the state of this <code>ItemManager</code> instance
125      * (used for diagnostic purposes).
126      *
127      * @param ps
128      * @throws RepositoryException
129      */
130     void dump(PrintStream ps) throws RepositoryException {
131 	ps.println("ItemManager (" + this + ")");
132 	ps.println();
133 	ps.println("Items in cache:");
134 	ps.println();
135 	Iterator iter = itemCache.keySet().iterator();
136 	while (iter.hasNext()) {
137 	    ItemId id = (ItemId) iter.next();
138 	    ItemImpl item = (ItemImpl) itemCache.get(id);
139 	    ps.print(item.isNode() ? "Node: " : "Prop: ");
140 	    ps.print(item.isTransient() ? "transient " : "          ");
141 	    ps.println(id + "\t" + item.getPath() + " (" + item + ")");
142 	}
143 	ps.println();
144     }

193      * @throws AccessDeniedException
194      * @throws RepositoryException

196     synchronized ItemImpl getItem(Path path)
197 	    throws PathNotFoundException, AccessDeniedException, RepositoryException {
198 	ItemId id = hierMgr.resolvePath(path);
199 	try {
200 	    return getItem(id);
201 	} catch (ItemNotFoundException infe) {
202 	    throw new PathNotFoundException(safeGetJCRPath(path));
203 	}

206     /***
207      * @param id
208      * @return
209      * @throws RepositoryException
210      */
211     public synchronized ItemImpl getItem(ItemId id)
212 	    throws ItemNotFoundException, AccessDeniedException, RepositoryException {
213 	// check privileges
214 	if (!session.getAccessManager().isGranted(id, Permission.READ_ITEM)) {
215 	    // clear cache
216 	    if (isCached(id)) {
217 		evictItem(id);
218 	    }
219 	    throw new AccessDeniedException("cannot read item " + id);
220 	}
221 
222 	// check cache
223 	if (isCached(id)) {
224 	    return retrieveItem(id);
225 	}
226 
227 	// shortcut
228 	if (id.denotesNode() && ((NodeId) id).equals(rootNodeId)) {
229 	    return getRoot();
230 	}
231 
232 	// create instance of item using its state object
233 	return createItemInstance(id);
234     }

236     /***
237      * @param parentId
238      * @return
239      * @throws ItemNotFoundException
240      * @throws AccessDeniedException
241      * @throws RepositoryException
242      */
243     synchronized NodeIterator getChildNodes(NodeId parentId)
244 	    throws ItemNotFoundException, AccessDeniedException, RepositoryException {
245 	// check privileges
246 	if (!session.getAccessManager().isGranted(parentId, Permission.READ_ITEM)) {
247 	    // clear cache
248 	    ItemImpl item = retrieveItem(parentId);
249 	    if (item != null) {
250 		evictItem(parentId);
251 	    }
252 	    throw new AccessDeniedException("cannot read item " + parentId);
253 	}
254 
255 	ArrayList children = new ArrayList();
256 
257 	ItemState state = null;
258 	try {
259 	    state = itemStateProvider.getItemState(parentId);
260 	} catch (NoSuchItemStateException nsise) {
261 	    String msg = "no such item: " + parentId;
262 	    log.error(msg);
263 	    throw new ItemNotFoundException(msg);
264 	} catch (ItemStateException ise) {
265 	    String msg = "failed to retrieve item state of node " + parentId;
266 	    log.error(msg);
267 	    throw new RepositoryException(msg);
268 	}
269 
270 	if (!state.isNode()) {
271 	    String msg = "can't list child nodes of property " + parentId;
272 	    log.error(msg);
273 	    throw new RepositoryException(msg);
274 	}
275 	NodeState nodeState = (NodeState) state;
276 	Iterator iter = nodeState.getChildNodeEntries().iterator();
277 
278 	while (iter.hasNext()) {
279 	    NodeState.ChildNodeEntry entry = (NodeState.ChildNodeEntry) iter.next();
280 	    try {
281 		Item item = getItem(new NodeId(entry.getUUID()));
282 		children.add(item);
283 	    } catch (AccessDeniedException ade) {
284 		// ignore
285 		continue;
286 	    }
287 	}
288 
289 	return new IteratorHelper(Collections.unmodifiableList(children));
290     }

347     //-------------------------------------------------< item factory methods >
348     private ItemImpl createItemInstance(ItemId id) throws ItemNotFoundException, RepositoryException {
349 	// create instance of item using its state object
350 	ItemImpl item = null;
351 	ItemState state = null;
352 	try {
353 	    state = itemStateProvider.getItemState(id);
354 	} catch (NoSuchItemStateException ise) {
355 	    throw new ItemNotFoundException(id.toString());
356 	} catch (ItemStateException ise) {
357 	    String msg = "failed to retrieve item state of item " + id;
358 	    log.error(msg);
359 	    throw new RepositoryException(msg);
360 	}
361 
362 	if (state.isNode()) {
363 	    item = createNodeInstance((NodeState) state);
364 	} else {
365 	    item = createPropertyInstance((PropertyState) state);
366 	}
367 	return item;
368     }

370     NodeImpl createNodeInstance(NodeState state, NodeDef def)
371 	    throws RepositoryException {
372 	NodeId id = new NodeId(state.getUUID());
373 	// we want to be informed on life cycle changes of the new node object
374 	// in order to maintain item cache consistency
375 	ItemLifeCycleListener[] listeners = new ItemLifeCycleListener[]{this};
376 
377 	// create node object; create specialized nodes for nodes of specific
378 	// primary types (i.e. nt:version & nt:versionHistory)
379 	if (state.getNodeTypeName().equals(NodeTypeRegistry.NT_VERSION_HISTORY)) {
380 	    return new VersionHistoryImpl(this, session, id, state, def, listeners);
381 	} else if (state.getNodeTypeName().equals(NodeTypeRegistry.NT_FROZEN)) {
382 	    return new FrozenNode(this, session, id, state, def, listeners);
383 	} else if (state.getNodeTypeName().equals(NodeTypeRegistry.NT_VERSION)) {
384 	    return new VersionImpl(this, session, id, state, def, listeners);
385 	} else {
386 	    return new NodeImpl(this, session, id, state, def, listeners);
387 	}
388     }

390     NodeImpl createNodeInstance(NodeState state) throws RepositoryException {
391 	// 1. get definition of the specified node
392 	NodeDef def = session.getNodeTypeManager().getNodeDef(state.getDefinitionId());
393 	if (def == null) {
394 	    String msg = "internal error: no definition found for node " + safeGetJCRPath(state.getId());
395 	    log.error(msg);
396 	    throw new RepositoryException(msg);
397 	}
398 	// 2. create instance
399 	return createNodeInstance(state, def);
400     }

402     PropertyImpl createPropertyInstance(PropertyState state, PropertyDef def) {
403 	PropertyId id = new PropertyId(state.getParentUUID(), state.getName());
404 	// we want to be informed on life cycle changes of the new property object
405 	// in order to maintain item cache consistency
406 	ItemLifeCycleListener[] listeners = new ItemLifeCycleListener[]{this};
407 	// create property object
408 	PropertyImpl prop = new PropertyImpl(this, session, id, state, def, listeners);
409 	return prop;
410     }

412     PropertyImpl createPropertyInstance(PropertyState state) throws RepositoryException {
413 	// 1. get definition for the specified property
414 	PropertyDef def = session.getNodeTypeManager().getPropDef(state.getDefinitionId());
415 	if (def == null) {
416 	    String msg = "internal error: no definition found for property " + safeGetJCRPath(state.getId());
417 	    log.error(msg);
418 	    throw new RepositoryException(msg);
419 	}
420 	// 2. create instance
421 	return createPropertyInstance(state, def);
422     }

424     /***
425      * Removes the specified item from the cache and renders it
426      * 'invalid'.
427      *
428      * @param id
429      * @throws ItemNotFoundException
430      * @throws RepositoryException
431      */
432     void removeItem(ItemId id)
433 	    throws ItemNotFoundException, RepositoryException {
434 	// the removed instance is not directly removed from the cache;
435 	// it will be removed when the instance notifies the item manager
436 	// that it has been invalidated (see itemInvalidated method)
437 	ItemImpl item = retrieveItem(id);
438 	if (item == null) {
439 	    // need to instantiate item first
440 	    item = createItemInstance(id);
441 	}
442 	item.setRemoved();
443     }

449      * @param id the id to be checked
450      * @return true if there's a corresponding cache entry, otherwise false.

456     /***
457      * Returns an item reference from the cache.
458      *
459      * @param id id of the item that should be retrieved.
460      * @return the item reference stored in the corresponding cache entry
461      *         or <code>null</code> if there's no corresponding cache entry.
462      */
463     private ItemImpl retrieveItem(ItemId id) {
464 	return (ItemImpl) itemCache.get(id);
465     }

467     /***
468      * Puts the reference of an item in the cache with
469      * the item's path as the key.
470      *
471      * @param item the item to cache
472      */
473     private void cacheItem(ItemImpl item) {
474 	ItemId id = item.getId();
475 	if (itemCache.containsKey(id)) {
476 	    log.warn("overwriting cached item " + id);
477 	}
478 	log.debug("caching item " + id);
479 	itemCache.put(id, item);
480     }

482     /***
483      * Removes a cache entry for a specific item.
484      *
485      * @param id id of the item to remove from the cache
486      */
487     private void evictItem(ItemId id) {
488 	log.debug("removing item " + id + " from cache");
489 	itemCache.remove(id);
490     }

493     /***
494      * Failsafe conversion of internal <code>Path</code> to JCR path for use in
495      * error messages etc.
496      *
497      * @param path path to convert
498      * @return JCR path
499      */
500     String safeGetJCRPath(Path path) {
501 	try {
502 	    return path.toJCRPath(session.getNamespaceResolver());
503 	} catch (NoPrefixDeclaredException npde) {
504 	    log.error("failed to convert " + path.toString() + " to JCR path.");
505 	    // return string representation of internal path as a fallback
506 	    return path.toString();
507 	}
508     }
