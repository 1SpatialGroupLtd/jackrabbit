120     RepositoryImpl(FileSystem repStore, StableWorkspaceDef[] swda) throws RepositoryException {
121 	// setup file systems
122 	String fsRootPath = "/meta";
123 	try {
124 	    if (!repStore.exists(fsRootPath) || !repStore.isFolder(fsRootPath)) {
125 		repStore.createFolder(fsRootPath);
126 	    }
127 	} catch (FileSystemException fse) {
128 	    String msg = "failed to create folder for repository meta data";
129 	    log.error(msg, fse);
130 	    throw new RepositoryException(msg, fse);
131 	}
132 	this.repStore = repStore;
133 	metaDataStore = new BasedFileSystem(repStore, fsRootPath);
134 
135 	fsRootPath = "/versions";
136 	try {
137 	    if (!repStore.exists(fsRootPath) || !repStore.isFolder(fsRootPath)) {
138 		repStore.createFolder(fsRootPath);
139 	    }
140 	} catch (FileSystemException fse) {
141 	    String msg = "failed to create folder for repository version store";
142 	    log.error(msg, fse);
143 	    throw new RepositoryException(msg, fse);
144 	}
145 	versionStore = new BasedFileSystem(repStore, fsRootPath);
146 
147 	FileSystemResource uuidFile = new FileSystemResource(metaDataStore, "rootUUID");
148 	try {
149 	    if (uuidFile.exists()) {
150 		try {
151 		    // load uuid of the repository's root node
152 		    InputStream in = uuidFile.getInputStream();
153 /*
154 		    // uuid is stored in binary format (16 bytes)
155 		    byte[] bytes = new byte[16];
156 		    try {
157 			in.read(bytes);
158 		    } finally {
159 			try {
160 			    in.close();
161 			} catch (IOException ioe) {
162 			    // ignore
163 			}
164 		    }
165 		    rootNodeUUID = new UUID(bytes).toString();
166 */
167 		    // uuid is stored in text format (36 characters) for better readability
168 		    char[] chars = new char[36];
169 		    InputStreamReader reader = new InputStreamReader(in);
170 		    try {
171 			reader.read(chars);
172 		    } finally {
173 			try {
174 			    reader.close();
175 			} catch (IOException ioe) {
176 			    // ignore
177 			}
178 		    }
179 		    rootNodeUUID = new UUID(new String(chars)).toString();
180 
181 		} catch (Exception e) {
182 		    String msg = "failed to load persisted repository state";
183 		    log.error(msg, e);
184 		    throw new RepositoryException(msg, e);
185 		}
186 	    } else {
187 		// create new uuid
188 		UUID rootUUID = UUID.randomUUID();	// version 4 uuid
189 		rootNodeUUID = rootUUID.toString();
190 		try {
191 		    // persist uuid of the repository's root node
192 		    OutputStream out = uuidFile.getOutputStream();
193 /*
194 		    // store uuid in binary format
195 		    try {
196 			out.write(rootUUID.getBytes());
197 		    } finally {
198 			try {
199 			    out.close();
200 			} catch (IOException ioe) {
201 			    // ignore
202 			}
203 		    }
204 */
205 		    // store uuid in text format for better readability
206 		    OutputStreamWriter writer = new OutputStreamWriter(out);
207 		    try {
208 			writer.write(rootUUID.toString());
209 		    } finally {
210 			try {
211 			    writer.close();
212 			} catch (IOException ioe) {
213 			    // ignore
214 			}
215 		    }
216 		} catch (Exception e) {
217 		    String msg = "failed to persist repository state";
218 		    log.error(msg, e);
219 		    throw new RepositoryException(msg, e);
220 		}
221 	    }
222 	} catch (FileSystemException fse) {
223 	    String msg = "failed to access repository state";
224 	    log.error(msg, fse);
225 	    throw new RepositoryException(msg, fse);
226 	}
227 
228 	// workspaces
229 	for (int i = 0; i < swda.length; i++) {
230 	    StableWorkspaceDef swd = swda[i];
231 	    if (wspDefs.containsKey(swd.getName())) {
232 		String msg = "workspace '" + swd.getName() + "' already defined";
233 		log.error(msg);
234 		throw new RepositoryException(msg);
235 	    }
236 	    wspDefs.put(swd.getName(), swd);
237 	    DynamicWorkspaceDef[] dwda = swd.getDynWorkspaces();
238 	    for (int j = 0; j < dwda.length; j++) {
239 		DynamicWorkspaceDef dwd = dwda[j];
240 		if (wspDefs.containsKey(dwd.getName())) {
241 		    String msg = "workspace '" + dwd.getName() + "' already defined";
242 		    log.error(msg);
243 		    throw new RepositoryException(msg);
244 		}
245 		wspDefs.put(dwd.getName(), dwd);
246 	    }
247 	}
248 	WorkspaceDef wd = (WorkspaceDef) wspDefs.get(DEFAULT_WORKSPACE_NAME);
249 	if (wd == null || wd.isDynamic()) {
250 	    String msg = "mandatory stable workspace 'default' not defined";
251 	    log.error(msg);
252 	    throw new RepositoryException(msg);
253 	}
254 
255 	nsReg = new NamespaceRegistryImpl(new BasedFileSystem(repStore, "/namespaces"));
256 
257 	ntReg = NodeTypeRegistry.create(nsReg, new BasedFileSystem(repStore, "/nodetypes"));
258 
259 	/***
260 	 * todo implement 'System' workspace
261 	 * - the system workspace should have the /jcr:system node
262 	 * - versions, version history and node types should be reflected in
263 	 *   this system workspace as content under /jcr:system
264 	 * - all other workspaces should be dynamic workspaces based on
265 	 *   this 'read-only' system workspace
266 	 */
267 
268 	// FIXME version manager should not operate on default workspace
269 	// check system root node of system workspace
270 	SessionImpl sysSession = getSystemSession(DEFAULT_WORKSPACE_NAME);
271 	NodeImpl rootNode = (NodeImpl) sysSession.getRootNode();
272 	if (!rootNode.hasNode(SYSTEM_ROOT_NAME)) {
273 	    rootNode.addNode(SYSTEM_ROOT_NAME, NodeTypeRegistry.NT_UNSTRUCTURED);
274 	    rootNode.save();
275 	}
276 	// init version manager
277 	vMgr = new VersionManager(sysSession);
278 
279 	// load repository properties
280 	repProps = new Properties();
281 	loadRepProps();
282 
283 	nodesCount = Long.parseLong(repProps.getProperty(STATS_NODE_COUNT_PROPERTY, "0"));
284 	propsCount = Long.parseLong(repProps.getProperty(STATS_PROP_COUNT_PROPERTY, "0"));
285 
286 	// get the system session for every defined workspace and
287 	// register as an event listener
288 	Iterator iter = wspDefs.values().iterator();
289 	while (iter.hasNext()) {
290 	    String wspName = ((WorkspaceDef) iter.next()).getName();
291 	    Session s = getSystemSession(wspName);
292 	    s.getWorkspace().getObservationManager().addEventListener(this,
293 		    EventType.CHILD_NODE_ADDED | EventType.CHILD_NODE_REMOVED
294 		    | EventType.PROPERTY_ADDED | EventType.PROPERTY_REMOVED,
295 		    "/", true, null, null, false);
296 
297 	    // register SearchManager as EventListener
298 	    SearchManager searchMgr = getSearchManager(wspName);
299 
300 	    if (searchMgr != null) {
301 		s.getWorkspace().getObservationManager().addEventListener(searchMgr,
302 			EventType.CHILD_NODE_ADDED | EventType.CHILD_NODE_REMOVED |
303 			EventType.PROPERTY_ADDED | EventType.PROPERTY_REMOVED |
304 			EventType.PROPERTY_CHANGED,
305 			"/", true, null, null, false);
306 	    }
307 	}
308     }

326     synchronized PersistentItemStateManager getWorkspaceStateManager(String workspaceName)
327 	    throws NoSuchWorkspaceException, RepositoryException {
328 	WorkspaceDef wd = (WorkspaceDef) wspDefs.get(workspaceName);
329 	if (wd == null) {
330 	    throw new NoSuchWorkspaceException(workspaceName);
331 	}
332 	// get/create per named workspace (i.e. per physical storage) item state manager
333 	PersistentItemStateManager stateMgr =
334 		(PersistentItemStateManager) wspStateMgrs.get(workspaceName);
335 	if (stateMgr == null) {
336 	    if (wd.isDynamic()) {
337 /*
338 		// create dynamic (i.e. transparent) state manager backed
339 		// by a 'master' state manager
340 		DynamicWorkspaceDef dwd = (DynamicWorkspaceDef) wd;
341 		StableWorkspaceDef swd = (StableWorkspaceDef) wspDefs.get(dwd.getStableWorkspace());
342 		stateMgr = new TransparentItemStateManager(dwd.getFS(), getWorkspaceStateManager(swd));
343 */
344 		// @todo implement dynamic workspace support
345 		throw new RepositoryException("dynamic workspaces are not supported");
346 	    } else {
347 		// create stable (i.e. opaque) state manager
348 		StableWorkspaceDef swd = (StableWorkspaceDef) wd;
349 		PersistenceManager persistMgr = createPersistenceManager(swd);
350 		try {
351 		    stateMgr = new PersistentItemStateManager(persistMgr, rootNodeUUID, ntReg);
352 		} catch (ItemStateException ise) {
353 		    String msg = "failed to instantiate the persistent state manager";
354 		    log.error(msg, ise);
355 		    throw new RepositoryException(msg, ise);
356 		}
357 	    }
358 	    wspStateMgrs.put(workspaceName, stateMgr);
359 	}
360 	return stateMgr;
361     }

408     synchronized SearchManager getSearchManager(String workspaceName)
409 	    throws NoSuchWorkspaceException, RepositoryException {
410 	SearchManager searchMgr
411 		= (SearchManager) wspSearchMgrs.get(workspaceName);
412 	if (searchMgr == null) {
413 	    try {
414 		StableWorkspaceDef wspDef = (StableWorkspaceDef)wspDefs.get(workspaceName);
415                 if (wspDef.getSearchIndexPath() == null) {
416 		    // no search index location configured
417 		    return null;
418 		}
419 		ItemStateProvider stateProvider = getWorkspaceStateManager(workspaceName);
420 		SystemSession s = getSystemSession(workspaceName);
421 		searchMgr = new SearchManager(stateProvider, s.hierMgr, s, wspDef.getSearchIndexPath());
422 	    } catch (IOException e) {
423 		throw new RepositoryException("Exception opening search index.", e);
424 	    }
425 	    wspSearchMgrs.put(workspaceName, searchMgr);
426 	}
427 	return searchMgr;

443      * @return
444      * @throws RepositoryException
445      */
446     private PersistenceManager createPersistenceManager(WorkspaceDef wspDef) throws RepositoryException {
447 	PersistenceManager perstistMgr;
448 	String className = wspDef.getPersistenceManagerClass();
449 	try {
450 	    // Create the persistence manager object
451 	    Class c = Class.forName(className);
452 	    perstistMgr = (PersistenceManager) c.newInstance();
453 	    // set the properties of the persistence manager object from the
454 	    // param hashmap
455 	    BeanMap bm = new BeanMap(perstistMgr);
456 	    HashMap params = wspDef.getPersistenceManagerParams();
457 	    Iterator iter = params.keySet().iterator();
458 	    while (iter.hasNext()) {
459 		Object name = iter.next();
460 		Object value = params.get(name);
461 		bm.put(name, value);
462 	    }
463 	    perstistMgr.init(wspDef);
464 	} catch (Exception e) {
465 	    log.error("Cannot instantiate implementing class " + className, e);
466 	    throw new RepositoryException("Cannot instantiate implementing class " + className, e);
467 	}
468 	return perstistMgr;
469     }

507 	FileSystemResource propFile = new FileSystemResource(metaDataStore, PROPERTIES_RESOURCE);
508 	try {
509 	    repProps.clear();
510 	    if (!propFile.exists() || propFile.length() == 0) {
511 		// initialize properties with pre-defined values
512 		repProps.setProperty(SPEC_VERSION_PROPERTY, SPEC_VERSION);
513 		repProps.setProperty(SPEC_NAME_PROPERTY, SPEC_NAME);

516 		repProps.setProperty(REP_NAME_PROPERTY, REP_NAME);
517 		repProps.setProperty(REP_VERSION_PROPERTY, REP_VERSION);
518 
519 		repProps.setProperty(STATS_NODE_COUNT_PROPERTY, Long.toString(nodesCount));
520 		repProps.setProperty(STATS_PROP_COUNT_PROPERTY, Long.toString(propsCount));
521 
522 		// persist properties
523 		storeRepProps();
524 		return;
525 	    }
526 
527 	    InputStream in = propFile.getInputStream();
528 	    try {
529 		repProps.load(in);
530 	    } finally {
531 		in.close();
532 	    }
533 	} catch (Exception e) {
534 	    String msg = "failed to load repository properties";
535 	    log.error(msg, e);
536 	    throw new RepositoryException(msg, e);

612     /***
613      * @see EventListener#onEvent(EventIterator)
614      */
615     public synchronized void onEvent(EventIterator events) {
616 	while (events.hasNext()) {
617 	    Event event = events.nextEvent();
618 	    long type = event.getType();
619 	    if ((type & EventType.CHILD_NODE_ADDED) == EventType.CHILD_NODE_ADDED) {
620 		nodesCount++;
621 		repProps.setProperty(STATS_NODE_COUNT_PROPERTY, Long.toString(nodesCount));
622 	    }
623 	    if ((type & EventType.CHILD_NODE_REMOVED) == EventType.CHILD_NODE_REMOVED) {
624 		nodesCount--;
625 		repProps.setProperty(STATS_NODE_COUNT_PROPERTY, Long.toString(nodesCount));
626 	    }
627 	    if ((type & EventType.PROPERTY_ADDED) == EventType.PROPERTY_ADDED) {
628 		propsCount++;
629 		repProps.setProperty(STATS_PROP_COUNT_PROPERTY, Long.toString(propsCount));
630 	    }
631 	    if ((type & EventType.PROPERTY_REMOVED) == EventType.PROPERTY_REMOVED) {
632 		propsCount--;
633 		repProps.setProperty(STATS_PROP_COUNT_PROPERTY, Long.toString(propsCount));
634 	    }
635 	}
