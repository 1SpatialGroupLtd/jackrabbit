29  /***
30   * <code>InternalValue</code> represents the internal format of a property value.
31   * <p/>
32   * The following table specifies the internal format for every property type:
33   * <pre>
34   * <table>
35   * <tr><b>PropertyType</b><td></td><td><b>Internal Format</b></td></tr>
36   * <tr>STRING<td></td><td>String</td></tr>
37   * <tr>LONG<td></td><td>Long</td></tr>
38   * <tr>DOUBLE<td></td><td>Double</td></tr>
39   * <tr>DATE<td></td><td>Calendar</td></tr>
40   * <tr>BOOLEAN<td></td><td>Boolean</td></tr>
41   * <tr>NAME<td></td><td>QName</td></tr>
42   * <tr>PATH<td></td><td>Path</td></tr>
43   * <tr>BINARY<td></td><td>BLOBFileValue</td></tr>
44   * <tr>REFERENCE<td></td><td>UUID</td></tr>
45   * </table>
46   * </pre>

57      /***
58       * @param value
59       * @param nsResolver
60       * @return
61       * @throws ValueFormatException
62       * @throws RepositoryException
63       */
64      public static InternalValue create(Value value, NamespaceResolver nsResolver)
65  	    throws ValueFormatException, RepositoryException {
66  	if (value == null) {
67  	    throw new IllegalArgumentException("null value");
68  	}
69  
70  	switch (value.getType()) {
71  	    case PropertyType.BINARY:
72  		try {
73  		    if (value instanceof BLOBFileValue) {
74  			return new InternalValue((BLOBFileValue) value);
75  		    } else {
76  			return new InternalValue(new BLOBFileValue(value.getStream()));
77  		    }
78  		} catch (IOException ioe) {
79  		    throw new ValueFormatException(ioe.getMessage());
80  		}
81  	    case PropertyType.BOOLEAN:
82  		return new InternalValue(value.getBoolean());
83  	    case PropertyType.DATE:
84  		return new InternalValue(value.getDate());
85  	    case PropertyType.DOUBLE:
86  		return new InternalValue(value.getDouble());
87  	    case PropertyType.LONG:
88  		return new InternalValue(value.getLong());
89  	    case PropertyType.REFERENCE:
90  		return new InternalValue(new UUID(value.getString()));
91  	    case PropertyType.NAME:
92  		try {
93  		    return new InternalValue(QName.fromJCRName(value.getString(), nsResolver));
94  		} catch (IllegalNameException ine) {
95  		    throw new ValueFormatException(ine.getMessage());
96  		} catch (UnknownPrefixException upe) {
97  		    throw new ValueFormatException(upe.getMessage());
98  		}
99  	    case PropertyType.PATH:
100 		try {
101 		    return new InternalValue(Path.create(value.getString(), nsResolver, false));
102 		} catch (MalformedPathException mpe) {
103 		    throw new ValueFormatException(mpe.getMessage());
104 		}
105 	    case PropertyType.STRING:
106 		return new InternalValue(value.getString());
107 
108 	    default:
109 		throw new IllegalArgumentException("illegal value");
110 	}
111     }

237     /***
238      * @param nsResolver
239      * @return
240      * @throws RepositoryException
241      */
242     public Value toJCRValue(NamespaceResolver nsResolver) throws RepositoryException {
243 	switch (type) {
244 	    case PropertyType.BINARY:
245 		return new BinaryValue(((BLOBFileValue) val).getStream());
246 	    case PropertyType.BOOLEAN:
247 		return new BooleanValue(((Boolean) val));
248 	    case PropertyType.DATE:
249 		return new DateValue((Calendar) val);
250 	    case PropertyType.DOUBLE:
251 		return new DoubleValue((Double) val);
252 	    case PropertyType.LONG:
253 		return new LongValue((Long) val);
254 	    case PropertyType.REFERENCE:
255 		return ReferenceValue.valueOf(((UUID) val).toString());
256 	    case PropertyType.PATH:
257 		try {
258 		    return PathValue.valueOf(((Path) val).toJCRPath(nsResolver));
259 		} catch (NoPrefixDeclaredException npde) {
260 		    // should never get here...
261 		    throw new RepositoryException("internal error: encountered unregistered namespace", npde);
262 		}
263 	    case PropertyType.NAME:
264 		try {
265 		    return NameValue.valueOf(((QName) val).toJCRName(nsResolver));
266 		} catch (NoPrefixDeclaredException npde) {
267 		    // should never get here...
268 		    throw new RepositoryException("internal error: encountered unregistered namespace", npde);
269 		}
270 	    case PropertyType.STRING:
271 		return new StringValue((String) val);
272 	    default:
273 		throw new RepositoryException("illegal internal value type");
274 	}
275     }

295     public InternalValue createCopy() throws RepositoryException {
296 	switch (type) {
297 	    case PropertyType.BINARY:
298 		try {
299 		    return new InternalValue(new BLOBFileValue(((BLOBFileValue) val).getStream()));
300 		} catch (IOException ioe) {
301 		    throw new RepositoryException("failed to copy binary value", ioe);
302 		}
303 	    case PropertyType.BOOLEAN:
304 		return new InternalValue(((Boolean) val).booleanValue());
305 	    case PropertyType.DATE:
306 		return new InternalValue((Calendar) val);
307 	    case PropertyType.DOUBLE:
308 		return new InternalValue(((Double) val).doubleValue());
309 	    case PropertyType.LONG:
310 		return new InternalValue(((Long) val).longValue());
311 	    case PropertyType.REFERENCE:
312 		return new InternalValue((UUID) val);
313 	    case PropertyType.PATH:
314 		return new InternalValue((Path) val);
315 	    case PropertyType.NAME:
316 		return new InternalValue((QName) val);
317 	    case PropertyType.STRING:
318 		return new InternalValue((String) val);
319 	    default:
320 		throw new RepositoryException("illegal internal value type");
321 	}
322     }

324     /***
325      * Returns the string representation of this internal value. If this is a
326      * <i>binary</i> value then the path of its backing file will be returned.

328      * @return string representation of this internal value

330     public String toString() {
331 	if (type == PropertyType.DATE) {
332 	    return ISO8601.format((Calendar) val);
333 	} else {
334 	    return val.toString();

354     public static InternalValue valueOf(String s, int type) {
355 	switch (type) {
356 	    case PropertyType.BOOLEAN:
357 		return new InternalValue(Boolean.valueOf(s).booleanValue());
358 	    case PropertyType.DATE:
359 		return new InternalValue(ISO8601.parse(s));
360 	    case PropertyType.DOUBLE:
361 		return new InternalValue(Double.valueOf(s).doubleValue());
362 	    case PropertyType.LONG:
363 		return new InternalValue(Long.valueOf(s).longValue());
364 	    case PropertyType.REFERENCE:
365 		return new InternalValue(new UUID(s));
366 	    case PropertyType.PATH:
367 		return new InternalValue(Path.valueOf(s));
368 	    case PropertyType.NAME:
369 		return new InternalValue(QName.valueOf(s));
370 	    case PropertyType.STRING:
371 		return new InternalValue(s);
372 
373 	    case PropertyType.BINARY:
374 		throw new IllegalArgumentException("this method does not support the type PropertyType.BINARY");
375 	    default:
376 		throw new IllegalArgumentException("illegal type");
377 	}
