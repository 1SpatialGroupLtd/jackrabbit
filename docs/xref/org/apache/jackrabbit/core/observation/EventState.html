81      /***
82       * Creates a new <code>EventState</code> instance.
83       *
84       * @param type       the {@link javax.jcr.observation.EventType} of this
85       *                   event.
86       * @param parentUUID the uuid of the parent node associated with this
87       *                   event.
88       * @param parentPath the path of the parent node associated with this
89       *                   event.
90       * @param childUUID  the uuid of the child node associated with this event.
91       *                   If the event type is one of: <code>PROPERTY_ADDED</code>,
92       *                   <code>PROPERTY_CHANGED</code> or <code>PROPERTY_REMOVED</code>
93       *                   this parameter must be <code>null</code>.
94       * @param childName  the qualified name of the child item associated with
95       *                   this event.
96       * @param nodeType   the node type of the parent node.
97       * @param session    the {@link javax.jcr.Session} that caused this event.
98       */
99      private EventState(long type,
100 		       String parentUUID,
101 		       Path parentPath,
102 		       String childUUID,
103 		       QName childName,
104 		       NodeTypeImpl nodeType,
105 		       Session session) {
106 	long mask = (EventType.PROPERTY_ADDED | EventType.PROPERTY_CHANGED | EventType.PROPERTY_REMOVED);
107 	if ((type & mask) > 0) {
108 	    if (childUUID != null) {
109 		throw new IllegalArgumentException("childUUID only allowed for Node events.");
110 	    }
111 	} else {
112 	    if (childUUID == null) {
113 		throw new IllegalArgumentException("childUUID must not be null for Node events.");
114 	    }
115 	}
116 	this.type = type;
117 	this.parentUUID = parentUUID;
118 	this.parentPath = parentPath;
119 	this.childUUID = childUUID;
120 	this.childName = childName;
121 	this.nodeType = nodeType;
122 	this.session = session;
123     }

394     /***
395      * Returns a String representation of <code>eventType</code>.
396      *
397      * @param eventType an event type defined by {@link EventType}.
398      * @return a String representation of <code>eventType</code>.
399      */
400     public static String valueOf(long eventType) {
401 	if (eventType == EventType.CHILD_NODE_ADDED) {
402 	    return "ChildNodeAdded";
403 	} else if (eventType == EventType.CHILD_NODE_REMOVED) {
404 	    return "ChildNodeRemoved";
405 	} else if (eventType == EventType.PROPERTY_ADDED) {
406 	    return "PropertyAdded";
407 	} else if (eventType == EventType.PROPERTY_CHANGED) {
408 	    return "PropertyChanged";
409 	} else if (eventType == EventType.PROPERTY_REMOVED) {
410 	    return "PropertyRemoved";
411 	} else {
412 	    return "UnknownEventType";
413 	}
