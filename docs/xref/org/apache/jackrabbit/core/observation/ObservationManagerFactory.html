38  /***
39   * The class <code>ObservationManagerFactory</code> creates new
40   * <code>ObservationManager</code> instances for sessions. It also implements the
41   * {@link EventDispatcher} interface where {@link EventStateCollection}s can be
42   * dispatched to {@link javax.jcr.observation.EventListener}s.
43   *
44   * @author Marcel Reutegger
45   */
46  public final class ObservationManagerFactory implements EventDispatcher, Runnable {
47  
48      /***
49       * Logger instance for this class
50       */
51      private static final Logger log
52  	    = Logger.getLogger(ObservationManagerFactory.class);
53  
54      /***
55       * Dummy DispatchAction indicating the notification thread to end
56       */
57      private static final DispatchAction DISPOSE_MARKER = new DispatchAction(null, null);
58  
59      /***
60       * Currently active <code>EventConsumer</code>s for notification
61       */
62      private Set activeConsumers = new HashSet();
63  
64      /***
65       * Set of <code>EventConsumer</code>s for read only Set access
66       */
67      private Set readOnlyConsumers;
68  
69      /***
70       * synchronization monitor for listener changes
71       */
72      private Object consumerChange = new Object();
73  
74      /***
75       * Contains the pending events that will be delivered to event listeners
76       */
77      private Buffer eventQueue
78  	    = BufferUtils.blockingBuffer(new UnboundedFifoBuffer());
79  
80      /***
81       * The background notification thread
82       */
83      private Thread notificationThread;
84  
85      /***
86       * Creates a new <code>ObservationManagerFactory</code> instance
87       * and starts the notification thread deamon.
88       */
89      public ObservationManagerFactory() {
90  	notificationThread = new Thread(this, "ObservationManager");
91  	notificationThread.setDaemon(true);
92  	notificationThread.start();
93      }
94  
95      /***
96       * Disposes this <code>ObservationManager</code>. This will
97       * effectively stop the background notification thread.
98       */
99      public void dispose() {
100 	// dispatch dummy event to mark end of notification
101 	eventQueue.add(DISPOSE_MARKER);
102 	try {
103 	    notificationThread.join();
104 	} catch (InterruptedException e) {
105 	    // FIXME log exception ?
106 	}
107 	log.info("Notification of EventListeners stopped.");
108     }
109 
110     /***
111      * Returns an unmodifieable <code>Set</code> of <code>EventConsumer</code>s.
112      *
113      * @return <code>Set</code> of <code>EventConsumer</code>s.
114      */
115     private Set getConsumers() {
116 	synchronized (consumerChange) {
117 	    if (readOnlyConsumers == null) {
118 		readOnlyConsumers = Collections.unmodifiableSet(new HashSet(activeConsumers));
119 	    }
120 	    return readOnlyConsumers;
121 	}
122     }
123 
124     /***
125      * Creates a new <code>session</code> local <code>ObservationManager</code>
126      * with an associated <code>NamespaceResolver</code>.
127      *
128      * @param session the session.
129      * @param itemMgr the <code>ItemManager</code> of the <code>session</code>.
130      * @return an <code>ObservationManager</code>.
131      */
132     public ObservationManager createObservationManager(SessionImpl session,
133 						       ItemManager itemMgr) {
134 	return new SessionLocalObservationManager(session, itemMgr);
135     }
136 
137 
138     /***
139      * Implements the run method of the background notification
140      * thread.
141      */
142     public void run() {
143 	DispatchAction action;
144 	while ((action = (DispatchAction) eventQueue.remove()) != DISPOSE_MARKER) {
145 
146 	    log.debug("got EventStateCollection");
147 	    log.debug("event delivery to " + action.eventConsumers.size() + " consumers started...");
148 	    for (Iterator it = action.eventConsumers.iterator(); it.hasNext();) {
149 		EventConsumer c = (EventConsumer) it.next();
150 		try {
151 		    c.consumeEvents(action.eventStates);
152 		} catch (Throwable t) {
153 		    log.error("EventConsumer threw exception.", t);
154 		    // move on to the next consumer
155 		}
156 	    }
157 	    log.debug("event delivery finished.");
158 
159 	}
160     }

164     /***
165      * @see EventDispatcher#dispatchEvents
166      */
167     public void dispatchEvents(EventStateCollection events) {
168 	eventQueue.add(new DispatchAction(events, getConsumers()));
169     }

173     /***
174      * Each <code>Session</code> instance has its own <code>ObservationManager</code>
175      * instance. The class <code>SessionLocalObservationManager</code> implements
176      * this behaviour.
177      */
178     class SessionLocalObservationManager implements ObservationManager {
179 
180 	/***
181 	 * The <code>Session</code> this <code>ObservationManager</code>
182 	 * belongs to.
183 	 */
184 	private SessionImpl session;
185 
186 	/***
187 	 * The <code>ItemManager</code> for this <code>ObservationManager</code>.
188 	 */
189 	private ItemManager itemMgr;
190 
191 	/***
192 	 * Creates an <code>ObservationManager</code> instance.
193 	 *
194 	 * @param session the <code>Session</code> this ObservationManager
195 	 *                belongs to.
196 	 * @param itemMgr {@link org.apache.jackrabbit.core.ItemManager} of the passed
197 	 *                <code>Session</code>.
198 	 * @throws NullPointerException if <code>session</code> or <code>itemMgr</code>
199 	 *                              is <code>null</code>.
200 	 */
201 	SessionLocalObservationManager(SessionImpl session,
202 				       ItemManager itemMgr) {
203 	    if (session == null) {
204 		throw new NullPointerException("session");
205 	    }
206 	    if (itemMgr == null) {
207 		throw new NullPointerException("itemMgr");
208 	    }
209 
210 	    this.session = session;
211 	    this.itemMgr = itemMgr;
212 	}
213 
214 	/***
215 	 * @see ObservationManager#addEventListener
216 	 */
217 	public void addEventListener(EventListener listener,
218 				     long eventTypes,
219 				     String absPath,
220 				     boolean isDeep,
221 				     String[] uuid,
222 				     String[] nodeTypeName,
223 				     boolean noLocal)
224 		throws RepositoryException {
225 
226 	    // create NodeType instances from names
227 	    NodeType[] nodeTypes;
228 	    if (nodeTypeName == null) {
229 		nodeTypes = null;
230 	    } else {
231 		NodeTypeManager ntMgr = session.getNodeTypeManager();
232 		nodeTypes = new NodeType[nodeTypeName.length];
233 		for (int i = 0; i < nodeTypes.length; i++) {
234 		    nodeTypes[i] = ntMgr.getNodeType(nodeTypeName[i]);
235 		}
236 	    }
237 
238 	    Path path;
239 	    try {
240 		path = Path.create(absPath, session.getNamespaceResolver(), true);
241 	    } catch (MalformedPathException mpe) {
242 		String msg = "invalid path syntax: " + absPath;
243 		log.error(msg, mpe);
244 		throw new RepositoryException(msg, mpe);
245 	    }
246 	    // create filter
247 	    EventFilter filter = new EventFilter(itemMgr,
248 		    session,
249 		    eventTypes,
250 		    path,
251 		    isDeep,
252 		    uuid,
253 		    nodeTypes,
254 		    noLocal);
255 
256 	    EventConsumer consumer =
257 		    new EventConsumer(session, listener, filter);
258 
259 	    synchronized (consumerChange) {
260 		// remove existing if any
261 		activeConsumers.remove(consumer);
262 		// re-add it
263 		activeConsumers.add(consumer);
264 		// reset read only consumer set
265 		readOnlyConsumers = null;
266 	    }
267 	}
268 
269 	/***
270 	 * @see ObservationManager#removeEventListener(javax.jcr.observation.EventListener)
271 	 */
272 	public void removeEventListener(EventListener listener)
273 		throws RepositoryException {
274 	    EventConsumer consumer =
275 		    new EventConsumer(session, listener, EventFilter.BLOCK_ALL);
276 
277 	    synchronized (consumerChange) {
278 		activeConsumers.remove(consumer);
279 		// reset read only listener set
280 		readOnlyConsumers = null;
281 	    }
282 	}
283 
284 	/***
285 	 * @see ObservationManager#getRegisteredEventListeners()
286 	 */
287 	public EventListenerIterator getRegisteredEventListeners()
288 		throws RepositoryException {
289 	    return new EventListenerIteratorImpl(session, getConsumers());
290 	}
291     }
