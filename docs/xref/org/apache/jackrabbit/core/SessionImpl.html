99      /***
100      * Package private constructor.
101      *
102      * @param rep
103      * @param credentials
104      * @param wspName
105      */
106     SessionImpl(RepositoryImpl rep, Credentials credentials, String wspName)
107 	    throws RepositoryException {
108 	this.rep = rep;
109 
110 	if (credentials instanceof SimpleCredentials) {
111 	    SimpleCredentials sc = (SimpleCredentials) credentials;
112 	    // clear password for security reasons
113 	    char[] pwd = sc.getPassword();
114 	    if (pwd != null) {
115 		for (int i = 0; i < pwd.length; i++) {
116 		    pwd[i] = 0;
117 		}
118 	    }
119 	    userId = sc.getUserId();
120 	    String[] names = sc.getAttributeNames();
121 	    for (int i = 0; i < names.length; i++) {
122 		attributes.put(names[i], sc.getAttribute(names[i]));
123 	    }
124 	} else {
125 	    userId = null;
126 	}
127 
128 	nsMappings = new TransientNamespaceMappings(rep.getNamespaceRegistry());
129 
130 	ntMgr = new NodeTypeManagerImpl(rep.getNodeTypeRegistry(), getNamespaceResolver());
131 	wsp = new WorkspaceImpl(wspName, rep.getWorkspaceStateManager(wspName),
132 		rep.getWorkspaceReferenceManager(wspName), rep, this);
133 	itemStateMgr = new SessionItemStateManager(rep.getRootNodeUUID(), wsp.getPersistentStateManager(), getNamespaceResolver());
134 	hierMgr = itemStateMgr.getHierarchyMgr();
135 	itemMgr = new ItemManager(itemStateMgr, hierMgr, this, ntMgr.getRootNodeDefinition(), rep.getRootNodeUUID());
136 	accessMgr = new AccessManagerImpl(credentials, hierMgr, getNamespaceResolver());
137     }

340     /***
341      * @see Session#move(String, String)
342      */
343     public void move(String srcAbsPath, String destAbsPath)
344 	    throws ItemExistsException, PathNotFoundException,
345 	    ConstraintViolationException, RepositoryException {
346 
347 	// check paths & get node instances
348 
349 	Path srcPath;
350 	Path.PathElement srcName;
351 	Path srcParentPath;
352 	NodeImpl targetNode;
353 	NodeImpl srcParentNode;
354 	try {
355 	    srcPath = Path.create(srcAbsPath, getNamespaceResolver(), true);
356 	    srcName = srcPath.getNameElement();
357 	    srcParentPath = srcPath.getAncestor(1);
358 	    ItemImpl item = itemMgr.getItem(srcPath);
359 	    if (!item.isNode()) {
360 		throw new PathNotFoundException(srcAbsPath);
361 	    }
362 	    targetNode = (NodeImpl) item;
363 	    srcParentNode = (NodeImpl) itemMgr.getItem(srcParentPath);
364 	} catch (AccessDeniedException ade) {
365 	    throw new PathNotFoundException(srcAbsPath);
366 	} catch (MalformedPathException mpe) {
367 	    String msg = srcAbsPath + ": invalid path";
368 	    log.error(msg, mpe);
369 	    throw new RepositoryException(msg, mpe);
370 	}
371 
372 	Path destPath;
373 	Path.PathElement destName;
374 	Path destParentPath;
375 	NodeImpl destParentNode;
376 	try {
377 	    destPath = Path.create(destAbsPath, getNamespaceResolver(), true);
378 	    destName = destPath.getNameElement();
379 	    destParentPath = destPath.getAncestor(1);
380 	    destParentNode = (NodeImpl) itemMgr.getItem(destParentPath);
381 	} catch (AccessDeniedException ade) {
382 	    throw new PathNotFoundException(destAbsPath);
383 	} catch (MalformedPathException mpe) {
384 	    String msg = destAbsPath + ": invalid path";
385 	    log.error(msg, mpe);
386 	    throw new RepositoryException(msg, mpe);
387 	}
388 	int ind = destName.getIndex();
389 	if (ind > 0) {
390 	    // subscript in name element
391 	    String msg = destAbsPath + ": invalid destination path (subscript in name element is not allowed)";
392 	    log.error(msg);
393 	    throw new RepositoryException(msg);
394 	}
395 
396 	// check for name collisions
397 
398 	try {
399 	    ItemImpl item = itemMgr.getItem(destPath);
400 	    if (!item.isNode()) {
401 		// there's already a property with that name
402 		throw new ItemExistsException(item.safeGetJCRPath());
403 	    } else {
404 		// there's already a node with that name
405 		// check same-name sibling setting of both new and existing node
406 		if (!destParentNode.getDefinition().allowSameNameSibs()
407 			|| !((NodeImpl) item).getDefinition().allowSameNameSibs()) {
408 		    throw new ItemExistsException(item.safeGetJCRPath());
409 		}
410 	    }
411 	} catch (AccessDeniedException ade) {
412 	    // FIXME by throwing ItemExistsException we're disclosing too much information
413 	    throw new ItemExistsException(destAbsPath);
414 	} catch (PathNotFoundException pnfe) {
415 	    // no name collision
416 	}
417 
418 	// check constraints
419 
420 	// get applicable definition of target node at new location
421 	NodeTypeImpl nt = (NodeTypeImpl) targetNode.getPrimaryNodeType();
422 	NodeDefImpl newTargetDef;
423 	try {
424 	    newTargetDef = destParentNode.getApplicableChildNodeDef(destName.getName(), nt.getQName());
425 	} catch (RepositoryException re) {
426 	    String msg = destAbsPath + ": no definition found in parent node's node type for new node";
427 	    log.error(msg, re);
428 	    throw new ConstraintViolationException(msg, re);
429 	}
430 	// check protected flag of old & new parent
431 	if (destParentNode.getDefinition().isProtected()) {
432 	    String msg = destAbsPath + ": cannot add a child node to a protected node";
433 	    log.error(msg);
434 	    throw new ConstraintViolationException(msg);
435 	}
436 	if (srcParentNode.getDefinition().isProtected()) {
437 	    String msg = srcAbsPath + ": cannot remove a child node from a protected node";
438 	    log.error(msg);
439 	    throw new ConstraintViolationException(msg);
440 	}
441 
442 	// do move operation
443 
444 	String targetUUID = ((NodeState) targetNode.getItemState()).getUUID();
445 	// add target to new parent
446 	destParentNode.createChildNodeLink(destName.getName(), targetUUID);
447 	// remove target from old parent
448 	srcParentNode.removeChildNode(srcName.getName(), srcName.getIndex() == 0 ? 1 : srcName.getIndex());
449 	// change definition of target if necessary
450 	NodeDefImpl oldTargetDef = (NodeDefImpl) targetNode.getDefinition();
451 	NodeDefId oldTargetDefId = new NodeDefId(oldTargetDef.unwrap());
452 	NodeDefId newTargetDefId = new NodeDefId(newTargetDef.unwrap());
453 	if (!oldTargetDefId.equals(newTargetDefId)) {
454 	    targetNode.onRedefine(newTargetDefId);
455 	}
456     }

613     //--------------------------------------------------------< inner classes >
614     class TransientNamespaceMappings implements NamespaceResolver {
615 
616 	// the global persistent namespace registry
617 	private NamespaceRegistryImpl nsReg;
618 
619 	// local prefix/namespace mappings
620 	private HashMap prefixToURI = new HashMap();
621 	private HashMap uriToPrefix = new HashMap();
622 
623 	// prefixes in global namespace registry hidden by local mappings
624 	private Set hiddenPrefixes = new HashSet();
625 
626 	TransientNamespaceMappings(NamespaceRegistryImpl nsReg) {
627 	    this.nsReg = nsReg;
628 	}
629 
630 	void setNamespacePrefix(String prefix, String uri)
631 		throws NamespaceException, RepositoryException {
632 	    if (prefix == null || uri == null) {
633 		throw new IllegalArgumentException("prefix/uri can not be null");
634 	    }
635 	    if (NamespaceRegistryImpl.NS_EMPTY_PREFIX.equals(prefix)
636 		    || NamespaceRegistryImpl.NS_DEFAULT_URI.equals(uri)) {
637 		throw new NamespaceException("default namespace is reserved and can not be changed");
638 	    }
639 	    // check if namespace exists (the following call will
640 	    // trigger a NamespaceException if it doesn't)
641 	    String globalPrefix = nsReg.getPrefix(uri);
642 
643 	    // check new prefix for collision
644 	    String globalURI = null;
645 	    try {
646 		globalURI = nsReg.getURI(prefix);
647 	    } catch (NamespaceException nse) {
648 		// ignore
649 	    }
650 	    if (globalURI != null) {
651 		// prefix is already mapped in global namespace registry;
652 		// check if it refers to a namespace that has been locally
653 		// remapped, thus hiding it
654 		if (!hiddenPrefixes.contains(prefix)) {
655 		    // we don't allow to hide a namespace because we can't
656 		    // guarantee that there are no references to it
657 		    // (in names of nodes/properties/node types etc.)
658 		    throw new NamespaceException(prefix + ": prefix is already mapped to the namespace: " + globalURI);
659 		}
660 	    }
661 
662 	    // check if namespace is already locally mapped
663 	    String oldPrefix = (String) uriToPrefix.get(uri);
664 	    if (oldPrefix != null) {
665 		// resurrect hidden global prefix
666 		hiddenPrefixes.remove(nsReg.getPrefix(uri));
667 		// remove old mapping
668 		uriToPrefix.remove(uri);
669 		prefixToURI.remove(oldPrefix);
670 	    }
671 
672 	    // check if prefix is already locally mapped
673 	    String oldURI = (String) prefixToURI.get(prefix);
674 	    if (oldURI != null) {
675 		// resurrect hidden global prefix
676 		hiddenPrefixes.remove(nsReg.getPrefix(oldURI));
677 		// remove old mapping
678 		uriToPrefix.remove(oldURI);
679 		prefixToURI.remove(prefix);
680 	    }
681 
682 	    if (!prefix.equals(globalPrefix)) {
683 		// store new mapping
684 		prefixToURI.put(prefix, uri);
685 		uriToPrefix.put(uri, prefix);
686 		hiddenPrefixes.add(globalPrefix);
687 	    }
688 	}
689 
690 	String[] getPrefixes() {
691 	    if (prefixToURI.isEmpty()) {
692 		// shortcut
693 		return nsReg.getPrefixes();
694 	    }
695 
696 	    HashSet prefixes = new HashSet();
697 	    // global prefixes
698 	    String[] globalPrefixes = nsReg.getPrefixes();
699 	    for (int i = 0; i < globalPrefixes.length; i++) {
700 		if (!hiddenPrefixes.contains(globalPrefixes[i])) {
