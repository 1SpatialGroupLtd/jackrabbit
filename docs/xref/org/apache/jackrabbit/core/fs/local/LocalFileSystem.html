51      private String osPath(String genericPath) {
52  	if (File.separator.equals(SEPARATOR)) {
53  	    return genericPath;
54  	}
55  	return genericPath.replace(SEPARATOR_CHAR, File.separatorChar);
56      }

72      /***
73       * @see FileSystem#init()
74       */
75      public void init() throws FileSystemException {
76  	if (rootPath == null) {
77  	    String msg = "path not set";
78  	    log.error(msg);
79  	    throw new FileSystemException(msg);
80  	}
81  	root = new File(rootPath);
82  
83  	if (root.exists()) {
84  	    if (!root.isDirectory()) {
85  		String msg = "path does not denote a folder";
86  		log.error(msg);
87  		throw new FileSystemException(msg);
88  	    }
89  	} else {
90  	    if (!root.mkdirs()) {
91  		String msg = "failed to create root";
92  		log.error(msg);
93  		throw new FileSystemException(msg);
94  	    }
95  	}
96      }

279     public String[] listFiles(String folderPath) throws FileSystemException {
280 	File folder = new File(root, osPath(folderPath));
281 	File[] files = folder.listFiles(new FileFilter() {
282 	    public boolean accept(File f) {
283 		return f.isFile();
284 	    }
285 	});
286 	if (files == null) {
287 	    String msg = folderPath + " does not denote a folder";
288 	    log.error(msg);
289 	    throw new FileSystemException(msg);
290 	}
291 	String[] entries = new String[files.length];
292 	for (int i = 0; i < files.length; i++) {
293 	    entries[i] = files[i].getName();
294 	}
295 	return entries;

323     public void move(String srcPath, String destPath) throws FileSystemException {
324 	File src = new File(root, osPath(srcPath));
325 	File dest = new File(root, osPath(destPath));
326 
327 	if (dest.exists()) {
328 	    // we need to move the existing file/folder out of the way first
329 	    try {
330 		FileUtil.delete(dest);
331 	    } catch (IOException ioe) {
332 		String msg = "moving " + src.getPath() + " to " + dest.getPath() + " failed";
333 		log.error(msg, ioe);
334 		throw new FileSystemException(msg, ioe);
335 	    }
336 	}
337 	File destParent = dest.getParentFile();
338 	if (!destParent.exists()) {
339 	    // create destination parent folder first
340 	    if (!destParent.mkdirs()) {
341 		String msg = "moving " + src.getPath() + " to " + dest.getPath() + " failed";
342 		log.error(msg);
343 		throw new FileSystemException(msg);
344 	    }
345 	}
346 
347 	// now we're ready to move/rename the file/folder
348 	if (!src.renameTo(dest)) {
349 	    String msg = "moving " + src.getPath() + " to " + dest.getPath() + " failed";
350 	    log.error(msg);
351 	    throw new FileSystemException(msg);
352 	}
