264     public static void importNode(File file, Node parent) throws Exception {
265 	if (file.isDirectory()) {
266 	    Node newNode = parent.addNode(file.getName(), "nt:folder");
267 	    File[] children = file.listFiles();
268 	    if (children != null) {
269 		for (int i = 0; i < children.length; i++) {
270 		    importNode(children[i], newNode);
271 		}
272 	    }
273 	} else {
274 	    Node newNode = parent.addNode(file.getName(), "nt:file");
275 	    Node content = newNode.addNode("jcr:content", "nt:mimeResource");
276 	    content.setProperty("jcr:data", new FileInputStream(file));
277 	    content.setProperty("jcr:lastModified", Calendar.getInstance());
278 	    content.setProperty("jcr:mimeType", "application/octet-stream");
279 	}
280     }

282     public static void dumpTree(Node start, final PrintStream ps) throws RepositoryException {
283 	start.accept(new TraversingItemVisitor.Default() {
284 	    protected void entering(Node node, int i) throws RepositoryException {
285 		while (i-- > 0) {
286 		    System.out.print('\t');
287 		}
288 		ps.println("[node] " + node.getName());
289 
290 		super.entering(node, i);
291 	    }
292 
293 	    protected void entering(Property property, int i) throws RepositoryException {
294 		while (i-- > 0) {
295 		    System.out.print('\t');
296 		}
297 		StringBuffer sb = new StringBuffer();
298 		if (property.getDefinition().isMultiple()) {
299 		    Value[] values = property.getValues();
300 		    for (int j = 0; j < values.length; j++) {
301 			if (sb.length() > 0) {
302 			    sb.append(", ");
303 			}
304 			Value val = values[j];
305 			if (val.getType() == PropertyType.BINARY) {
306 			    sb.append("[binary]");
307 			} else {
308 			    sb.append(val.getString());
309 			}
310 		    }
311 		} else {
312 		    Value val = property.getValue();
313 		    if (val.getType() == PropertyType.BINARY) {
314 			sb.append("[binary]");
315 		    } else {
316 			sb.append(val.getString());
317 		    }
318 		}
319 		ps.println("[prop] " + property.getName() + " " + sb.toString());
320 
321 		super.entering(property, i);
322 	    }
323 	});
324     }
