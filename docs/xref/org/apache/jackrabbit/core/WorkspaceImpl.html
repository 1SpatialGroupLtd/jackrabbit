145     /***
146      * Disposes this <code>WorkspaceImpl</code> and frees resources.
147      */
148     void dispose() {
149 	try {
150 	    ObservationManager om = getObservationManager();
151 	    EventListenerIterator it = om.getRegisteredEventListeners();
152 	    while (it.hasNext()) {
153 		EventListener l = it.nextEventListener();
154 		log.debug("removing EventListener: " + l);
155 		om.removeEventListener(l);
156 	    }
157 	} catch (RepositoryException e) {
158 	    log.error("Exception while disposing Workspace:", e);
159 	}
160     }

218     protected static PersistentNodeState getNodeState(Path nodePath,
219 						      HierarchyManagerImpl hierMgr,
220 						      PersistentItemStateManager stateMgr)
221 	    throws PathNotFoundException, RepositoryException {
222 	try {
223 	    ItemId id = hierMgr.resolvePath(nodePath);
224 	    if (!id.denotesNode()) {
225 		throw new PathNotFoundException(hierMgr.safeGetJCRPath(nodePath));
226 	    }
227 	    return getNodeState((NodeId) id, stateMgr);
228 	} catch (NoSuchItemStateException nsise) {
229 	    throw new PathNotFoundException(hierMgr.safeGetJCRPath(nodePath));
230 	} catch (ItemStateException ise) {
231 	    String msg = "internal error: failed to retrieve state of " + hierMgr.safeGetJCRPath(nodePath);
232 	    log.error(msg, ise);
233 	    throw new RepositoryException(msg, ise);
234 	}
235     }

250     /***
251      * @param nodePath
252      * @param nodeTypeName
253      * @param ntReg
254      * @param accessMgr
255      * @param hierMgr
256      * @param stateMgr
257      * @throws ConstraintViolationException
258      * @throws AccessDeniedException
259      * @throws PathNotFoundException
260      * @throws ItemExistsException
261      * @throws RepositoryException
262      */
263     protected static void checkAddNode(Path nodePath, QName nodeTypeName,
264 				       NodeTypeRegistry ntReg,
265 				       AccessManagerImpl accessMgr,
266 				       HierarchyManagerImpl hierMgr,
267 				       PersistentItemStateManager stateMgr)
268 	    throws ConstraintViolationException, AccessDeniedException,
269 	    PathNotFoundException, ItemExistsException, RepositoryException {
270 
271 	Path parentPath = nodePath.getAncestor(1);
272 	PersistentNodeState parentState = getNodeState(parentPath, hierMgr, stateMgr);
273 
274 	// 1. check path & access rights
275 
276 	Path.PathElement nodeName = nodePath.getNameElement();
277 	try {
278 	    // check access rights
279 	    if (!accessMgr.isGranted(parentState.getId(), Permission.READ_ITEM)) {
280 		throw new PathNotFoundException(hierMgr.safeGetJCRPath(parentPath));
281 	    }
282 	    if (!accessMgr.isGranted(parentState.getId(), Permission.ADD_NODE)) {
283 		throw new AccessDeniedException(hierMgr.safeGetJCRPath(parentPath) + ": not allowed to add child node");
284 	    }
285 	} catch (ItemNotFoundException infe) {
286 	    String msg = "internal error: failed to check access rights for " + hierMgr.safeGetJCRPath(parentPath);
287 	    log.error(msg, infe);
288 	    throw new RepositoryException(msg, infe);
289 	}
290 
291 	// 2. check node type constraints
292 
293 	ChildNodeDef parentDef = ntReg.getNodeDef(parentState.getDefinitionId());
294 	if (parentDef.isProtected()) {
295 	    throw new ConstraintViolationException(hierMgr.safeGetJCRPath(parentPath) + ": cannot add child node to protected parent node");
296 	}
297 	EffectiveNodeType entParent = getEffectiveNodeType(parentState, ntReg);
298 	entParent.checkAddNodeConstraints(nodeName.getName(), nodeTypeName);
299 	ChildNodeDef newNodeDef = findApplicableDefinition(nodeName.getName(), nodeTypeName, parentState, ntReg);
300 
301 	// 3. check for name collisions
302 
303 	if (parentState.hasPropertyEntry(nodeName.getName())) {
304 	    // there's already a property with that name
305 	    throw new ItemExistsException(hierMgr.safeGetJCRPath(nodePath));
306 	}
307 	if (parentState.hasChildNodeEntry(nodeName.getName())) {
308 	    // there's already a node with that name...
309 
310 	    // get definition of existing conflicting node
311 	    NodeState.ChildNodeEntry entry = parentState.getChildNodeEntry(nodeName.getName(), 1);
312 	    NodeState conflictingState;
313 	    NodeId conflictingId = new NodeId(entry.getUUID());
314 	    try {
315 		conflictingState = (NodeState) stateMgr.getItemState(conflictingId);
316 	    } catch (ItemStateException ise) {
317 		String msg = "internal error: failed to retrieve state of " + hierMgr.safeGetJCRPath(conflictingId);
318 		log.error(msg, ise);
319 		throw new RepositoryException(msg, ise);
320 	    }
321 	    ChildNodeDef conflictingTargetDef = ntReg.getNodeDef(conflictingState.getDefinitionId());
322 	    // check same-name sibling setting of both target and existing node
323 	    if (!conflictingTargetDef.allowSameNameSibs()
324 		    || !newNodeDef.allowSameNameSibs()) {
325 		throw new ItemExistsException(hierMgr.safeGetJCRPath(nodePath));
326 	    }
327 	}
328     }

330     /***
331      * @param nodePath
332      * @param ntReg
333      * @param accessMgr
334      * @param hierMgr
335      * @param stateMgr
336      * @throws ConstraintViolationException
337      * @throws AccessDeniedException
338      * @throws PathNotFoundException
339      * @throws RepositoryException
340      */
341     protected static void checkRemoveNode(Path nodePath,
342 					  NodeTypeRegistry ntReg,
343 					  AccessManagerImpl accessMgr,
344 					  HierarchyManagerImpl hierMgr,
345 					  PersistentItemStateManager stateMgr)
346 	    throws ConstraintViolationException, AccessDeniedException,
347 	    PathNotFoundException, RepositoryException {
348 
349 	PersistentNodeState targetState = getNodeState(nodePath, hierMgr, stateMgr);
350 	Path parentPath = nodePath.getAncestor(1);
351 	PersistentNodeState parentState = getNodeState(parentPath, hierMgr, stateMgr);
352 
353 	// 1. check path & access rights
354 
355 	try {
356 	    // check access rights
357 	    if (!accessMgr.isGranted(targetState.getId(), Permission.READ_ITEM)) {
358 		throw new PathNotFoundException(hierMgr.safeGetJCRPath(nodePath));
359 	    }
360 	    if (!accessMgr.isGranted(parentState.getId(), Permission.REMOVE_ITEM)) {
361 		throw new AccessDeniedException(hierMgr.safeGetJCRPath(parentPath) + ": not allowed to remove child node");
362 	    }
363 	} catch (ItemNotFoundException infe) {
364 	    String msg = "internal error: failed to check access rights for " + hierMgr.safeGetJCRPath(nodePath);
365 	    log.error(msg, infe);
366 	    throw new RepositoryException(msg, infe);
367 	}
368 
369 	// 2. check node type constraints
370 
371 	ChildNodeDef parentDef = ntReg.getNodeDef(parentState.getDefinitionId());
372 	if (parentDef.isProtected()) {
373 	    throw new ConstraintViolationException(hierMgr.safeGetJCRPath(parentPath) + ": cannot remove child node of protected parent node");
374 	}
375 	ChildNodeDef targetDef = ntReg.getNodeDef(targetState.getDefinitionId());
376 	if (targetDef.isMandatory()) {
377 	    throw new ConstraintViolationException(hierMgr.safeGetJCRPath(nodePath) + ": cannot remove mandatory node");
378 	}
379 	if (targetDef.isProtected()) {
380 	    throw new ConstraintViolationException(hierMgr.safeGetJCRPath(nodePath) + ": cannot remove protected node");
381 	}
382     }

395 							    NodeTypeRegistry ntReg)
396 	    throws RepositoryException {
397 	// build effective node type of mixins & primary type:
398 	// existing mixin's

402 	try {
403 	    return ntReg.buildEffectiveNodeType((QName[]) set.toArray(new QName[set.size()]));
404 	} catch (NodeTypeConflictException ntce) {
405 	    String msg = "internal error: failed to build effective node type for node " + state.getUUID();
406 	    log.error(msg, ntce);
407 	    throw new RepositoryException(msg, ntce);
408 	}
409     }

434     private static PersistentNodeState copyNodeState(NodeState srcState,
435 						     String parentUUID,
436 						     NodeTypeRegistry ntReg,
437 						     HierarchyManagerImpl srcHierMgr,
438 						     PersistentItemStateManager srcStateMgr,
439 						     PersistentItemStateManager destStateMgr,
440 						     boolean clone)
441 	    throws RepositoryException {
442 	PersistentNodeState newState;
443 	try {
444 	    String uuid;
445 	    if (clone) {
446 		uuid = srcState.getUUID();
447 	    } else {
448 		uuid = UUID.randomUUID().toString();	// create new version 4 uuid
449 	    }
450 	    newState = destStateMgr.createNodeState(uuid, srcState.getNodeTypeName(), parentUUID);
451 	    // copy node state
452 	    // @todo special handling required for nodes with special semantics (e.g. those defined by mix:versionable, et.al.)
453 	    // FIXME delegate to 'node type instance handler'
454 	    newState.setMixinTypeNames(srcState.getMixinTypeNames());
455 	    newState.setDefinitionId(srcState.getDefinitionId());
456 	    // copy child nodes
457 	    Iterator iter = srcState.getChildNodeEntries().iterator();
458 	    while (iter.hasNext()) {
459 		NodeState.ChildNodeEntry entry = (NodeState.ChildNodeEntry) iter.next();
460 		NodeState srcChildState = (NodeState) srcStateMgr.getItemState(new NodeId(entry.getUUID()));
461 		// recursive copying of child node
462 		PersistentNodeState newChildState = copyNodeState(srcChildState, uuid,
463 			ntReg, srcHierMgr, srcStateMgr, destStateMgr, clone);
464 		// persist new child node
465 		newChildState.store();
466 		// add new child node entry to new node
467 		newState.addChildNodeEntry(entry.getName(), newChildState.getUUID());
468 	    }
469 	    // copy properties
470 	    iter = srcState.getPropertyEntries().iterator();
471 	    while (iter.hasNext()) {
472 		NodeState.PropertyEntry entry = (NodeState.PropertyEntry) iter.next();
473 		PropertyState srcChildState = (PropertyState) srcStateMgr.getItemState(new PropertyId(srcState.getUUID(), entry.getName()));
474 		PersistentPropertyState newChildState = copyPropertyState(srcChildState, uuid, entry.getName(),
475 			ntReg, srcHierMgr, srcStateMgr, destStateMgr);
476 		// persist new property
477 		newChildState.store();
478 		// add new property entry to new node
479 		newState.addPropertyEntry(entry.getName());
480 	    }
481 	    return newState;
482 	} catch (ItemStateException ise) {
483 	    String msg = "internal error: failed to copy state of " + srcHierMgr.safeGetJCRPath(srcState.getId());
484 	    log.error(msg, ise);
485 	    throw new RepositoryException(msg, ise);
486 	}
487     }

489     private static PersistentPropertyState copyPropertyState(PropertyState srcState,
490 							     String parentUUID,
491 							     QName propName,
492 							     NodeTypeRegistry ntReg,
493 							     HierarchyManagerImpl srcHierMgr,
494 							     PersistentItemStateManager srcStateMgr,
495 							     PersistentItemStateManager destStateMgr)
496 	    throws RepositoryException {
497 	// @todo special handling required for properties with special semantics (e.g. those defined by mix:versionable, mix:lockable, et.al.)
498 	PersistentPropertyState newState;
499 	try {
500 	    newState = destStateMgr.createPropertyState(parentUUID, propName);
501 	    PropDefId defId = srcState.getDefinitionId();
502 	    newState.setDefinitionId(defId);
503 	    newState.setType(srcState.getType());
504 	    InternalValue[] values = srcState.getValues();
505 	    if (values != null) {
506 		InternalValue[] newValues = new InternalValue[values.length];
507 		for (int i = 0; i < values.length; i++) {
508 		    newValues[i] = values[i] != null ? values[i].createCopy() : null;
509 		}
510 		newState.setValues(values);
511 		// FIXME delegate to 'node type instance handler'
512 		if (defId != null) {
513 		    PropDef def = ntReg.getPropDef(defId);
514 		    if (def.getDeclaringNodeType().equals(NodeTypeRegistry.MIX_REFERENCEABLE)) {
515 			if (propName.equals(ItemImpl.PROPNAME_UUID)) {
516 			    // set correct value of jcr:uuid property
517 			    newState.setValues(new InternalValue[]{InternalValue.create(parentUUID)});
518 			}
519 		    }
520 		}
521 	    }
522 	    return newState;
523 	} catch (ItemStateException ise) {
524 	    String msg = "internal error: failed to copy state of " + srcHierMgr.safeGetJCRPath(srcState.getId());
525 	    log.error(msg, ise);
526 	    throw new RepositoryException(msg, ise);
527 	}
528     }

530     private static void internalCopy(String srcAbsPath,
531 				     PersistentItemStateManager srcStateMgr,
532 				     HierarchyManagerImpl srcHierMgr,
533 				     String destAbsPath,
534 				     PersistentItemStateManager destStateMgr,
535 				     HierarchyManagerImpl destHierMgr,
536 				     AccessManagerImpl accessMgr,
537 				     NamespaceResolver nsResolver,
538 				     NodeTypeRegistry ntReg,
539 				     boolean clone)
540 	    throws ConstraintViolationException, AccessDeniedException,
541 	    PathNotFoundException, ItemExistsException, RepositoryException {
542 
543 	// 1. check paths & retrieve state
544 
545 	Path srcPath;
546 	PersistentNodeState srcState;
547 	try {
548 	    srcPath = Path.create(srcAbsPath, nsResolver, true);
549 	    srcState = getNodeState(srcPath, srcHierMgr, srcStateMgr);
550 	} catch (MalformedPathException mpe) {
551 	    String msg = "invalid path: " + srcAbsPath;
552 	    log.error(msg, mpe);
553 	    throw new RepositoryException(msg, mpe);
554 	}
555 
556 	Path destPath;
557 	Path.PathElement destName;
558 	Path destParentPath;
559 	PersistentNodeState destParentState;
560 	try {
561 	    destPath = Path.create(destAbsPath, nsResolver, true);
562 	    destName = destPath.getNameElement();
563 	    destParentPath = destPath.getAncestor(1);
564 	    destParentState = getNodeState(destParentPath, destHierMgr, destStateMgr);
565 	} catch (MalformedPathException mpe) {
566 	    String msg = "invalid path: " + destAbsPath;
567 	    log.error(msg, mpe);
568 	    throw new RepositoryException(msg, mpe);
569 	}
570 	int ind = destName.getIndex();
571 	if (ind > 0) {
572 	    // subscript in name element
573 	    String msg = destAbsPath + ": invalid destination path (subscript in name element is not allowed)";
574 	    log.error(msg);
575 	    throw new RepositoryException(msg);
576 	}
577 
578 	// 2. check access rights & node type constraints
579 
580 	try {
581 	    // check read access right on source node
582 	    if (!accessMgr.isGranted(srcState.getId(), Permission.READ_ITEM)) {
583 		throw new PathNotFoundException(srcAbsPath);
584 	    }
585 	} catch (ItemNotFoundException infe) {
586 	    String msg = "internal error: failed to check access rights for " + srcAbsPath;
587 	    log.error(msg, infe);
588 	    throw new RepositoryException(msg, infe);
589 	}
590 	// check node type constraints
591 	checkAddNode(destPath, srcState.getNodeTypeName(), ntReg, accessMgr, destHierMgr, destStateMgr);
592 /*
593 	// check if target node needs to be inserted at specific location in child node entries list
594 	boolean insertTargetEntry = false;
595 	int ind = destName.getIndex();
596 	if (ind > 0) {
597 	    // target name contains subscript:
598 	    // validate subscript
599 	    List sameNameSibs = destParentState.getChildNodeEntries(destName.getName());
600 	    if (ind > sameNameSibs.size() + 1) {
601 		String msg = "invalid subscript in name: " + destAbsPath;
602 		log.error(msg);
603 		throw new RepositoryException(msg);
604 	    }
605 	    insertTargetEntry = (ind < sameNameSibs.size() + 1) ? true : false;
606 	}
607 	if (insertTargetEntry) {
608 	    // check hasOrderableChildNodes flag
609 	    if (!ntReg.getNodeTypeDef(destParentState.getNodeTypeName()).hasOrderableChildNodes()) {
610 		throw new ConstraintViolationException(destAbsPath + ": parent node's node type does not allow explicit ordering of child nodes");
611 	    }
612 	}
613 */
614 	// 3. do copy operation (modify and persist affected states)
