139     /***
140      * Disposes this <code>WorkspaceImpl</code> and frees resources.
141      */
142     void dispose() {
143 	try {
144 	    ObservationManager om = getObservationManager();
145 	    EventListenerIterator it = om.getRegisteredEventListeners();
146 	    while (it.hasNext()) {
147 		EventListener l = it.nextEventListener();
148 		log.debug("removing EventListener: " + l);
149 		om.removeEventListener(l);
150 	    }
151 	} catch (RepositoryException e) {
152 	    log.error("Exception while disposing Workspace:", e);
153 	}
154     }

209      * @throws PathNotFoundException
210      * @throws RepositoryException
211      */
212     protected static PersistentNodeState getNodeState(Path nodePath,
213 						      HierarchyManagerImpl hierMgr,
214 						      PersistentItemStateManager stateMgr)
215 	    throws PathNotFoundException, RepositoryException {
216 	try {
217 	    ItemId id = hierMgr.resolvePath(nodePath);
218 	    if (!id.denotesNode()) {
219 		throw new PathNotFoundException(hierMgr.safeGetJCRPath(nodePath));
220 	    }
221 	    return getNodeState((NodeId) id, stateMgr);
222 	} catch (NoSuchItemStateException nsise) {
223 	    throw new PathNotFoundException(hierMgr.safeGetJCRPath(nodePath));
224 	} catch (ItemStateException ise) {
225 	    String msg = "internal error: failed to retrieve state of " + hierMgr.safeGetJCRPath(nodePath);
226 	    log.error(msg, ise);
227 	    throw new RepositoryException(msg, ise);
228 	}
229     }

244     /***
245      * @param nodePath
246      * @param nodeTypeName
247      * @param ntReg
248      * @param accessMgr
249      * @param hierMgr
250      * @param stateMgr
251      * @throws ConstraintViolationException
252      * @throws AccessDeniedException
253      * @throws PathNotFoundException
254      * @throws ItemExistsException
255      * @throws RepositoryException
256      */
257     protected static void checkAddNode(Path nodePath, QName nodeTypeName,
258 				       NodeTypeRegistry ntReg,
259 				       AccessManagerImpl accessMgr,
260 				       HierarchyManagerImpl hierMgr,
261 				       PersistentItemStateManager stateMgr)
262 	    throws ConstraintViolationException, AccessDeniedException,
263 	    PathNotFoundException, ItemExistsException, RepositoryException {
264 
265 	Path parentPath = nodePath.getAncestor(1);
266 	PersistentNodeState parentState = getNodeState(parentPath, hierMgr, stateMgr);
267 
268 	// 1. check path & access rights
269 
270 	Path.PathElement nodeName = nodePath.getNameElement();
271 	try {
272 	    // check access rights
273 	    if (!accessMgr.isGranted(parentState.getId(), Permission.READ_ITEM)) {
274 		throw new PathNotFoundException(hierMgr.safeGetJCRPath(parentPath));
275 	    }
276 	    if (!accessMgr.isGranted(parentState.getId(), Permission.ADD_NODE)) {
277 		throw new AccessDeniedException(hierMgr.safeGetJCRPath(parentPath) + ": not allowed to add child node");
278 	    }
279 	} catch (ItemNotFoundException infe) {
280 	    String msg = "internal error: failed to check access rights for " + hierMgr.safeGetJCRPath(parentPath);
281 	    log.error(msg, infe);
282 	    throw new RepositoryException(msg, infe);
283 	}
284 
285 	// 2. check node type constraints
286 
287 	ChildNodeDef parentDef = ntReg.getNodeDef(parentState.getDefinitionId());
288 	if (parentDef.isProtected()) {
289 	    throw new ConstraintViolationException(hierMgr.safeGetJCRPath(parentPath) + ": cannot add child node to protected parent node");
290 	}
291 	EffectiveNodeType entParent = getEffectiveNodeType(parentState, ntReg);
292 	entParent.checkAddNodeConstraints(nodeName.getName(), nodeTypeName);
293 	ChildNodeDef newNodeDef = findApplicableDefinition(nodeName.getName(), nodeTypeName, parentState, ntReg);
294 
295 	// 3. check for name collisions
296 
297 	if (parentState.hasPropertyEntry(nodeName.getName())) {
298 	    // there's already a property with that name
299 	    throw new ItemExistsException(hierMgr.safeGetJCRPath(nodePath));
300 	}
301 	if (parentState.hasChildNodeEntry(nodeName.getName())) {
302 	    // there's already a node with that name...
303 
304 	    // get definition of existing conflicting node
305 	    NodeState.ChildNodeEntry entry = parentState.getChildNodeEntry(nodeName.getName(), 1);
306 	    NodeState conflictingState;
307 	    NodeId conflictingId = new NodeId(entry.getUUID());
308 	    try {
309 		conflictingState = (NodeState) stateMgr.getItemState(conflictingId);
310 	    } catch (ItemStateException ise) {
311 		String msg = "internal error: failed to retrieve state of " + hierMgr.safeGetJCRPath(conflictingId);
312 		log.error(msg, ise);
313 		throw new RepositoryException(msg, ise);
314 	    }
315 	    ChildNodeDef conflictingTargetDef = ntReg.getNodeDef(conflictingState.getDefinitionId());
316 	    // check same-name sibling setting of both target and existing node
317 	    if (!conflictingTargetDef.allowSameNameSibs()
318 		    || !newNodeDef.allowSameNameSibs()) {
319 		throw new ItemExistsException(hierMgr.safeGetJCRPath(nodePath));
320 	    }
321 	}
322     }

324     /***
325      * @param nodePath
326      * @param ntReg
327      * @param accessMgr
328      * @param hierMgr
329      * @param stateMgr
330      * @throws ConstraintViolationException
331      * @throws AccessDeniedException
332      * @throws PathNotFoundException
333      * @throws RepositoryException
334      */
335     protected static void checkRemoveNode(Path nodePath,
336 					  NodeTypeRegistry ntReg,
337 					  AccessManagerImpl accessMgr,
338 					  HierarchyManagerImpl hierMgr,
339 					  PersistentItemStateManager stateMgr)
340 	    throws ConstraintViolationException, AccessDeniedException,
341 	    PathNotFoundException, RepositoryException {
342 
343 	PersistentNodeState targetState = getNodeState(nodePath, hierMgr, stateMgr);
344 	Path parentPath = nodePath.getAncestor(1);
345 	PersistentNodeState parentState = getNodeState(parentPath, hierMgr, stateMgr);
346 
347 	// 1. check path & access rights
348 
349 	try {
350 	    // check access rights
351 	    if (!accessMgr.isGranted(targetState.getId(), Permission.READ_ITEM)) {
352 		throw new PathNotFoundException(hierMgr.safeGetJCRPath(nodePath));
353 	    }
354 	    if (!accessMgr.isGranted(parentState.getId(), Permission.REMOVE_ITEM)) {
355 		throw new AccessDeniedException(hierMgr.safeGetJCRPath(parentPath) + ": not allowed to remove child node");
356 	    }
357 	} catch (ItemNotFoundException infe) {
358 	    String msg = "internal error: failed to check access rights for " + hierMgr.safeGetJCRPath(nodePath);
359 	    log.error(msg, infe);
360 	    throw new RepositoryException(msg, infe);
361 	}
362 
363 	// 2. check node type constraints
364 
365 	ChildNodeDef parentDef = ntReg.getNodeDef(parentState.getDefinitionId());
366 	if (parentDef.isProtected()) {
367 	    throw new ConstraintViolationException(hierMgr.safeGetJCRPath(parentPath) + ": cannot remove child node of protected parent node");
368 	}
369 	ChildNodeDef targetDef = ntReg.getNodeDef(targetState.getDefinitionId());
370 	if (targetDef.isMandatory()) {
371 	    throw new ConstraintViolationException(hierMgr.safeGetJCRPath(nodePath) + ": cannot remove mandatory node");
372 	}
373 	if (targetDef.isProtected()) {
374 	    throw new ConstraintViolationException(hierMgr.safeGetJCRPath(nodePath) + ": cannot remove protected node");
375 	}
376     }

388     protected static EffectiveNodeType getEffectiveNodeType(NodeState state,
389 							    NodeTypeRegistry ntReg)
390 	    throws RepositoryException {
391 	// build effective node type of mixins & primary type:
392 	// existing mixin's
393 	HashSet set = new HashSet(((NodeState) state).getMixinTypeNames());
394 	// primary type
395 	set.add(state.getNodeTypeName());
396 	try {
397 	    return ntReg.buildEffectiveNodeType((QName[]) set.toArray(new QName[set.size()]));
398 	} catch (NodeTypeConflictException ntce) {
399 	    String msg = "internal error: failed to build effective node type for node " + state.getUUID();
400 	    log.error(msg, ntce);
401 	    throw new RepositoryException(msg, ntce);
402 	}
403     }

428     private static PersistentNodeState copyNodeState(NodeState srcState,
429 						     String parentUUID,
430 						     NodeTypeRegistry ntReg,
431 						     HierarchyManagerImpl srcHierMgr,
432 						     PersistentItemStateManager srcStateMgr,
433 						     PersistentItemStateManager destStateMgr,
434 						     boolean clone)
435 	    throws RepositoryException {
436 	PersistentNodeState newState;
437 	try {
438 	    String uuid;
439 	    if (clone) {
440 		uuid = srcState.getUUID();
441 	    } else {
442 		uuid = UUID.randomUUID().toString();	// create new version 4 uuid
443 	    }
444 	    newState = destStateMgr.createNodeState(uuid, srcState.getNodeTypeName(), parentUUID);
445 	    // copy node state
446 	    // @todo special handling required for nodes with special semantics (e.g. those defined by mix:versionable, et.al.)
447 	    // FIXME delegate to 'node type instance handler'
448 	    newState.setMixinTypeNames(srcState.getMixinTypeNames());
449 	    newState.setDefinitionId(srcState.getDefinitionId());
450 	    // copy child nodes
451 	    Iterator iter = srcState.getChildNodeEntries().iterator();
452 	    while (iter.hasNext()) {
453 		NodeState.ChildNodeEntry entry = (NodeState.ChildNodeEntry) iter.next();
454 		NodeState srcChildState = (NodeState) srcStateMgr.getItemState(new NodeId(entry.getUUID()));
455 		// recursive copying of child node
456 		PersistentNodeState newChildState = copyNodeState(srcChildState, uuid,
457 			ntReg, srcHierMgr, srcStateMgr, destStateMgr, clone);
458 		// persist new child node
459 		newChildState.store();
460 		// add new child node entry to new node
461 		newState.addChildNodeEntry(entry.getName(), newChildState.getUUID());
462 	    }
463 	    // copy properties
464 	    iter = srcState.getPropertyEntries().iterator();
465 	    while (iter.hasNext()) {
466 		NodeState.PropertyEntry entry = (NodeState.PropertyEntry) iter.next();
467 		PropertyState srcChildState = (PropertyState) srcStateMgr.getItemState(new PropertyId(srcState.getUUID(), entry.getName()));
468 		PersistentPropertyState newChildState = copyPropertyState(srcChildState, uuid, entry.getName(),
469 			ntReg, srcHierMgr, srcStateMgr, destStateMgr);
470 		// persist new property
471 		newChildState.store();
472 		// add new property entry to new node
473 		newState.addPropertyEntry(entry.getName());
474 	    }
475 	    return newState;
476 	} catch (ItemStateException ise) {
477 	    String msg = "internal error: failed to copy state of " + srcHierMgr.safeGetJCRPath(srcState.getId());
478 	    log.error(msg, ise);
479 	    throw new RepositoryException(msg, ise);
480 	}
481     }

483     private static PersistentPropertyState copyPropertyState(PropertyState srcState,
484 							     String parentUUID,
485 							     QName propName,
486 							     NodeTypeRegistry ntReg,
487 							     HierarchyManagerImpl srcHierMgr,
488 							     PersistentItemStateManager srcStateMgr,
489 							     PersistentItemStateManager destStateMgr)
490 	    throws RepositoryException {
491 	// @todo special handling required for properties with special semantics (e.g. those defined by mix:versionable, mix:lockable, et.al.)
492 	PersistentPropertyState newState;
493 	try {
494 	    newState = destStateMgr.createPropertyState(parentUUID, propName);
495 	    PropDefId defId = srcState.getDefinitionId();
496 	    newState.setDefinitionId(defId);
497 	    newState.setType(srcState.getType());
498 	    InternalValue[] values = srcState.getValues();
499 	    if (values != null) {
500 		InternalValue[] newValues = new InternalValue[values.length];
501 		for (int i = 0; i < values.length; i++) {
502 		    newValues[i] = values[i] != null ? values[i].createCopy() : null;
503 		}
504 		newState.setValues(values);
505 		// FIXME delegate to 'node type instance handler'
506 		if (defId != null) {
507 		    PropDef def = ntReg.getPropDef(defId);
508 		    if (def.getDeclaringNodeType().equals(NodeTypeRegistry.MIX_REFERENCEABLE)) {
509 			if (propName.equals(ItemImpl.PROPNAME_UUID)) {
510 			    // set correct value of jcr:uuid property
511 			    newState.setValues(new InternalValue[]{InternalValue.create(parentUUID)});
512 			}
513 		    }
514 		}
515 	    }
516 	    return newState;
517 	} catch (ItemStateException ise) {
518 	    String msg = "internal error: failed to copy state of " + srcHierMgr.safeGetJCRPath(srcState.getId());
519 	    log.error(msg, ise);
520 	    throw new RepositoryException(msg, ise);
521 	}
522     }

524     private static void internalCopy(String srcAbsPath,
525 				     PersistentItemStateManager srcStateMgr,
526 				     HierarchyManagerImpl srcHierMgr,
527 				     String destAbsPath,
528 				     PersistentItemStateManager destStateMgr,
529 				     HierarchyManagerImpl destHierMgr,
530 				     AccessManagerImpl accessMgr,
531 				     NamespaceResolver nsResolver,
532 				     NodeTypeRegistry ntReg,
533 				     boolean clone)
534 	    throws ConstraintViolationException, AccessDeniedException,
535 	    PathNotFoundException, ItemExistsException, RepositoryException {
536 
537 	// 1. check paths & retrieve state
538 
539 	Path srcPath;
540 	PersistentNodeState srcState;
541 	try {
542 	    srcPath = Path.create(srcAbsPath, nsResolver, true);
543 	    srcState = getNodeState(srcPath, srcHierMgr, srcStateMgr);
544 	} catch (MalformedPathException mpe) {
545 	    String msg = "invalid path: " + srcAbsPath;
546 	    log.error(msg, mpe);
547 	    throw new RepositoryException(msg, mpe);
548 	}
549 
550 	Path destPath;
551 	Path.PathElement destName;
552 	Path destParentPath;
553 	PersistentNodeState destParentState;
554 	try {
555 	    destPath = Path.create(destAbsPath, nsResolver, true);
556 	    destName = destPath.getNameElement();
557 	    destParentPath = destPath.getAncestor(1);
558 	    destParentState = getNodeState(destParentPath, destHierMgr, destStateMgr);
559 	} catch (MalformedPathException mpe) {
560 	    String msg = "invalid path: " + destAbsPath;
561 	    log.error(msg, mpe);
562 	    throw new RepositoryException(msg, mpe);
563 	}
564 	int ind = destName.getIndex();
565 	if (ind > 0) {
566 	    // subscript in name element
567 	    String msg = destAbsPath + ": invalid destination path (subscript in name element is not allowed)";
568 	    log.error(msg);
569 	    throw new RepositoryException(msg);
570 	}
571 
572 	// 2. check access rights & node type constraints
573 
574 	try {
575 	    // check read access right on source node
576 	    if (!accessMgr.isGranted(srcState.getId(), Permission.READ_ITEM)) {
577 		throw new PathNotFoundException(srcAbsPath);
578 	    }
579 	} catch (ItemNotFoundException infe) {
580 	    String msg = "internal error: failed to check access rights for " + srcAbsPath;
581 	    log.error(msg, infe);
582 	    throw new RepositoryException(msg, infe);
583 	}
584 	// check node type constraints
585 	checkAddNode(destPath, srcState.getNodeTypeName(), ntReg, accessMgr, destHierMgr, destStateMgr);
586 /*
587 	// check if target node needs to be inserted at specific location in child node entries list
588 	boolean insertTargetEntry = false;
589 	int ind = destName.getIndex();
590 	if (ind > 0) {
591 	    // target name contains subscript:
592 	    // validate subscript
593 	    List sameNameSibs = destParentState.getChildNodeEntries(destName.getName());
594 	    if (ind > sameNameSibs.size() + 1) {
595 		String msg = "invalid subscript in name: " + destAbsPath;
596 		log.error(msg);
597 		throw new RepositoryException(msg);
598 	    }
599 	    insertTargetEntry = (ind < sameNameSibs.size() + 1) ? true : false;
600 	}
601 	if (insertTargetEntry) {
602 	    // check hasOrderableChildNodes flag
603 	    if (!ntReg.getNodeTypeDef(destParentState.getNodeTypeName()).hasOrderableChildNodes()) {
604 		throw new ConstraintViolationException(destAbsPath + ": parent node's node type does not allow explicit ordering of child nodes");
605 	    }
606 	}
607 */
608 	// 3. do copy operation (modify and persist affected states)
609 
610 	// create deep copy of source node state
611 	PersistentNodeState newState = copyNodeState(srcState, destParentState.getUUID(),
612 		ntReg, srcHierMgr, srcStateMgr, destStateMgr, clone);
