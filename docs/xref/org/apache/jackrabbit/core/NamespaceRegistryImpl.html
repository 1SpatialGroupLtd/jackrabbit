114     private void load() throws RepositoryException {
115 	FileSystemResource propFile = new FileSystemResource(nsRegStore, NS_REG_RESOURCE);
116 	try {
117 	    if (!propFile.exists()) {
118 		// clear existing mappings
119 		prefixToURI.clear();
120 		uriToPrefix.clear();
121 
122 		// default namespace (if no prefix is specified)
123 		prefixToURI.put(NS_EMPTY_PREFIX, NS_DEFAULT_URI);
124 		uriToPrefix.put(NS_DEFAULT_URI, NS_EMPTY_PREFIX);
125 		// declare the predefined mappings
126 		// jcr:
127 		prefixToURI.put(NS_JCR_PREFIX, NS_JCR_URI);
128 		uriToPrefix.put(NS_JCR_URI, NS_JCR_PREFIX);
129 		// nt:
130 		prefixToURI.put(NS_NT_PREFIX, NS_NT_URI);
131 		uriToPrefix.put(NS_NT_URI, NS_NT_PREFIX);
132 		// mix:
133 		prefixToURI.put(NS_MIX_PREFIX, NS_MIX_URI);
134 		uriToPrefix.put(NS_MIX_URI, NS_MIX_PREFIX);
135 		// pt:
136 		prefixToURI.put(NS_PT_PREFIX, NS_PT_URI);
137 		uriToPrefix.put(NS_PT_URI, NS_PT_PREFIX);
138 		// sv:
139 		prefixToURI.put(NS_SV_PREFIX, NS_SV_URI);
140 		uriToPrefix.put(NS_SV_URI, NS_SV_PREFIX);
141 
142 		// persist mappings
143 		store();
144 		return;
145 	    }
146 
147 	    InputStream in = propFile.getInputStream();
148 	    try {
149 		Properties props = new Properties();
150 		props.load(in);
151 
152 		// clear existing mappings
153 		prefixToURI.clear();
154 		uriToPrefix.clear();
155 
156 		// read mappings from properties
157 		Iterator iter = props.keySet().iterator();
158 		while (iter.hasNext()) {
159 		    String prefix = (String) iter.next();
160 		    String uri = props.getProperty(prefix);
161 
162 		    prefixToURI.put(prefix, uri);
163 		    uriToPrefix.put(uri, prefix);
164 		}
165 	    } finally {
166 		in.close();
167 	    }
168 	} catch (Exception e) {
169 	    String msg = "failed to load namespace registry";
170 	    log.error(msg, e);
171 	    throw new RepositoryException(msg, e);
172 	}
173     }
174 
175     private void store() throws RepositoryException {
176 	FileSystemResource propFile = new FileSystemResource(nsRegStore, NS_REG_RESOURCE);
177 	try {
178 	    propFile.makeParentDirs();
179 	    OutputStream os = propFile.getOutputStream();
180 	    Properties props = new Properties();
181 
182 	    // store mappings in properties
183 	    Iterator iter = prefixToURI.keySet().iterator();
184 	    while (iter.hasNext()) {
185 		String prefix = (String) iter.next();
186 		String uri = (String) prefixToURI.get(prefix);
187 		props.setProperty(prefix, uri);
188 	    }
189 
190 	    try {
191 		props.store(os, null);
192 	    } finally {
193 		// make sure stream is closed
194 		os.close();
195 	    }
196 	} catch (Exception e) {
197 	    String msg = "failed to persist namespace registry";
198 	    log.error(msg, e);
199 	    throw new RepositoryException(msg, e);
200 	}
201     }

204     /***
205      * @see NamespaceRegistry#registerNamespace
206      */
207     public void registerNamespace(String prefix, String uri)
208 	    throws NamespaceException, RepositoryException {
209 	if (prefix == null || uri == null) {
210 	    throw new IllegalArgumentException("prefix/uri can not be null");
211 	}
212 	if (NS_EMPTY_PREFIX.equals(prefix) || NS_DEFAULT_URI.equals(uri)) {
213 	    throw new NamespaceException("default namespace is reserved and can not be changed");
214 	}
215 	if (reservedURIs.contains(uri)) {
216 	    throw new NamespaceException("failed to register namespace " + prefix + " -> " + uri + ": reserved URI");
217 	}
218 	if (reservedPrefixes.contains(prefix)) {
219 	    throw new NamespaceException("failed to register namespace " + prefix + " -> " + uri + ": reserved prefix");
220 	}
221 
222 	String oldPrefix = (String) uriToPrefix.get(uri);
223 	if (oldPrefix != null) {
224 	    // existing namespace
225 	    if (oldPrefix.equals(prefix)) {
226 		throw new NamespaceException("failed to register namespace " + prefix + " -> " + uri + ": mapping already exists");
227 	    }
228 	    // remove old prefix
229 	    prefixToURI.remove(oldPrefix);
230 	    uriToPrefix.remove(uri);
231 	}
232 
233 	if (prefixToURI.containsKey(prefix)) {
234 	    // prevent remapping of existing prefixes because this would in effect
235 	    // remove the previously assigned namespace;
236 	    // as we can't guarantee that there are no references to this namespace
237 	    // (in names of nodes/properties/node types etc.) we simply don't allow it.
238 	    throw new NamespaceException("failed to register namespace " + prefix + " -> " + uri + ": remapping existing prefixes is not supported.");
239 	}
240 
241 	prefixToURI.put(prefix, uri);
242 	uriToPrefix.put(uri, prefix);
243 
244 	// persist mappings
245 	store();
246     }
247 
248     /***
249      * @see NamespaceRegistry#unregisterNamespace
250      */
251     public void unregisterNamespace(String prefix)
252 	    throws NamespaceException, RepositoryException {
253 	if (reservedPrefixes.contains(prefix)) {
254 	    throw new NamespaceException("reserved prefix: " + prefix);
255 	}
256 	if (!prefixToURI.containsKey(prefix)) {
257 	    throw new NamespaceException("unknown prefix: " + prefix);
258 	}
259 	// as we can't guarantee that there are no references to the specified
260 	// namespace (in names of nodes/properties/node types etc.) we simply
261 	// don't allow it.
262 	throw new NamespaceException("unregistering namespaces is not supported.");
263     }
