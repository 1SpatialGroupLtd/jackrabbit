66      protected synchronized ItemState getOrCreateTransientItemState()
67  	    throws RepositoryException {
68  	if (!isTransient()) {
69  	    // make transient (copy-on-write)
70  	    try {
71  		PropertyState transientState =
72  			itemStateMgr.createTransientPropertyState((PropertyState) state, ItemState.STATUS_EXISTING_MODIFIED);
73  		// remove listener on persistent state
74  		state.removeListener(this);
75  		// add listener on transient state
76  		transientState.addListener(this);
77  		// swap persistent with transient state
78  		state = transientState;
79  	    } catch (ItemStateException ise) {
80  		String msg = "failed to create transient state";
81  		log.error(msg, ise);
82  		throw new RepositoryException(msg, ise);
83  	    }
84  	}
85  	return state;
86      }
87  
88      protected void makePersistent() throws RepositoryException {
89  	if (!isTransient()) {
90  	    String msg = "there's no transient state to persist";
91  	    log.error(msg);
92  	    throw new RepositoryException(msg);
93  	}
94  
95  	try {
96  	    PropertyState transientState = (PropertyState) state;
97  	    PersistentPropertyState persistentState = (PersistentPropertyState) transientState.getOverlayedState();
98  	    if (persistentState == null) {
99  		// this property is 'new'
100 		persistentState = itemStateMgr.createPersistentPropertyState(transientState.getParentUUID(), transientState.getName());
101 	    }
102 	    // copy state from transient state
103 	    persistentState.setDefinitionId(transientState.getDefinitionId());
104 	    persistentState.setType(transientState.getType());
105 	    persistentState.setValues(transientState.getValues());
106 	    // make state persistent
107 	    persistentState.store();
108 	    // remove listener from transient state
109 	    transientState.removeListener(this);
110 	    // add listener to persistent state
111 	    persistentState.addListener(this);
112 	    // swap transient state with persistent state
113 	    state = persistentState;
114 	    // reset status
115 	    status = STATUS_NORMAL;
116 	} catch (ItemStateException ise) {
117 	    String msg = "failed to persist transient state of " + safeGetJCRPath();
118 	    log.error(msg, ise);
119 	    throw new RepositoryException(msg, ise);
120 	}
121     }

123     /***
124      * @param values
125      * @param type
126      * @throws ConstraintViolationException
127      * @throws RepositoryException
128      */
129     protected void internalSetValue(InternalValue[] values, int type)
130 	    throws ConstraintViolationException, RepositoryException {
131 	// check state of this instance
132 	checkItemState();
133 
134 	// check for null value
135 	if (values == null) {
136 	    // setting a property to null removes it automatically
137 	    ((NodeImpl) getParent()).removeChildProperty(((PropertyId) id).getName());
138 	    return;
139 	}
140 	ArrayList list = new ArrayList();
141 	// compact array (purge null entries)
142 	if (values != null) {
143 	    for (int i = 0; i < values.length; i++) {
144 		if (values[i] != null) {
145 		    list.add(values[i]);
146 		}
147 	    }
148 	}
149 	values = (InternalValue[]) list.toArray(new InternalValue[list.size()]);
150 
151 	// modify the state of this property
152 	PropertyState thisState = (PropertyState) getOrCreateTransientItemState();
153 
154 	// free old values as necessary
155 	InternalValue[] oldValues = thisState.getValues();
156 	if (oldValues != null) {
157 	    for (int i = 0; i < oldValues.length; i++) {
158 		InternalValue old = oldValues[i];
159 		if (old != null) {
160 		    switch (old.getType()) {
161 			case PropertyType.BINARY:
162 			    // BINARY value
163 			    BLOBFileValue blob = (BLOBFileValue) old.internalValue();
164 			    blob.delete();
165 			    break;
166 		    }
167 		}
168 	    }
169 	}
170 
171 	// set new values
172 	thisState.setValues(values);
173 	// set type
174 	if (type == PropertyType.UNDEFINED) {
175 	    // fallback to default type
176 	    type = PropertyType.STRING;
177 	}
178 	thisState.setType(type);
179     }

181     /***
182      * Same as <code>{@link Property#setValue(String)}</code> except that
183      * this method takes a <code>QName</code> instead of a <code>String</code>
184      * value.
185      *
186      * @param name
187      * @throws ValueFormatException
188      * @throws RepositoryException
189      */
190     public void setValue(QName name) throws ValueFormatException, RepositoryException {
191 	// check state of this instance
192 	checkItemState();
193 
194 	// check if versioning allows write
195 	if (!((NodeImpl) getParent()).safeIsCheckedOut()) {
196 	    throw new VersionException("Cannot alter the value of a property of a checked-in node " + safeGetJCRPath());
197 	}
198 
199 	// check protected flag
200 	if (definition.isProtected()) {
201 	    throw new ConstraintViolationException("cannot set the value of a protected property " + safeGetJCRPath());
202 	}
203 
204 	// check multi-value flag
205 	if (definition.isMultiple()) {
206 	    throw new ValueFormatException(safeGetJCRPath() + " is multi-valued and can therefore only be set to an array of values");
207 	}
208 
209 	// check type according to definition of this property
210 	int reqType = definition.getRequiredType();
211 	if (reqType == PropertyType.UNDEFINED) {
212 	    reqType = PropertyType.NAME;
213 	}
214 
215 	if (name == null) {
216 	    internalSetValue(null, reqType);
217 	    return;
218 	}
219 
220 	InternalValue internalValue;
221 	if (reqType != PropertyType.NAME) {
222 	    // type conversion required
223 	    internalValue =
224 		    InternalValue.create(InternalValue.create(name).toJCRValue(session.getNamespaceResolver()),
225 			    reqType, session.getNamespaceResolver());
226 	} else {
227 	    // no type conversion required
228 	    internalValue = InternalValue.create(name);
229 	}
230 
231 	internalSetValue(new InternalValue[]{internalValue}, reqType);
232     }

235      * Same as <code>{@link Property#setValue(String[])}</code> except that

242      */

244 	// check state of this instance
245 	checkItemState();
246 
247 	// check if versioning allows write
248 	if (!((NodeImpl) getParent()).safeIsCheckedOut()) {
249 	    throw new VersionException("Cannot alter the value of a property of a checked-in node " + safeGetJCRPath());
250 	}
251 
252 	// check protected flag
253 	if (definition.isProtected()) {
254 	    throw new ConstraintViolationException("cannot set the value of a protected property " + safeGetJCRPath());
255 	}
256 
257 	// check multi-value flag
258 	if (!definition.isMultiple()) {
259 	    throw new ValueFormatException(safeGetJCRPath() + " is not multi-valued");
260 	}
261 
262 	// check type according to definition of this property
263 	int reqType = definition.getRequiredType();
264 	if (reqType == PropertyType.UNDEFINED) {
265 	    reqType = PropertyType.NAME;
266 	}
267 
268 	InternalValue[] internalValues = null;
269 	// convert to internal values of correct type
270 	if (names != null) {
271 	    internalValues = new InternalValue[names.length];
272 	    for (int i = 0; i < names.length; i++) {
273 		QName name = names[i];
274 		InternalValue internalValue = null;
275 		if (name != null) {
276 		    if (reqType != PropertyType.NAME) {
277 			// type conversion required
278 			internalValue =
279 				InternalValue.create(InternalValue.create(name).toJCRValue(session.getNamespaceResolver()),
280 					reqType, session.getNamespaceResolver());
281 		    } else {
282 			// no type conversion required
283 			internalValue = InternalValue.create(name);
284 		    }
285 		}
286 		internalValues[i] = internalValue;
287 	    }
288 	}
289 
290 	internalSetValue(internalValues, reqType);
291     }
